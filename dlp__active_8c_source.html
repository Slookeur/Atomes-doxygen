51struct field_molecule * get_active_field_molecule_from_model_id (struct project * this_proj, int aid)
52{
53 int i;
54 struct field_molecule * fmol = this_proj -> force_field[activef] -> first_molecule;
55 struct field_atom * fat;
56 while (fmol)
57 {
58 fat = fmol -> first_atom;
59 while (fat)
60 {
61 for (i=0; i<fat -> num; i++)
62 {
63 if (fat -> list[i] == aid) return fmol;
64 }
65 fat = fat -> next;
66 }
67 fmol = fmol -> next;
68 }
69 return NULL;
70}

279{
280 struct field_prop * prop = NULL;

282 gboolean done;
283 int i;

293 }
