51G_MODULE_EXPORT void update_restart_parameter (GtkEntry * res, gpointer data)
52{
53 int i, j;
54 i = GPOINTER_TO_INT(data);
55 const gchar * m = entry_get_text (res);
56 double v = atof(m);
57 j = (int)v;
58 if (j != tmp_cpmd -> restart[i])
59 {
60 tmp_cpmd -> restart[i] = j;
61 }
62 update_entry_int (res, tmp_cpmd -> restart[i]);
63 for (i=1; i<4; i++) print_the_section (i, 0, qmbuffer[i]);
64}
