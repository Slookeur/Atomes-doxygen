51void set_quality (int q, glwin * view)

117GtkWidget * menu_quality (glwin * view, int id)
118{
119 GtkWidget * menuq = gtk_menu_new ();
120 GtkWidget * fixed = gtk_fixed_new ();
121 if (id == 0)
122 {
123 view -> ogl_quality = create_hscale (2.0, 1000.0, 1.0, view -> anim -> last -> img -> quality, GTK_POS_RIGHT, 0, 100, NULL, NULL, NULL);
124 gtk_fixed_put (GTK_FIXED(fixed), view -> ogl_quality, 0, 0);
125 }
126 else
127 {
128 GtkWidget * qscale = create_hscale (2.0, 1000.0, 1.0, view -> anim -> last -> img -> quality, GTK_POS_RIGHT, 0, 100, NULL, NULL, NULL);
129 gtk_fixed_put (GTK_FIXED(fixed), qscale, 0, 0);
130 }
131 GtkWidget * qs = create_menu_item_from_widget (fixed, FALSE, FALSE, FALSE);
132 g_signal_connect (G_OBJECT (qs), "activate", G_CALLBACK(window_quality), view);
133 gtk_menu_shell_append ((GtkMenuShell *)menuq, qs);
134 return menuq;
135}
