160GtkWidget * menu_edit (glwin * view, int id)
161{
162 int i, j;
163 struct project * this_proj = get_project_by_id(view -> proj);
164 GtkWidget * menu = gtk_menu_new ();
165 GtkWidget * widg;
166 if (id == 0)
167 {

170 gtk_menu_shell_append ((GtkMenuShell *)menu, view -> cbuilder);

172 else
173 {
174 widg = create_menu_item (FALSE, "Crystal Builder");
175 g_signal_connect (G_OBJECT (widg), "activate", G_CALLBACK(crystal_window), & view -> colorp[0][0]);
176 gtk_menu_shell_append ((GtkMenuShell *)menu, widg);
177 }
178 j = (this_proj -> cell.ltype && this_proj -> steps == 1) ? 1 : 0;
179 if (id == 0)
180 {
181 view -> ogl_box[1] = menu_item_new_with_submenu ("Cell", (this_proj -> natomes) ? this_proj -> cell.ltype : 0, menu_cell_edit(view, id, j));
182 gtk_menu_shell_append ((GtkMenuShell *)menu, view -> ogl_box[1]);
183 }
184 else
185 {
186 gtk_menu_shell_append ((GtkMenuShell *)menu, menu_item_new_with_submenu ("Cell", (this_proj -> natomes) ? this_proj -> cell.ltype : 0, menu_cell_edit(view, id, j)));
187 }
188
189 GtkWidget * ats = create_menu_item (FALSE, "Atoms");
190 gtk_menu_shell_append ((GtkMenuShell *)menu, ats);
191 GtkWidget * menua = gtk_menu_new ();
192 gtk_menu_item_set_submenu ((GtkMenuItem *)ats, menua);
193 j = (this_proj -> steps == 1) ? 1 : 0;
194 for (i=0; i<5; i++)
195 {
196 widg = create_menu_item (TRUE, action_name[i]);
197 g_signal_connect (G_OBJECT (widg), "activate", G_CALLBACK(action_window), & view -> colorp[i][0]);
198 gtk_menu_shell_append ((GtkMenuShell *)menua, widg);
199 widget_set_sensitive (widg, this_proj -> nspec);
200 widget_set_sensitive (widg, (i == 3) ? j : (this_proj -> natomes) ? j : 0);
201 }
202
203 if (id == 0)
204 {
205 gchar * rtext[2] = {"Extract/Rebuild on Motion", "Extract/Rebuild on Copy"};
206 for (i=0; i<2; i++)
207 {
208 view -> rbuild[i] = gtk3_menu_item (menu, rtext[i], IMG_STOCK, (gpointer)ECUT, G_CALLBACK(turn_rebuild), & view -> colorp[i][0], FALSE, 0, 0, TRUE, FALSE, view -> rebuild[i][0]);
209 widget_set_sensitive (view -> rbuild[i], (this_proj -> steps == 1) ? 1 : 0);
210 }
211 }
212 return menu;

224GMenu * menu_cell_edit (glwin * view, int popm, int sensitive)
225{
226 GMenu * menu = g_menu_new ();
227 gboolean sens;
228 GCallback edit_handler[3] = {G_CALLBACK(wrapping_coord), G_CALLBACK(super_cell), G_CALLBACK(edition_win)};
229 int i, j, k, l;
230 gchar * act;
231 for (i=0; i<6; i++)
232 {
233 j = (i == 0) ? i : (i == 3) ? 1 : 2;
234 if (i > 2) sens = sensitive;
235 if (i == 0) sens = ! view -> wrapped;
236 if (i == 3)
237 {
238 k = 0;
239 for (l=0; l<3; l++) k += view -> anim -> last -> img-> extra_cell[l];
240 sens = k;
241 }
242 act = g_strdup_printf ("ceed-%d", i);
243 if (i == 0 || i == 3)
244 {
245 append_opengl_item (view, menu, edit_names[i], act, popm, i, NULL, IMG_NONE, NULL, FALSE, edit_handler[j], (gpointer)view, FALSE, FALSE, FALSE, sens);
246 }
247 else
248 {
249 k = (i < 3) ? i-1 : i-2;
250 append_opengl_item (view, menu, edit_names[i], act, popm, i, NULL, IMG_NONE, NULL, FALSE, edit_handler[j], & view -> colorp[k][0], FALSE, FALSE, FALSE, sens);
251 }
252 g_free (act);
253 }
254 return menu;

291G_MODULE_EXPORT void to_run_rebuild (GSimpleAction * action, GVariant * parameter, gpointer data)
292{
293 tint * dat =(tint *)data;
294 glwin * view = get_project_by_id(dat -> a) -> modelgl;
295 gboolean doit = TRUE;
296 GVariant * state = g_action_get_state (G_ACTION (action));
297 const gchar * rebuild = g_action_get_name ((GAction *)action);
298 int lgt = strlen (rebuild);
299 gchar * name = g_strdup_printf ("%c%c", rebuild[lgt-2], rebuild[lgt-1]);
300 if (g_strcmp0(name, ".1") == 0)
301 {
302 g_free (name);
303 name = g_strdup_printf ("%.*s.0", lgt-2, rebuild);
304 g_action_group_activate_action ((GActionGroup *)view -> action_group, (const gchar *)name, NULL);
305 g_free (name);
306 doit = FALSE;
307 }
308 if (doit)
309 {
310 turn_rebuild (NULL, NULL, data);
311 g_action_change_state (G_ACTION (action), g_variant_new_boolean (! g_variant_get_boolean (state)));
312 g_variant_unref (state);
313 }
314}
