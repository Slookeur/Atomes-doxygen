145void init_map_range (colormap * map, int pts)
146{
147 map -> points = pts;
148 if (map -> positions != NULL)
149 {
150 g_free (map -> positions);
151 map -> positions = NULL;
152 }
153 if (map -> values != NULL)
154 {
155 g_free (map -> values);
156 map -> values = NULL;
157 }
158 map -> positions = allocfloat (pts);
159 map -> values = g_malloc (pts*sizeof*map -> values);
160 int i;
161 for (i=0; i<pts; i++)
162 {
163 map -> positions[i] = map -> cmax - i*(map -> cmax - map -> cmin)/(pts - 1.0);
164 map -> values[i].red = 1.0 - i / (pts - 1.0);
165 map -> values[i].green = 0.0;
166 map -> values[i].blue = i / (pts - 1.0);
167 map -> values[i].alpha = 1.0;
168 }
169}

202gboolean setup_custom_color_map (float * data, struct project * this_proj, gboolean init)
203{
204 int i, j, k;
205 gboolean action;
206 float cmin, cmax;
207 float delta;
208
209 if (data != NULL)
210 {
211 k = 0;
212 cmin = cmax = data[0];
213 for (i=0; i<this_proj -> steps; i++)
214 {
215 for (j=0; j<this_proj -> natomes; j++)
216 {
217 cmin = min(cmin, data[k]);
218 cmax = max(cmax, data[k]);
219 k ++;
220 }
221 }
222 }
223
224 if (init)
225 {
226 gchar * str = g_strdup_printf ("Range: [min-max] = %f - %f", cmin, cmax);
227 action = ask_yes_no ("Use this data set ?", str, GTK_MESSAGE_QUESTION, NULL);
228 }
229 else
230 {
231 action = TRUE;
232 }
233 if (action)
234 {
235 if (init)
236 {
237 this_proj -> modelgl -> custom_map = allocate_color_map (2, this_proj);
238 this_proj -> modelgl -> custom_map -> points = 2;
239 }
240 the_map = this_proj -> modelgl -> custom_map;
241 if (data != NULL)
242 {
243 the_map -> cmin = cmin;
244 the_map -> cmax = cmax;
245 init_map_range (the_map, the_map -> points);
246 k = 0;
247 for (i=0; i<this_proj -> steps; i++)
248 {
249 for (j=0; j<this_proj -> natomes; j++)
250 {
251 the_map -> data[i][j] = data[k];
252 k ++;
253 }
254 }
255 }
256 gboolean done;
257 for (i=0; i<this_proj -> steps; i++)
258 {
259 for (j=0; j<this_proj -> natomes; j++)
260 {
261 done = FALSE;
262 for (k=0; k<the_map -> points-1; k++)
263 {
264 if (the_map -> data[i][j] <= the_map -> positions[k] && the_map -> data[i][j] > the_map -> positions[k+1])
265 {
266 delta = (the_map -> data[i][j]-the_map -> positions[k])/(the_map -> positions[k] - the_map -> positions[k+1]);
267 the_map -> colors[i][j].red = the_map -> values[k].red + delta * (the_map -> values[k].red - the_map -> values[k+1].red);
268 the_map -> colors[i][j].green = the_map -> values[k].green + delta * (the_map -> values[k].green - the_map -> values[k+1].green);
269 the_map -> colors[i][j].blue = the_map -> values[k].blue + delta * (the_map -> values[k].blue - the_map -> values[k+1].blue);
270 the_map -> colors[i][j].alpha = 1.0;
271 done = TRUE;
272 }
273 }
274 if (! done)
275 {
276 if (the_map -> data[i][j] > the_map -> positions[0])
277 {
278 the_map -> colors[i][j].red = the_map -> values[0].red;
279 the_map -> colors[i][j].green = the_map -> values[0].green;
280 the_map -> colors[i][j].blue = the_map -> values[0].blue;
281 the_map -> colors[i][j].alpha = 1.0;
282 }
283 else if (the_map -> data[i][j] < the_map -> positions[the_map -> points-1])
284 {
285 the_map -> colors[i][j].red = the_map -> values[the_map -> points-1].red;
286 the_map -> colors[i][j].green = the_map -> values[the_map -> points-1].green;
287 the_map -> colors[i][j].blue = the_map -> values[the_map -> points-1].blue;
288 the_map -> colors[i][j].alpha = 1.0;
289 }
290 }
291 }
292 }
293 }
294 return action;
295}

305G_MODULE_EXPORT void set_point_position (GtkSpinButton * res, gpointer data)
306{
307 int pts = GPOINTER_TO_INT (data);
308 float pos = (float) gtk_spin_button_get_value (res);
309 gboolean update = FALSE;
310 if (pos >= tmp_map -> cmin && pos <= tmp_map -> cmax)
311 {
312 if (pos != tmp_map -> positions[pts])
313 {
314 if (pts == 0)
315 {
316 update = (pos > tmp_map -> positions[1]) ? TRUE : FALSE;
317 }
318 else if (pts == tmp_map -> points-1)
319 {
320 update = (pos < tmp_map -> positions[pts-1]) ? TRUE : FALSE;
321 }
322 else
323 {
324 update = (pos < tmp_map -> positions[pts-1] && pos > tmp_map -> positions[pts+1]) ? TRUE : FALSE;
325 }
326 if (update) tmp_map -> positions[pts] = pos;
327 }
328 }
329 gtk_spin_button_set_value (GTK_SPIN_BUTTON(res), tmp_map -> positions[pts]);
330 destroy_this_widget (color_bar);
331 color_bar = create_css_color_bar (tmp_map);

587void fill_map_model (GtkTreeStore * store, struct project * this_proj)
588{
589 GtkTreeIter step_level, atom_level;
590 int h, i, j;
591 if (this_proj -> steps > 1)
592 {
593 for (h=0; h < this_proj -> steps; h++)
594 {
595 gtk_tree_store_append (store, & step_level, NULL);
596 gtk_tree_store_set (store, & step_level, 0, h+1,
597 1, 0,
598 2, 0,
599 3, 0, -1);
600 for (i=0; i < this_proj -> natomes; i++)
601 {
602 gtk_tree_store_append (store, & atom_level, & step_level);
603 j = this_proj -> atoms[h][i].sp;
604 gtk_tree_store_set (store, & atom_level, 0, -h-1, 1, this_proj -> chemistry -> label[j] , 2, i+1, 3, the_map -> data[h][i], -1);
605 }
606 }
607 }
608 else
609 {
610 for (i=0; i < this_proj -> natomes; i++)
611 {
612 gtk_tree_store_append (store, & atom_level, NULL);
613 j = this_proj -> atoms[0][i].sp;
614 gtk_tree_store_set (store, & atom_level, 0, this_proj -> chemistry -> label[j] , 1, i+1, 2, the_map -> data[0][i], -1);
615 }
616 }
