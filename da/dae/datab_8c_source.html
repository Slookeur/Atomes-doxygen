106void get_tree_data (GtkWidget * tree)
107{
108 sel = gtk_tree_view_get_selection (GTK_TREE_VIEW(tree));
109 curve_model = gtk_tree_view_get_model(GTK_TREE_VIEW(tree));
110 lrows = gtk_tree_selection_get_selected_rows (sel, & curve_model);
111 nrows = gtk_tree_selection_count_selected_rows (sel);
112}
113
122void save_row (gpointer data, gpointer user_data)
123{
124 path = data;
125 tint * id = (tint *)user_data;
126 if (gtk_tree_model_get_iter (curve_model, & row, path))
127 {
128
129 gtk_tree_model_get (curve_model, & row, 1, & get_project_by_id(id -> a) -> curves[id -> b][id -> c] -> data[0][nrows],
130 2, & get_project_by_id(id -> a) -> curves[id -> b][id -> c] -> data[1][nrows], -1);
131 }
132 nrows = nrows + 1;
133}

143void update_first_row (gpointer data, gpointer user_data)
144{
145 path = data;
146 if (gtk_tree_model_get_iter (curve_model, & row, path))
147 {
148 nrows = nrows + 1;
149 gtk_list_store_set (GTK_LIST_STORE(curve_model), & row, 0, nrows, -1);
150 }
151}

159void update_first_col ()
160{
161 gtk_tree_selection_select_all (sel);
162 lrows = gtk_tree_selection_get_selected_rows (sel, & curve_model);
163 nrows = 0;
164 g_list_foreach (lrows, (GFunc)update_first_row, NULL);
165 gtk_tree_selection_unselect_all (sel);
166}

176void add_to_last_row (gpointer data, gpointer user_data)
177{
178 path = data;
179 double vold;
180 double * cte = (double *)user_data;
181 if (gtk_tree_model_get_iter (curve_model, & row, path))
182 {
183 gtk_tree_model_get (curve_model, & row, 2, & vold, -1);
184 gtk_list_store_set (GTK_LIST_STORE(curve_model), & row, 2, * cte + vold, -1);
185 }
186}

196void add_to_last_col (double cte, gpointer data)
197{
198 qint * dat = (qint *)data;
199 struct project * this_proj = get_project_by_id(dat -> a);
200 curve_model = gtk_tree_view_get_model (GTK_TREE_VIEW(this_proj -> curves[dat -> b][dat -> c] -> datatree));
201 sel = gtk_tree_view_get_selection (GTK_TREE_VIEW(this_proj -> curves[dat -> b][dat -> c] -> datatree));
202 gtk_tree_selection_select_all (sel);
203 get_tree_data (this_proj -> curves[dat -> b][dat -> c] -> datatree);
204 g_list_foreach (lrows, (GFunc)add_to_last_row, & cte);
205 gtk_tree_selection_unselect_all (sel);
206}

273void copy_row (gpointer data, gpointer user_data)
274{
275 path = data;
276 int v1;
277 double v2, v3;
278 gchar * tmp;
279
280 if (gtk_tree_model_get_iter (curve_model, & row, path))
281 {
282 gtk_tree_model_get (curve_model, & row, 0, & v1, 1, & v2, 2, & v3, -1);
283 if (text == NULL)
284 {
285 text = g_strdup_printf ("%d\t%f\t%f\n", v1, v2, v3);
286 }
287 else
288 {
289 tmp = g_strdup_printf ("%s%d\t%f\t%f\n", text, v1, v2, v3);
290 g_free (text);
291 text = g_strdup_printf ("%s", tmp);
292 g_free (tmp);
293 }
294 }
295}

310 gtk_clipboard_store (gtk_clipboard_get(GDK_SELECTION_CLIPBOARD));
311#endif
312 g_free (text);
313}

323void add_row (gpointer data, gpointer user_data)
324{
325 GtkTreeIter newrow;
326 if (gtk_tree_model_get_iter (curve_model, & row, path))
327 {
328 if (GPOINTER_TO_INT(user_data) == 0)
329 {
330 gtk_list_store_insert_before (GTK_LIST_STORE(curve_model), & newrow, & row);
331 }
332 else
333 {
334 gtk_list_store_insert_after (GTK_LIST_STORE(curve_model), & newrow, & row);
335 }
336 gtk_list_store_set (GTK_LIST_STORE(curve_model), & newrow, 0, 0, 1, 0.00000, 2, 0.00000, -1);
337 }
338}

352 {
353 gtk_list_store_remove (GTK_LIST_STORE(curve_model), & row);

364void insert_cell (gpointer data)
365{
366 if (GPOINTER_TO_INT(data) == 0)
367 {
368 path = (GtkTreePath *) g_list_nth_data (lrows, 0);
369 }
370 else
371 {
372 path = (GtkTreePath *) g_list_nth_data (lrows, nrows-1);
373 }
374 g_list_foreach (lrows, (GFunc)add_row, data);
375 update_first_col ();
376}

385void delete_cell (gpointer data)

647{

664 }

666 {
667 add_to_column (data);
668 }
669 else if (g_strcmp0 (name, "mul") == 0)
670 {
671 multiply_column (data);
672 }
673}

846static void fill_data_model (GtkListStore * store, struct project * this_proj, int b, int c)
847{
848 GtkTreeIter datalevel;
849 int i;
850 for (i=0; i<this_proj -> curves[b][c] -> ndata; i++)
851 {
852 gtk_list_store_append (store, & datalevel);
853 gtk_list_store_set (store, & datalevel,
854 0, i+1,
855 1, this_proj -> curves[b][c] -> data[0][i],
856 2, this_proj -> curves[b][c] -> data[1][i], -1);
857 }
858}
