100void camera_has_changed (gdouble value, gpointer data)
101{
102 tint * cid = (tint *)data;
103 project * this_proj = get_project_by_id(cid -> a);
104 double v;
105 switch (cid -> b)
106 {
107 case 0:
108 this_proj -> modelgl -> anim -> last -> img -> zoom = 2.0*(1.0-value);
109 // gtk_spin_button_set_increments ((GtkSpinButton *)this_proj -> modelgl -> camera_widg[0], this_proj -> modelgl -> zoom_factor, this_proj -> modelgl -> zoom_factor);
110 break;
111 case 1:
112 // > camera depth
113 if (value > this_proj -> modelgl -> anim -> last -> img -> gnear)
114 {
115 this_proj -> modelgl -> anim -> last -> img -> p_depth = value;
116 }
117 else
118 {
119 this_proj -> modelgl -> anim -> last -> img -> p_depth = this_proj -> modelgl -> anim -> last -> img -> gnear + 0.01;
120 gtk_spin_button_set_value ((GtkSpinButton *)this_proj -> modelgl -> camera_widg[1], this_proj -> modelgl -> anim -> last -> img -> p_depth);
121 }
122 break;
123 case 2:
124 // < perspective depth
125 if (value < this_proj -> modelgl -> anim -> last -> img -> p_depth)
126 {
127 this_proj -> modelgl -> anim -> last -> img -> gnear = value;
128 }
129 else
130 {
131 this_proj -> modelgl -> anim -> last -> img -> gnear = this_proj -> modelgl -> anim -> last -> img -> p_depth - 0.01;
132 gtk_spin_button_set_value ((GtkSpinButton *)this_proj -> modelgl -> camera_widg[2], this_proj -> modelgl -> anim -> last -> img -> gnear);
133 }
134 break;
135 default:
136 if (cid -> b < 5)
137 {
138 if (value != this_proj -> modelgl -> anim -> last -> img -> c_angle[cid -> b - 3])
139 {
140 v = this_proj -> modelgl -> anim -> last -> img -> c_angle[cid -> b - 3] - value;
141 save_rotation_quaternion (this_proj -> modelgl);
142 if (cid -> b == 3)
143 {
144 rotate_x_y (this_proj -> modelgl, v, 0.0);
145 }
146 else
147 {
148 rotate_x_y (this_proj -> modelgl, 0.0, v);
149 }
150 }
151 }
152 else
153 {
154 this_proj -> modelgl -> anim -> last -> img -> c_shift[cid -> b - 5] = (value == 0.0) ? value : - value;
155 }
156 break;
157 }
158 update_labels (this_proj -> modelgl);
159 update (this_proj -> modelgl);
160}

290G_MODULE_EXPORT void representation_advanced (GtkWidget * widg, gpointer data)
291{
292 glwin * view = (glwin *)data;
293 gchar * cam_opts[7]={"Zoom:", "Perspective depth:", "Camera depth:",
294 "Camera pitch:", "Camera heading:",
295 "Camera right/left", "Camera up/down"};
296 gchar * str;
297 double smax[7] = {1.0, 100.0, 100.0, 180.0, 180.0, 100.0, 100.0};
298 double smin[7] = {-2.0, 0.0, 0.0, -180.0, -180.0, -100.0, -100.0};
299 double sdel[7] = {0.001, 0.01, 0.01, 0.1, 0.1, 0.01, 0.01};
300 int sdig[7] = {3, 2, 2, 1, 1, 2, 2};
301 int i;
302 double v;
303 str = g_strdup_printf ("OpenGL camera set-up - %s", get_project_by_id(view -> proj)->name);
304 GtkWidget * arep = create_win (str, view -> win, FALSE, FALSE);
305 g_free (str);
306 GtkWidget * vbox = create_vbox (5);
307 add_container_child (CONTAINER_WIN, arep, vbox);
308 GtkWidget * box;
309 for (i=0; i<7; i++)
310 {
311 box = abox (vbox, cam_opts[i], 0);
312 switch (i)
313 {
314 case 0:
315 v = 1.0-0.5*view -> anim -> last -> img -> zoom;
316 break;
317 case 1:
318 v = view -> anim -> last -> img -> p_depth;
319 break;
320 case 2:
321 v = view -> anim -> last -> img -> gnear;
322 break;
323 default:
324 if (i < 5)
325 {
326 v = view -> anim -> last -> img -> c_angle[i-3];
327 }
328 else
329 {
330 v = (view -> anim -> last -> img -> c_shift[i-5] == 0.0) ? 0.0 : - view -> anim -> last -> img -> c_shift[i-5];
331 }
332 break;
333 }
334 if (view -> camera_widg[i]) view -> camera_widg[i] = destroy_this_widget(view -> camera_widg[i]);
335 view -> camera_widg[i] = spin_button (G_CALLBACK(set_camera_spin), v, smin[i], smax[i], sdel[i], sdig[i], 150, & view -> colorp[i][0]);
336 add_box_child_start (GTK_ORIENTATION_HORIZONTAL, box, view -> camera_widg[i], FALSE, FALSE, 10);
337 if (i > 2 || i == 0)
338 {
339 str = g_strdup_printf ("in [%.1f, %.1f]", smin[i], smax[i]);
340 }
341 else if (i == 1)
342 {
343 str = g_strdup_printf ("in [C. depth, %.1f]", smax[i]);
344 }
345 else
346 {
347 str = g_strdup_printf ("in [%.1f, P. depth]", smin[i]);
348 }
349 add_box_child_start (GTK_ORIENTATION_HORIZONTAL, box, markup_label(str, 25, -1, 0.0, 0.5), FALSE, FALSE, 5);
350 g_free (str);
351 if(i < 3 && view -> anim -> last -> img -> rep == ORTHOGRAPHIC) widget_set_sensitive (view -> camera_widg[i], 0);
352 }
353 box = create_hbox(0);
354 add_box_child_start (GTK_ORIENTATION_VERTICAL, vbox, box, FALSE, FALSE, 10);
355 add_box_child_start (GTK_ORIENTATION_HORIZONTAL, box, create_button("Reset view", IMG_NONE, NULL, 100, 25, GTK_RELIEF_NORMAL, G_CALLBACK(reset_view), view), FALSE, FALSE, 200);
356 add_gtk_close_event (arep, G_CALLBACK(on_rep_delete), view);
357 show_the_widgets (arep);
358}

457G_MODULE_EXPORT void change_rep_radio (GSimpleAction * action, GVariant * parameter, gpointer data)
458{
459 glwin * view = (glwin *)data;
460 const gchar * rep = g_variant_get_string (parameter, NULL);
461 int lgt = strlen (rep);
462 gchar * name = g_strdup_printf ("%c%c", rep[lgt-2], rep[lgt-1]);
463 if (g_strcmp0(name, ".1") == 0)
464 {
465 g_free (name);
466 name = g_strdup_printf ("%.*s.0", lgt-2, rep);
467 g_action_group_activate_action ((GActionGroup *)view -> action_group, "set-rep", g_variant_new_string((const gchar *)name));
468 g_free (name);
469 }
470 else
471 {
472 const gchar * rep = g_variant_get_string (parameter, NULL);
473 gchar * rep_name = NULL;
474 int i;
475 for (i=0; i<OGL_REPS; i++)
476 {
477 rep_name = g_strdup_printf ("set-rep.%d.0", i);
478 if (g_strcmp0(rep, (const gchar *)rep_name) == 0)
479 {
480 set_rep (NULL, & view -> colorp[i][0]);
481 g_free (rep_name);
482 rep_name = NULL;
483 break;
484 }
485 g_free (rep_name);
486 rep_name = NULL;
487 }
488 g_action_change_state (G_ACTION (action), parameter);
489 }
