58G_MODULE_EXPORT void run_window_color (GtkDialog * win, gint response_id, gpointer data)
59{
60 struct project * this_proj = (struct project *)data;
61
62 if (response_id == GTK_RESPONSE_OK)
63 {
64 ColRGBA colo = get_window_color (GTK_WIDGET(win));
65 if (wc_cid == -2)
66 {
67 this_proj -> modelgl -> anim -> last -> img -> backcolor = colo;
68 this_proj -> modelgl -> create_shaders[MEASU] = TRUE;
69 }
70 else if (wc_cid == -1)
71 {
72 this_proj -> modelgl -> anim -> last -> img -> box_color = colo;
73 this_proj -> modelgl -> create_shaders[MDBOX] = TRUE;
74 }
75 else

79 re_create_md_shaders (2, shaders, this_proj);
80 int shader[1] = {POLYS};
81 if (this_proj -> modelgl -> anim -> last -> img -> color_map[1] == 0) re_create_md_shaders (1, shader, this_proj);
82 }
83 update (this_proj -> modelgl);
84 }
85 destroy_this_dialog (win);
86}

97void window_color (struct project * this_proj, glwin * view, int wc_cid)
98{
99 gchar * str;
100 GdkRGBA col;
101 if (wc_cid == -2)
102 {
103 str = g_strdup_printf ("Background color");
104 col = colrgba_togtkrgba (view -> anim -> last -> img -> backcolor);
105 }
106 else if (wc_cid == -1)
107 {
108 str = g_strdup_printf ("Model box color");
109 col = colrgba_togtkrgba (view -> anim -> last -> img -> box_color);
110 }
111 else
112 {
113 if (wc_cid < this_proj -> nspec)
114 {
115 str = g_strdup_printf ("%s - atom(s) color", this_proj -> chemistry -> label[wc_cid]);
116 }
117 else
118 {
119 str = g_strdup_printf ("%s* - clone(s) color", this_proj -> chemistry -> label[wc_cid-this_proj -> nspec]);
120 }
121 col = colrgba_togtkrgba (view -> anim -> last -> img -> at_color[wc_cid]);
122 }
123 GtkWidget * win = gtk_color_chooser_dialog_new (str, GTK_WINDOW(view -> win));
124 gtk_window_set_modal (GTK_WINDOW(win), TRUE);
125 gtk_color_chooser_set_use_alpha (GTK_COLOR_CHOOSER(win), TRUE);
126 gtk_color_chooser_set_rgba (GTK_COLOR_CHOOSER(win), & col);
127 g_free (str);
128 run_this_gtk_dialog (win, G_CALLBACK(run_window_color), this_proj);
129}

230G_MODULE_EXPORT void run_window_color_coord (GtkDialog * win, gint response_id, gpointer data)
231{
232 qint * cid = (qint *)data;
233 int c, g, s;
234 struct project * this_proj = get_project_by_id(cid -> a);
235 s = cid -> b;
236 c = cid -> c;
237 g = cid -> d;
238 if (response_id == GTK_RESPONSE_OK)
239 {
240 if (g > 1) s = 0;
241 this_proj -> modelgl -> anim -> last -> img -> spcolor[g][s][c] = get_window_color (GTK_WIDGET(win));
242 int shaders[4] = {ATOMS, BONDS, POLYS, RINGS};
243 re_create_md_shaders (4, shaders, this_proj);
244 update (this_proj -> modelgl);
245 }
246 destroy_this_dialog (win);

299 break;

302 s = 0;
303 break;
