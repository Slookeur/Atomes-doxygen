46const gchar * default_title (int ax, int c)
47{
48 if (ax == 0)
49 {
50 if (activer == GR || activer == GK)
51 {
52 return ("r [Å]");
53 }
54 else if (activer == SQ || activer == SK)
55 {
56 return ("q [Å-1]");
57 }
58 else if (activer == BD)
59 {
60 return ("Dij [Å]");
61 }
62 else if (activer == AN)
63 {
64 return ("θ [°]");
65 }
66 else if (activer == RI)
67 {
68 return ("Size n of the ring [total number of nodes]");
69 }
70 else if (activer == CH)
71 {
72 return ("Size n of the chain [total number of nodes]");
73 }
74 else if (activer == SP)
75 {
76 return ("Ql");
77 }
78 else
79 {
80 return g_strdup_printf ("t [%s]", untime[active_project -> tunit]);
81 }
82 }
83 else
84 {
85 return active_project -> curves[activer][c] -> name;
86 }
87}
