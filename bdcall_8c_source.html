199void initcutoffs (chemical_data * chem, int species)
200{
201 int i, j;
202 for (i = 0; i < species; i++)
203 {
204 for (j = 0; j < species; j++)
205 {
206 if (chem -> cutoffs[i][j] == 0.0)
207 {
208 chem -> cutoffs[i][j] = chem -> chem_prop[CHEM_R][i] + chem -> chem_prop[CHEM_R][j];
209 chem -> cutoffs[i][j] = max(MINCUT, chem -> cutoffs[i][j]);
210 if (chem -> chem_prop[CHEM_Z][i] == 1.0 && chem -> chem_prop[CHEM_Z][j] == 6.0) chem -> cutoffs[i][j] = 1.2;
211 if (chem -> chem_prop[CHEM_Z][j] == 1.0 && chem -> chem_prop[CHEM_Z][i] == 6.0) chem -> cutoffs[j][i] = 1.2;
212 if (chem -> chem_prop[CHEM_Z][i] == 1.0 && chem -> chem_prop[CHEM_Z][j] == 8.0) chem -> cutoffs[i][j] = 1.2;
213 if (chem -> chem_prop[CHEM_Z][j] == 1.0 && chem -> chem_prop[CHEM_Z][i] == 8.0) chem -> cutoffs[j][i] = 1.2;
214 }
215 }
216 if (chem -> chem_prop[CHEM_Z][i] == 1.0) chem -> cutoffs[i][i] = 0.5;
217 }
218 if (chem -> grtotcutoff == 0.0)
219 {
220 for (i = 0; i < species; i++)
221 {
222 chem -> grtotcutoff += chem -> chem_prop[CHEM_R][i];
223 }
224 chem -> grtotcutoff *= 2.0;
225 chem -> grtotcutoff /= species;
226 chem -> grtotcutoff = max(MINCUT, chem -> grtotcutoff);
227 }

304gboolean run_distance_matrix (GtkWidget * widg, int calc, int up_ngb)
305{
306 int i, j, k;
307 gboolean res;
308 if (up_ngb)
309 {
310 for (i=0; i < active_project -> steps; i++)
311 {
312 for (j=0; j < active_project -> natomes; j++)
313 {
314 active_project -> atoms[i][j].cloned = FALSE;
315 if (active_project -> atoms[i][j].numv)
316 {
317 active_project -> atoms[i][j].numv = 0;
318 if (active_project -> atoms[i][j].vois)
319 {
320 g_free (active_project -> atoms[i][j].vois);
321 active_project -> atoms[i][j].vois = NULL;
322 }
323 }
324 }
325 }
326 }
327 prepostcalc (widg, FALSE, -1, (active_project -> steps > 1) ? 1 : 0, opac);
328 i = j = 0;
329 k = up_ngb;;
330 if (calc > 3 && calc < 6) i = 1;
331 if (calc > 0 && calc < 6)
332 {
333 j = active_project -> rsparam[calc-1][4];
334 }
335 else if (calc == 6)
336 {
337 j = active_project -> csparam[3];
338 }
339#ifdef DEBUG
340 g_debug ("Run dmtx Prim= %d, NOHP= %d, UPDATE= %d", i, j, k);
341#endif
342 clock_gettime (CLOCK_MONOTONIC, & start_time);
343 res = rundmtx_ (& i, & j, & k);
344 prepostcalc (widg, TRUE, -1, 0, 1.0);
345 clock_gettime (CLOCK_MONOTONIC, & stop_time);
346 g_print ("Time to calculate distance matrix: %s\n", calculation_time(FALSE, get_calc_time (start_time, stop_time)));
347 return res;
348}

465G_MODULE_EXPORT void on_calc_bonds_released (GtkWidget * widg, gpointer data)
466{
467 int j, k, l, m;
468 int statusb = 0;
469 int bonding = 0;
470 int * colm = NULL;
471 gboolean vis_bd = active_project -> visok[BD];
472
473 if (widg) bonds_update = 1;
474 bonding = (active_project -> runc[0]) ? 1 : 0;
475 if (! bonds_update && active_project -> runc[0]) bonding = 0;
476 if (active_glwin)
477 {
478 if (active_glwin -> init) colm = save_color_map (active_glwin);
479 prep_ogl_bonds ();
480 }
481 cutoffsend ();
482 if (! active_project -> dmtx) active_project -> dmtx = run_distance_matrix (widg, 0, 1);
483
484 if (active_project -> dmtx)
485 {
486 //if (bonding && active_project -> steps > 1) statusb = 1;
487 if (bonds_update || active_project -> runc[0] || active_project -> runc[2])
488 {
489 if (! active_project -> initok[BD] && bonding) initbd ();
490 if (active_project -> runc[0]) clean_curves_data (BD, 0, active_project -> numc[BD]);
491 prepostcalc (widg, FALSE, BD, statusb, opac);
492 l = 0;
493 m = 1;
494 if (active_project -> bondfile != NULL)
495 {
496 l = 1;
497 m = strlen (active_project -> bondfile);
498 }
499 // debug_chemical_information (active_project);
500 active_project -> delta[BD] = (active_project -> max[BD]-active_project -> min[BD]) / active_project -> num_delta[BD];
501 clock_gettime (CLOCK_MONOTONIC, & start_time);
502 j = bonding_ (& m, & l, & bonding, & active_project -> num_delta[BD], & active_project -> min[BD], & active_project -> delta[BD], active_project -> bondfile);
503 clock_gettime (CLOCK_MONOTONIC, & stop_time);

505 active_project -> runok[SP] = j;
506 prepostcalc (widg, bonding, BD, (bonding) ? j : vis_bd, 1.0);
507 if (! j)
508 {
509 show_error ("Unexpected error when calculating bond properties", 0, (widg) ? widg : MainWindow);
510 }
511 else
512 {
513 if (active_glwin -> init) print_info (calculation_time(TRUE, active_project -> calc_time[BD]), NULL, active_project -> text_buffer[BD+OT]);
514 bonding = 1;
515 if (frag_update)
516 {
517 prepostcalc (widg, FALSE, -1, statusb, opac);
518 k = active_glwin -> allbonds[0] + active_glwin -> allbonds[1];
519 clock_gettime (CLOCK_MONOTONIC, & start_time);
520 if (! molecules_ (& mol_update, & k))
521 {
522 show_error ("Unexpected error when looking for isolated fragment(s) and molecule(s)", 0, (widg) ? widg : MainWindow);
523 if (active_glwin)
524 {
525 for (k=0; k<2; k++)
526 {

528 if (k)
529 {

531 {
532 if (active_project -> force_field[l])

540 }
541 }

579 j = bond_diedrals_ (& active_project -> num_delta[AN]);

589 if (! active_project -> runc[0]) update_ang_view (active_project);
590 }

592 prepostcalc (widg, TRUE, AN, j, 1.0);
593 }
594 }
595 else
596 {
597 show_error ("The nearest neighbors table calculation has failed", 0, (widg) ? widg : MainWindow);
598 bonding = 0;

603 fill_tool_model ();
604 if (widg)
605 {
606 gtk_widget_show (curvetoolbox);
607 }
608 else
609 {
610 for (j=0; j<3; j++) active_project -> runc[j] = FALSE;
611 }
612 bonds_update = frag_update = mol_update = 0;
613}
