78int save_this_mol (FILE * fp, struct project * this_proj, struct molecule * tmp)
79{
80 if (fwrite (& tmp -> id, sizeof(int), 1, fp) != 1) return 0;
81 if (fwrite (& tmp -> md, sizeof(int), 1, fp) != 1) return 0;
82 if (fwrite (& tmp -> multiplicity, sizeof(int), 1, fp) != 1) return 0;
83 if (fwrite (tmp -> fragments, sizeof(int), tmp -> multiplicity, fp) != tmp -> multiplicity) return 0;
84 if (fwrite (& tmp -> natoms, sizeof(int), 1, fp) != 1) return 0;
85 if (fwrite (& tmp -> nspec, sizeof(int), 1, fp) != 1) return 0;
86 if (fwrite (tmp -> species, sizeof(int), this_proj -> nspec, fp) != this_proj -> nspec) return 0;
87 return 1;
88}

98int save_mol (FILE * fp, struct project * this_proj)
99{
100 int i, j;
101 i = 0;
102 if (! this_proj -> modelfc)
103 {
104 if (fwrite (& i, sizeof(int), 1, fp) != 1) return ERROR_MOL;
105 return OK;
106 }
107 i = 1;
108 if (fwrite (& i, sizeof(int), 1, fp) != 1) return ERROR_MOL;
109 for (i=1; i<4; i++)
110 {
111 if (fwrite (& this_proj -> coord -> totcoord[i], sizeof(int), 1, fp) != 1) return ERROR_MOL;
112 }
113 if (fwrite (this_proj -> modelfc -> mol_by_step, sizeof(int), this_proj -> steps, fp) != this_proj -> steps) return ERROR_MOL;
114 for (i=0; i<this_proj -> steps; i++)
115 {
116 for (j=0; j<this_proj -> modelfc -> mol_by_step[i]; j++)
117 {
118 if (! save_this_mol (fp, this_proj, & this_proj -> modelfc -> mols[i][j])) return ERROR_MOL;
119 }
120 }
121 for (i=0; i<this_proj -> steps; i++)
122 {
123 for (j=0; j< this_proj -> natomes; j++)
124 {
125 if (save_atom_m (fp, this_proj, i, j) != OK) return ERROR_MOL;
126 }
127 }
128 return OK;
129}
