261 tmp_fprop = tmp_fprop -> next;

269 if (save_field_prop(fp, fid, tmp_fstr -> st) != OK) return ERROR_RW;

365{
366 if (fwrite (& tmp_fbody -> bd, sizeof(int), 1, fp) != 1) return ERROR_RW;
367 if (fwrite (& tmp_fbody -> id, sizeof(int), 1, fp) != 1) return ERROR_RW;

369 if (tmp_fbody -> bd == 0) if (fwrite (tmp_fbody -> fpid, sizeof(int), 2, fp) != 2) return ERROR_RW;

410int save_dlp_field_data (FILE * fp, struct project * this_proj)
411{
412 int i, j;
413 if (this_proj -> force_field[0] == NULL)
414 {
415 i = 0;
416 if (fwrite (& i, sizeof(int), 1, fp) != 1) return ERROR_RW;
417 return OK;
418 }
419 i = 1;
420 if (fwrite (& i, sizeof(int), 1, fp) != 1) return ERROR_RW;
421 // CONTROL file
422 if (fwrite (this_proj -> force_field[0] -> sys_opts, sizeof(double), 17, fp) != 17) return ERROR_RW;
423 if (fwrite (this_proj -> force_field[0] -> io_opts, sizeof(double), 23, fp) != 23) return ERROR_RW;
424 if (fwrite (this_proj -> force_field[0] -> ana_opts, sizeof(double), 17, fp) != 17) return ERROR_RW;
425 if (fwrite (this_proj -> force_field[0] -> elec_opts, sizeof(double), 11, fp) != 11) return ERROR_RW;
426 if (fwrite (this_proj -> force_field[0] -> vdw_opts, sizeof(double), 6, fp) != 6) return ERROR_RW;
427 if (fwrite (this_proj -> force_field[0] -> met_opts, sizeof(double), 2, fp) != 2) return ERROR_RW;
428 if (fwrite (this_proj -> force_field[0] -> equi_opts, sizeof(double), 17, fp) != 17) return ERROR_RW;
429 if (fwrite (& this_proj -> force_field[0] -> ensemble, sizeof(int), 1, fp) != 1) return ERROR_RW;
430 if (fwrite (& this_proj -> force_field[0] -> thermostat, sizeof(int), 1, fp) != 1) return ERROR_RW;
431 if (fwrite (this_proj -> force_field[0] -> thermo_opts, sizeof(double), 10, fp) != 10) return ERROR_RW;
432 if (fwrite (this_proj -> force_field[0] -> md_opts, sizeof(double), 20, fp) != 20) return ERROR_RW;
433 if (fwrite (this_proj -> force_field[0] -> out_opts, sizeof(double), 31, fp) != 31) return ERROR_RW;
434 // FIELD file
435 if (fwrite (this_proj -> force_field[0] -> prepare_file, sizeof(gboolean), 2, fp) != 2) return ERROR_RW;
436 if (fwrite (this_proj -> force_field[0] -> afp, sizeof(gboolean), MAXDATC+MAXDATA, fp) != MAXDATC+MAXDATA) return ERROR_RW;
437 if (fwrite (& this_proj -> force_field[0] -> type, sizeof(int), 1, fp) != 1) return ERROR_RW;
438 if (fwrite (& this_proj -> force_field[0] -> energy_unit, sizeof(int), 1, fp) != 1) return ERROR_RW;
439 if (fwrite (& this_proj -> force_field[0] -> atom_init, sizeof(int), 1, fp) != 1) return ERROR_RW;
440 if (fwrite (& this_proj -> force_field[0] -> molecules, sizeof(int), 1, fp) != 1) return ERROR_RW;
441 if (! this_proj -> force_field[0] -> molecules) return ERROR_RW;
442 tmp_fmol = this_proj -> force_field[0] -> first_molecule;
443 while (tmp_fmol)
444 {
445 if (save_field_molecule (fp, 0) != OK) return ERROR_RW;
446 tmp_fmol = tmp_fmol -> next;
447 }
448 if (fwrite (this_proj -> force_field[0] -> nbody, sizeof(int), 5, fp) != 5) return ERROR_RW;
449 for (i=0; i<5; i++)
450 {
451 if (this_proj -> force_field[0] -> nbody[i])
452 {
453 tmp_fbody = this_proj -> force_field[0] -> first_body[i];
454 while (tmp_fbody)
455 {
456 if (save_field_body (fp, 0) != OK) return ERROR_RW;
457 tmp_fbody = tmp_fbody -> next;
458 }
459 }
460 }
461 // Tersoff potential cross terms
462 i = (this_proj -> force_field[0] -> cross) ? 1 : 0;
463 if (fwrite (& i, sizeof(int), 1, fp) != 1) return ERROR_RW;
464 if (this_proj -> force_field[0] -> cross)
465 {
466
467 i = this_proj -> force_field[0] -> nbody[2] * (this_proj -> force_field[0] -> nbody[2] - 1) / 2;
468 for (j=0; j<i; j++)
469 {
470 if (fwrite (& this_proj -> force_field[0] -> cross[j], sizeof(double), 3, fp) != 3) return ERROR_RW;
471 }
472 }
473 if (fwrite (& this_proj -> force_field[0] -> extern_fields, sizeof(int), 1, fp) != 1) return ERROR_RW;
474 tmp_fext = this_proj -> force_field[0] -> first_external;
475 while (tmp_fext)
476 {
477 if (save_field_external(fp, 0) != OK) return ERROR_RW;
478 tmp_fext = tmp_fext -> next;
479 }
480 return OK;
481}
