92 {
93 return "Primitive";
94 }
95 else if (groups[spg-1][0] == 'R')

112gchar * get_frac (float val)
113{
114 float vallist[12] = {1.0, 1.0/2.0, 1.0/4.0, 3.0/4.0, 1.0/8.0, 3.0/8.0, 5.0/8.0, 7.0/8.0, 1.0/3.0, 2.0/3.0, 1.0/6.0, 5.0/6.0};
115 gchar * charlist[12] = {"1", "1/2", "1/4", "3/4", "1/8", "3/8", "5/8", "7/8", "1/3", "2/3", "1/6", "5/6"};
116 int i;
117 for (i=0; i<12; i++)
118 {
119 if (fabs(val) - vallist[i] == 0.0) return g_strdup_printf ("%s", charlist[i]);
120 }
121 return NULL;

228void get_wyck_names (space_group * spg, int i, int j)
229{
230 int k, l;
231 double spgpos[3][4];
232 mat4_t wpos;
233 for (k=0; k<3; k++)
234 {
235 tmp_pos = g_strdup_printf ("%s", spg -> wyckoff[i].pos[j][k]);
236 for (l=0; l<3; l++)
237 {
238 spgpos[k][l] = get_val_from_wyckoff (vect_comp[l], spg -> wyckoff[i].pos[j][k]);
239 }
240
241 if (tmp_pos)
242 {
243 spgpos[k][3] = get_value_from_pos (tmp_pos);
244 g_free (tmp_pos);
245 tmp_pos = NULL;
246 }
247 else
248 {
249 spgpos[k][3] = 0.0;
250 }
251 }
252 wpos = mat4 (spgpos[0][0], spgpos[0][1], spgpos[0][2], spgpos[0][3],
253 spgpos[1][0], spgpos[1][1], spgpos[1][2], spgpos[1][3],
254 spgpos[2][0], spgpos[2][1], spgpos[2][2], spgpos[2][3],
255 0.0, 0.0, 0.0, 1.0);
256 if (i == spg -> numw - 1) m4_print (wpos);
257 wpos = m43_mul(spg -> wyck_origin, wpos);
258 for (k=0; k<3; k++)
259 {
260 if (wnpos[k])
261 {
262 g_free (wnpos[k]);
263 wnpos[k] = NULL;
264 }
265 }
266
267 get_wyck_char (wpos.m00, 0, 0);
268 get_wyck_char (wpos.m10, 0, 1);
269 get_wyck_char (wpos.m20, 0, 2);
270 get_extra_val (wpos.m30, 0);
271
272 get_wyck_char (wpos.m01, 1, 0);
273 get_wyck_char (wpos.m11, 1, 1);
274 get_wyck_char (wpos.m21, 1, 2);
275 get_extra_val (wpos.m31, 1);
276
277 get_wyck_char (wpos.m02, 2, 0);
278 get_wyck_char (wpos.m12, 2, 1);
279 get_wyck_char (wpos.m22, 2, 2);
280 get_extra_val (wpos.m32, 2);
281
282 for (k=0; k<3; k++) if (! wnpos[k]) wnpos[k] = g_strdup_printf ("0");
283}

293static void fill_wyckoff_model (GtkTreeStore * store, space_group * spg)
294{
295 GtkTreeIter wlevel;
296 GtkTreeIter mlevel;
297 int i, j, k;
298 gchar * str;
299 gchar * strw[3];
300 if (spg)
301 {
302 for (i=0; i<spg -> numw; i++)
303 {
304 gtk_tree_store_append (store, & wlevel, NULL);
305 str = g_strdup_printf ("%d", spg -> wyckoff[i].multi);
306 gtk_tree_store_set (store, & wlevel, 0, 1,
307 1, i+1,
308 2, str,
309 3, spg -> wyckoff[i].let,
310 4, spg -> wyckoff[i].site, -1);
311 g_free (str);
312 for (j=0; j<spg -> wyckoff[i].multi; j++)
313 {
314 get_wyck_names (spg, i, j);
315 gtk_tree_store_append (store, & mlevel, & wlevel);
316 for (k= 0; k<3; k++) strw[k] = frac_string (wnpos[k]);
317 str = g_strdup_printf ("(%s,%s,%s)", strw[0], strw[1], strw[2]);
318 for (k= 0; k<3; k++) g_free (strw[k]);
319 gtk_tree_store_set (store, & mlevel, 0, 0, 5, str, -1);
320 g_free (str);
321 }
322 }
