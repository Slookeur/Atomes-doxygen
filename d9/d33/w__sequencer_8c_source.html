77void update_selection (glwin * view, int o_step)
78{
79 int i, j, k;
80 i = view -> anim -> last -> img -> step;
81 struct project * this_proj = get_project_by_id(view -> proj);
82 for (k=0; k<2; k++)
83 {
84 view -> anim -> last -> img -> selected[k] -> selected = 0;
85 view -> anim -> last -> img -> selected[k] -> first = NULL;
86 view -> anim -> last -> img -> selected[k] -> last = NULL;
87 for (j=0; j<this_proj -> natomes; j++)
88 {
89 if (this_proj -> atoms[o_step][j].pick[k])
90 {
91 this_proj -> atoms[i][j].pick[k] = FALSE;
92 process_selected_atom (this_proj, view, j, 0, 0, k);
93 }
94 if (this_proj -> atoms[o_step][j].label[k])
95 {
96 this_proj -> atoms[i][j].label[k] = TRUE;
97 }
98 }
99 update_all_selections (view, k);
100 }
101}

144void sequence (glwin * view, int o_step, int n_step)
145{
146 int i;
147 for (i=0; i<2; i++) save_all_selections (view, i);
148 view -> anim -> last -> img -> step = n_step;
149 update_selection (view, o_step);
150 for (i=0; i<NGLOBAL_SHADERS; i++)
151 {
152 if (in_md_shaders (get_project_by_id(view -> proj), i)) view -> n_shaders[i][n_step] = -1;
153 }
154 recreate_all_shaders (view);
155 set_player_title (view);
156 update_step_button (view);
157 update (view);
158}

275static gboolean animate (gpointer data)
276{
277 glwin * view = (glwin *) data;
278 if (view -> play)
279 {
280 if (view -> anim -> last -> img -> step < get_project_by_id(view -> proj) -> steps-1)
281 {
282 sequence (view, view -> anim -> last -> img -> step, view -> anim -> last -> img -> step+1);
283 }
284 else if (view -> loop)
285 {
286 sequence (view, view -> anim -> last -> img -> step, 0);
287 }
288 else
289 {
290 view -> play = FALSE;
291 view -> stop = TRUE;
292 }
293 }
294 else
295 {
296 view -> play = FALSE;
297 view -> stop = TRUE;
298 }
299 return view -> play;
300}
