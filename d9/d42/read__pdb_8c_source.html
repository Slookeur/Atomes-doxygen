104 {
105 h ++;
106 j = omp_get_thread_num();
107 if (! first_at[j])
108 {

112 }
113 else
114 {
115 other_at -> next = g_malloc0(sizeof*other_at -> next);

135 add_spec = TRUE;
136 #pragma omp critical
137 if (other_at -> nz)
138 {
139 if (this_reader -> z)

142 {

171 }

173 ends:;
174 }

179 return 0;
180 }
