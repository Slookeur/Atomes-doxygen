97G_MODULE_EXPORT void set_window_size (GtkEntry * maj, gpointer data)
98{
99 const gchar *m;
100 char * text[2];
101 int i;
102 qint * ad = (qint *)data;
103
104 a = ad -> a;
105 b = ad -> b;
106 c = ad -> c;
107 d = ad -> d;
108 struct project * this_proj = get_project_by_id(a);
109 int shift = get_curve_shift (this_proj, b, c);
110 text[0] = "X size must be > 0";
111 text[1] = "Y size must be > 0";
112 m = entry_get_text (maj);
113
114 i = atof(m);
115 if (i > 0)
116 {
117 switch (d)
118 {
119 case 0:
120 resize_this_window (this_proj -> curves[b][c] -> window, i, this_proj -> curves[b][c] -> wsize[1] + shift);
121 break;
122 case 1:
123 resize_this_window (this_proj -> curves[b][c] -> window, this_proj -> curves[b][c] -> wsize[0], i + shift);
124 break;
125 }
126 this_proj -> curves[b][c] -> wsize[d] = i;
127 }
128 else
129 {
130 show_warning (text[d], this_proj -> curves[b][c] -> window);
131 }
132 update_entry_int (maj, this_proj -> curves[b][c] -> wsize[d]);
133}

154cairo_surface_t * draw_frame_surface (int tf,
155 int da,
156 double ti,
157 double x[2],
158 double y[2],
159 ColRGBA dcol,
160 ColRGBA bcol)
161{
162 cairo_surface_t * cst;
163 cairo_t * tcst;
164 int r[2];
165 r[0] = 100;
166 r[1] = 100;
167 cst = cairo_image_surface_create (CAIRO_FORMAT_ARGB32, 100, 100);
168 tcst = cairo_create (cst);
169 cairo_set_source_rgb (tcst, bcol.red, bcol.green, bcol.blue);
170 cairo_paint (tcst);
171 show_frame (tcst, tf, da, r, ti, x, y, dcol);
172 cairo_destroy (tcst);
173 return cst;
174}

526 qint * cd = (qint *)data;
527 int k;
528 const gchar * m;
529 double z;

531 b = cd -> b;
532 c = cd -> c;
533 d = cd -> d;
534 struct project * this_proj = get_project_by_id(a);
535 m = entry_get_text (fen);
536 z= atof(m);
537 if (d < 2)

539 if (d == 0)
540 {
541 k = 0;

555 {
556 this_proj -> curves[b][c] -> frame_pos[0][1] = z;
557 }
558 else
559 {
560 show_warning ("Frame x max must be > to frame x min", this_proj -> curves[b][c] -> window);
561 }
562 }
563 update_entry_double (fen, this_proj -> curves[b][c] -> frame_pos[0][k]);
564 }
565 else
566 {
567 if (d == 2)
568 {
569 k = 0;
570 if (z > this_proj -> curves[b][c] -> frame_pos[1][1])
571 {
572 this_proj -> curves[b][c] -> frame_pos[1][0] = z;
573 }
574 else
575 {
576 show_warning ("Frame y min must be > to frame y max", this_proj -> curves[b][c] -> window);
577 }
578 }
579 else
580 {
581 k = 1;
582 if (z < this_proj -> curves[b][c] -> frame_pos[1][0])
583 {
584 this_proj -> curves[b][c] -> frame_pos[1][1] = z;

590 }
591 update_entry_double (fen, this_proj -> curves[b][c] -> frame_pos[1][k]);

593 set_frame_style (data);
