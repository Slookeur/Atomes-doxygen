261void duplicate_other_prop (int oid, struct field_struct * old_fstr, struct field_struct * new_fstr)
262{
263 int i = struct_id(new_fstr -> st+7);
264 struct field_prop * tmp_new;
265 struct field_prop * tmp_old = old_fstr -> other;
266 while (tmp_old)
267 {
268 if (check_this_other_prop(oid, i, tmp_old))
269 {
270 if (new_fstr -> other)
271 {
272 tmp_new -> next = duplicate_field_prop (tmp_old, old_fstr -> st);
273 tmp_new = tmp_new -> next;
274 }
275 else
276 {
277 new_fstr -> other = duplicate_field_prop (tmp_old, old_fstr -> st);
278 tmp_new = new_fstr -> other;
279 }
280 }
281 tmp_old = tmp_old -> next;
282 }
283}
