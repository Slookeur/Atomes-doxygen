52struct distance distance_2d (struct atom * at, struct atom * bt)
53{
54 struct distance dist;
55 dist.pbc = FALSE;
56 dist.x = at -> x - bt -> x;
57 dist.y = at -> y - bt -> y;
58 dist.z = 0.0;
59 dist.length = sqrt(dist.x*dist.x + dist.y*dist.y);
60 return dist;
61}

73struct distance distance_3d (cell_info * cell, int mdstep, struct atom * at, struct atom * bt)
74{
75 struct distance dist;
76 double tmp;
77 vec3_t dij;
78 dist.pbc = FALSE;
79 dist.x = at -> x - bt -> x;
80 dist.y = at -> y - bt -> y;
81 dist.z = at -> z - bt -> z;
82 dist.length = sqrt(dist.x*dist.x + dist.y*dist.y + dist.z*dist.z);
83 if (cell -> pbc)
84 {
85 if (cell -> box[mdstep].param[1][0] == 90.0 && cell -> box[mdstep].param[1][1] == 90.0 && cell -> box[mdstep].param[1][2] == 90.0)
86 {
87 dij.x = dist.x - round((at -> x-bt -> x)/cell -> box[mdstep].param[0][0]) * cell -> box[mdstep].param[0][0];
88 dij.y = dist.y - round((at -> y-bt -> y)/cell -> box[mdstep].param[0][1]) * cell -> box[mdstep].param[0][1];
89 dij.z = dist.z - round((at -> z-bt -> z)/cell -> box[mdstep].param[0][2]) * cell -> box[mdstep].param[0][2];
90 }
91 else
92 {
93 vec3_t a = vec3(at -> x, at -> y, at -> z);
94 vec3_t b = vec3(bt -> x, bt -> y, bt -> z);
95 vec3_t af = m4_mul_coord (cell -> box[mdstep].cart_to_frac, a);
96 vec3_t bf = m4_mul_coord (cell -> box[mdstep].cart_to_frac, b);
97 vec3_t nij = v3_sub(af, bf);
98 nij.x -= round(nij.x);
99 nij.y -= round(nij.y);
100 nij.z -= round(nij.z);
101 dij = m4_mul_coord (cell -> box[mdstep].frac_to_cart, nij);
102 }
103 tmp = v3_length(dij);
104 if (dist.length - tmp > 0.001)
105 {
106 dist.pbc = TRUE;
107 dist.x = dij.x;
108 dist.y = dij.y;
109 dist.z = dij.z;
110 dist.length = tmp;
111 }
112 }
113 return dist;
114}

123double arc_cos (double val)
124{
125 if (val < -1.0)
126 {
127 return acos(-2.0 - val) * 180.0 / pi;
128 }
129 else if (val > 1.0)
130 {
131 return acos(2.0 - val) * 180.0 / pi;
132 }
133 else
134 {
135 return acos(val) * 180.0 / pi;
136 }
137}

148struct angle angle_2d (struct atom * at, struct atom * bt, struct atom * ct)
149{
150 struct angle theta;
151 struct distance dist_a = distance_2d (bt, at);
152 struct distance dist_b = distance_2d (bt, ct);
153 theta.pbc = FALSE;
154 double v = 0.0;
155 v = dist_a.x*dist_b.x + dist_a.y*dist_b.y;
156 theta.angle = arc_cos(v/(dist_a.length*dist_b.length));
157 return theta;
158}

196struct angle dihedral_3d (cell_info * cell, int mdstep, struct atom * at, struct atom * bt, struct atom * ct, struct atom * dt)
197{
198 struct angle phi;
199 struct distance dist_a = distance_3d (cell, mdstep, at, bt);
200 struct distance dist_b = distance_3d (cell, mdstep, bt, ct);
201 struct distance dist_c = distance_3d (cell, mdstep, ct, dt);
202 vec3_t u, v;
203
204 if (dist_a.pbc || dist_b.pbc || dist_c.pbc) phi.pbc = TRUE;
205
206 u = vec3(dist_a.y*dist_b.z - dist_a.z*dist_b.y, dist_a.z*dist_b.x - dist_a.x*dist_b.z, dist_a.x*dist_b.y - dist_a.y*dist_b.x);
207 v = vec3(dist_b.y*dist_c.z - dist_b.z*dist_c.y, dist_b.z*dist_c.x - dist_b.x*dist_c.z, dist_b.x*dist_c.y - dist_b.y*dist_c.x);
208
209 if (v3_length(u) == 0.0 || v3_length(v) == 0.0)
210 {
211 phi.angle = 0.0;
212 }
213 else
214 {
215 phi.angle = arc_cos(v3_dot(u, v)/(v3_length(u)*v3_length(v)));
216 }
217 return phi;
218}

232struct angle inversion_3d (cell_info * cell, int mdstep, struct atom * at, struct atom * bt, struct atom * ct, struct atom * dt)
233{
234 struct angle inv;
235 struct distance dist_a = distance_3d (cell, mdstep, bt, at);
236 struct distance dist_b = distance_3d (cell, mdstep, ct, at);
237 struct distance dist_c = distance_3d (cell, mdstep, at, dt);
238 vec3_t u, v, w, x;
239
240 if (dist_a.pbc || dist_b.pbc || dist_c.pbc) inv.pbc = TRUE;
241
242 u = vec3(dist_b.x, dist_b.y, dist_b.z);
243 v = vec3(dist_c.x, dist_c.y, dist_c.z);
244 w = v3_cross (u, v);
245 x = vec3(dist_a.x, dist_a.y, dist_a.z);
246 if (v3_length(w) == 0.0 || v3_length(x) == 0.0)
247 {
248 inv.angle = 0.0;
249 }
250 else
251 {
252 inv.angle = fabs(90.0 - arc_cos(v3_dot(w, x)/(v3_length(w)*v3_length(x))));
253 }
254 return inv;
