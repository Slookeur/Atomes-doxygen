117GtkWidget * attach_grid (project * this_proj, int init)
118{
119 GtkWidget * table = gtk_grid_new ();
120 gchar * str;
121 gboolean actif;
122 int i, j, k, l, m;
123 l = init+1;
124 for (i=0; i<this_proj -> pix[1]; i++)
125 {
126 for (j=0; j<this_proj -> pix[0]; j++)
127 {
128 actif = FALSE;
129 for (k=0; k<27; k++)
130 {
131 if (l == this_proj -> pixels[this_proj -> actif_pix-1][k])
132 {
133 actif = TRUE;
134 }
135 }
136 if (l == this_proj -> actif_pix)
137 {
138 str = g_strdup_printf ("<b>%d</b>", l);
139 m = 1;
140 }
141 else
142 {
143 str = g_strdup_printf ("%d", l);
144 m = 0;
145 }
146 l ++;
147 if (actif)
148 {
149 gtk_grid_attach (GTK_GRID (table), create_css_label(str, m), j, i, 1, 1);
150 }
151 else
152 {
153 gtk_grid_attach (GTK_GRID (table), markup_label(str, 50, 50, 0.5, 0.5), j, i, 1, 1);
154 }
155 g_free (str);
156 }
157 }
158 gtk_widget_set_size_request (table, this_proj -> pix[0]*50, this_proj -> pix[1]*50);
159 return table;
160}

169void update_pix_table (project * this_proj)
170{
171 int i;
172 int pix[3];
173
174 for (i=0; i<3; i++)
175 {
176 this_proj -> pix_tab[i] = destroy_this_widget (this_proj -> pix_tab[i]);
177 }
178
179 // Seed for bottom grid
180 if (this_proj -> actif_pix <= this_proj -> pix[0]*this_proj -> pix[1])
181 {
182 pix[0] = this_proj -> pix[0]*this_proj -> pix[1]*(this_proj -> pix[2]-1);
183 }
184 else
185 {
186 pix[0] = (this_proj -> actif_pix-1)/(this_proj -> pix[0]*this_proj -> pix[1]);
187 pix[0] -= 1;
188 pix[0] *= (this_proj -> pix[0]*this_proj -> pix[1]);
189 }
190
191 // Seed for pixel grid
192 pix[1] = (this_proj -> actif_pix-1)/(this_proj -> pix[0]*this_proj -> pix[1]);
193 pix[1] *= (this_proj -> pix[0]*this_proj -> pix[1]);
194
195 // Seed for top grid
196 if (this_proj -> actif_pix >= this_proj -> pix[0]*this_proj -> pix[1]*(this_proj -> pix[2]-1))
197 {
198 pix[2] = 0;
199 }
200 else
201 {
202 pix[2] = (this_proj -> actif_pix-1)/(this_proj -> pix[0]*this_proj -> pix[1]);
203 pix[2] += 1;
204 pix[2] *= (this_proj -> pix[0]*this_proj -> pix[1]);
205 }
206 for (i=0; i<3; i++)
207 {
208 this_proj -> pix_tab[i] = attach_grid (this_proj, pix[i]);
209 add_box_child_start (GTK_ORIENTATION_HORIZONTAL, this_proj -> pix_box, this_proj -> pix_tab[i], FALSE, FALSE, 20);
210 }
211 show_the_widgets (this_proj -> pix_box);
212}

222G_MODULE_EXPORT void set_pix (GtkEntry * res, gpointer data)
223{
224 int id = GPOINTER_TO_INT (data);
225 project * this_proj = get_project_by_id (id);
226 const gchar * m = entry_get_text (res);
227 double v = atof(m);
228 int p = (int)v;
229 int n = this_proj-> pix[0] * this_proj -> pix[1] * this_proj -> pix[2];
230 if (p > 0 && p <= n)
231 {
232 this_proj -> actif_pix = p;
233 }
234 update_entry_int (res, this_proj -> actif_pix);
235 update_pix_table (this_proj);
236}

245GtkWidget * pixels_tab (project * this_proj)
246{
247 GtkWidget * layout = create_layout (700, 750);
248 GtkWidget * vbox = add_vbox_to_layout (layout, -1, -1);
249 GtkWidget * hbox = create_hbox (0);
250 add_box_child_start (GTK_ORIENTATION_VERTICAL, vbox, hbox, FALSE, FALSE, 5);
251 gchar * str = g_strdup_printf ("\tSelect pixel to check, from 1 to %d:", this_proj -> pix[0]*this_proj -> pix[1]*this_proj->pix[2]);
252 add_box_child_start (GTK_ORIENTATION_HORIZONTAL, hbox, markup_label(str, -1, -1, 0.0, 0.5), FALSE, FALSE, 10);
253 g_free (str);
254 add_box_child_start (GTK_ORIENTATION_HORIZONTAL, hbox, create_entry(G_CALLBACK(set_pix), 100, 15, FALSE, GINT_TO_POINTER(this_proj -> id)), FALSE, FALSE, 10);
255 this_proj -> actif_pix = 1;
256 this_proj -> pix_box = create_hbox (0);
257 add_box_child_start (GTK_ORIENTATION_VERTICAL, vbox, this_proj -> pix_box, FALSE, FALSE, 5);
258 this_proj -> pix_tab[0] = this_proj -> pix_tab[1] = this_proj -> pix_tab[2] = NULL;
259 update_pix_table (this_proj);
260 return layout;
261}
