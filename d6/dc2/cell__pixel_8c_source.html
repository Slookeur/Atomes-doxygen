108GtkWidget * attach_grid (struct project * this_proj, int init)
109{
110 GtkWidget * table = gtk_grid_new ();
111 gchar * str;
112 gboolean actif;
113 int i, j, k, l, m;
114 l = init+1;
115 for (i=0; i<this_proj -> pix[1]; i++)
116 {
117 for (j=0; j<this_proj -> pix[0]; j++)
118 {
119 actif = FALSE;
120 for (k=0; k<27; k++)
121 {
122 if (l == this_proj -> pixels[this_proj -> actif_pix-1][k])
123 {
124 actif = TRUE;
125 }
126 }
127 if (l == this_proj -> actif_pix)
128 {
129 str = g_strdup_printf ("<b>%d</b>", l);
130 m = 1;
131 }
132 else
133 {
134 str = g_strdup_printf ("%d", l);
135 m = 0;
136 }
137 l ++;
138 if (actif)
139 {
140 gtk_grid_attach (GTK_GRID (table), create_css_label(str, m), j, i, 1, 1);
141 }
142 else
143 {
144 gtk_grid_attach (GTK_GRID (table), markup_label(str, 50, 50, 0.5, 0.5), j, i, 1, 1);
145 }
146 g_free (str);
147 }
148 }
149 gtk_widget_set_size_request (table, this_proj -> pix[0]*50, this_proj -> pix[1]*50);
150 return table;
151}

160void update_pix_table (struct project * this_proj)
161{
162 int i;
163 int pix[3];
164
165 for (i=0; i<3; i++)
166 {
167 this_proj -> pix_tab[i] = destroy_this_widget (this_proj -> pix_tab[i]);
168 }
169
170 // Seed for bottom grid
171 if (this_proj -> actif_pix <= this_proj -> pix[0]*this_proj -> pix[1])
172 {
173 pix[0] = this_proj -> pix[0]*this_proj -> pix[1]*(this_proj -> pix[2]-1);
174 }
175 else
176 {
177 pix[0] = (this_proj -> actif_pix-1)/(this_proj -> pix[0]*this_proj -> pix[1]);
178 pix[0] -= 1;
179 pix[0] *= (this_proj -> pix[0]*this_proj -> pix[1]);
180 }
181
182 // Seed for pixel grid
183 pix[1] = (this_proj -> actif_pix-1)/(this_proj -> pix[0]*this_proj -> pix[1]);
184 pix[1] *= (this_proj -> pix[0]*this_proj -> pix[1]);
185
186 // Seed for top grid
187 if (this_proj -> actif_pix >= this_proj -> pix[0]*this_proj -> pix[1]*(this_proj -> pix[2]-1))
188 {
189 pix[2] = 0;
190 }
191 else
192 {
193 pix[2] = (this_proj -> actif_pix-1)/(this_proj -> pix[0]*this_proj -> pix[1]);
194 pix[2] += 1;
195 pix[2] *= (this_proj -> pix[0]*this_proj -> pix[1]);
196 }
197 for (i=0; i<3; i++)
198 {
199 this_proj -> pix_tab[i] = attach_grid (this_proj, pix[i]);
200 add_box_child_start (GTK_ORIENTATION_HORIZONTAL, this_proj -> pix_box, this_proj -> pix_tab[i], FALSE, FALSE, 20);
201 }
202 show_the_widgets (this_proj -> pix_box);
203}

213G_MODULE_EXPORT void set_pix (GtkEntry * res, gpointer data)
214{
215 int id = GPOINTER_TO_INT (data);
216 struct project * this_proj = get_project_by_id (id);
217 const gchar * m = entry_get_text (res);
218 double v = atof(m);
219 int p = (int)v;
220 int n = this_proj-> pix[0] * this_proj -> pix[1] * this_proj -> pix[2];
221 if (p > 0 && p <= n)
222 {
223 this_proj -> actif_pix = p;
224 }
225 update_entry_int (res, this_proj -> actif_pix);
226 update_pix_table (this_proj);
227}

236GtkWidget * pixels_tab (struct project * this_proj)
237{
238 GtkWidget * layout = create_layout (700, 750);
239 GtkWidget * vbox = add_vbox_to_layout (layout, -1, -1);
240 GtkWidget * hbox = create_hbox (0);
241 add_box_child_start (GTK_ORIENTATION_VERTICAL, vbox, hbox, FALSE, FALSE, 5);
242 gchar * str = g_strdup_printf ("\tSelect pixel to check, from 1 to %d:", this_proj -> pix[0]*this_proj -> pix[1]*this_proj->pix[2]);
243 add_box_child_start (GTK_ORIENTATION_HORIZONTAL, hbox, markup_label(str, -1, -1, 0.0, 0.5), FALSE, FALSE, 10);
244 g_free (str);
245 add_box_child_start (GTK_ORIENTATION_HORIZONTAL, hbox, create_entry(G_CALLBACK(set_pix), 100, 15, FALSE, GINT_TO_POINTER(this_proj -> id)), FALSE, FALSE, 10);
246 this_proj -> actif_pix = 1;
247 this_proj -> pix_box = create_hbox (0);
248 add_box_child_start (GTK_ORIENTATION_VERTICAL, vbox, this_proj -> pix_box, FALSE, FALSE, 5);
249 this_proj -> pix_tab[0] = this_proj -> pix_tab[1] = this_proj -> pix_tab[2] = NULL;
250 update_pix_table (this_proj);
251 return layout;
252}
