82void set_motion_sensitive (glwin * view, int status)
83{
84 int i;
85#ifdef GTK3
86 // GTK3 Menu Action To Check
87 for (i=0; i<2; i++) widget_set_sensitive (view -> ogl_anim[i], status);
88#endif
89 if (view -> player != NULL) widget_set_sensitive (view -> player -> win, status);
90 if (view -> spiner != NULL) widget_set_sensitive (view -> spiner -> win, status);
91 if (view -> rec != NULL) widget_set_sensitive (view -> rec -> win, status);
92 if (! status)
93 {
94 if (view -> spin[0] || view -> spin[1])
95 {
96 for (i=0; i<2; i++)
97 {
98 view -> spin[i+2] = view -> spin[i];
99 view -> spin_speed[i+2] = view -> spin_speed[i];
100 view -> spin[i] = FALSE;
101 }
102 }
103 }
104 else if (view -> spin[2] || view -> spin[3])
105 {
106 for (i=0; i<2; i++)
107 {
108 view -> spin[i] = view -> spin[i+2];
109 view -> spin_speed[i] = view -> spin_speed[i+2];
110 g_timeout_add (REFRESH, (GSourceFunc) spin, & view -> colorp[0][i]);
111 view -> spin[i+2] = FALSE;
112 view -> spin_speed[i+2] = 0;
113 }
114 }

164G_MODULE_EXPORT void set_mode (GtkWidget * widg, gpointer data)
165{
166 tint * the_data = (tint *)data;
167 struct project * this_proj = get_project_by_id(the_data -> a);
168 int i = this_proj -> modelgl -> mode;
169 int j = the_data -> b;
170
171 if (! (j == EDITION && is_atom_win_active(this_proj -> modelgl)))
172 {
173#ifdef GTK4
174 if (i != j)
175#else
176 if (i != j && gtk_check_menu_item_get_active ((GtkCheckMenuItem *)widg))
177#endif
178 {
179 this_proj -> modelgl -> mode = NONE;
180#ifdef GTK3
181 // GTK3 Menu Action To Check
182 gtk_check_menu_item_set_active ((GtkCheckMenuItem *)this_proj -> modelgl -> ogl_mode[i], FALSE);
183 if (widg != this_proj -> modelgl -> ogl_mode[j])
184 {
185 gtk_check_menu_item_set_active ((GtkCheckMenuItem *)this_proj -> modelgl -> ogl_mode[j], TRUE);
186 }
187#endif
188 this_proj -> modelgl -> mode = j;
189 if (this_proj -> modelgl -> mode == EDITION)
190 {
191 for (i=1; i<3; i++) init_coordinates (this_proj, i, FALSE, TRUE);
192 set_motion_sensitive (this_proj -> modelgl, 0);
193 }
194 else
195 {
196 for (i=1; i<3; i++)
197 {
198 if (this_proj -> modelgl -> saved_coord[i] != NULL)
199 {
200 g_free (this_proj -> modelgl -> saved_coord[i]);
201 this_proj -> modelgl -> saved_coord[i] = NULL;
202 }
203 }
204 set_motion_sensitive (this_proj -> modelgl, 1);
205 if (this_proj -> modelgl -> selection_mode == NSELECTION-1)
206 {
207#ifdef GTK4
208 set_selection_mode (NULL, & this_proj -> modelgl -> colorp[0][0]);
209#else
210 // GTK3 Menu Action To Check
211 gtk_check_menu_item_set_active ((GtkCheckMenuItem *)this_proj -> modelgl -> ogl_smode[0], TRUE);
212 set_selection_mode (this_proj -> modelgl -> ogl_smode[0], & this_proj -> modelgl -> colorp[0][0]);
213#endif
214 }
215 }
216 i = activep;
217 active_project_changed (this_proj -> id);
218 active_project_changed (i);
219#ifdef GTK3
220 // GTK3 Menu Action To Check
221 widget_set_sensitive (this_proj -> modelgl -> ogl_smode[NSELECTION-1], (this_proj -> modelgl -> mode == EDITION) ? 1 : 0);
222#endif
223 i = (this_proj -> modelgl -> mode == EDITION) ? EDITION : ANALYZE;
224 gchar * str = g_strdup_printf ("%s - 3D view - [%s mode]", prepare_for_title(this_proj -> name), mode_name[i]);
225 gtk_window_set_title (GTK_WINDOW (this_proj -> modelgl -> win), str);
226 g_free (str);
227 this_proj -> modelgl -> create_shaders[MEASU] = TRUE;
228 update (this_proj -> modelgl);
229
230 }
231#ifdef GTK3
232 // GTK3 Menu Action To Check
233 else if (i == j && ! gtk_check_menu_item_get_active ((GtkCheckMenuItem *)widg))
234 {
235 gtk_check_menu_item_set_active ((GtkCheckMenuItem *)this_proj -> modelgl -> ogl_mode[j], TRUE);
236 }
237#endif
238 }

248void invert_visible (struct project * this_proj)
249{
250 int i, j, k;
251 for (i=0; i<this_proj -> steps; i++)
252 {
253 for (j=0; j<this_proj -> natomes; j++)
254 {
255 for (k=0; k<2; k++) this_proj -> atoms[i][j].show[k] = ! this_proj -> atoms[i][j].show[k];
256 }
257 }
258 check_hidden_visible (this_proj);
259 init_default_shaders (this_proj -> modelgl);
260}

523{

534 }

546 mode_name = NULL;
