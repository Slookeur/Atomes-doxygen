323 int * bdim, int bda[* bdim], int bdb[* bdim],
324 double * x, double * y, double * z)

339 active_glwin -> bondid[* stp][* bd][i][0] = j;

351 {
352 active_glwin -> clones[* stp][i].x = x[i];
353 active_glwin -> clones[* stp][i].y = y[i];
354 active_glwin -> clones[* stp][i].z = z[i];
355 j = bda[i] - 1;
356 k = bdb[i] - 1;
357 active_project -> atoms[* stp][j].cloned = TRUE;
358 active_project -> atoms[* stp][k].cloned = TRUE;

448void transform (glwin * view, double aspect)
449{
450 GLdouble w, h;
451 GLdouble dw, dh;
452
453 double zoom = view -> anim -> last -> img -> zoom;
454 view -> zoom_factor = zoom * 0.1 * view -> anim -> last -> img -> p_depth / (2.0 * view -> anim -> last -> img -> gfar);
455 if (view -> anim -> last -> img -> rep == ORTHOGRAPHIC)
456 {
457 zoom *= (view -> anim -> last -> img -> p_depth / view -> anim -> last -> img -> gnear);
458 }
459 dw = view -> anim -> last -> img -> c_shift[0]*2.0*zoom;
460 dh = view -> anim -> last -> img -> c_shift[1]*2.0*zoom;
461 if (aspect > 1.0)
462 {
463 w = zoom * aspect;
464 h = zoom;
465 }
466 else
467 {
468 w = zoom;
469 h = zoom / aspect;
470 }
471 view -> anim -> last -> img -> gleft = -w + dw;
472 view -> anim -> last -> img -> gright = w + dw;
473 view -> anim -> last -> img -> gbottom = -h + dh;
474 view -> anim -> last -> img -> gtop = h + dh;
475
476 if (view -> anim -> last -> img -> rep == PERSPECTIVE)
477 {
478 view -> projection_matrix = m4_frustum (view -> anim -> last -> img -> gleft,
479 view -> anim -> last -> img -> gright,
480 view -> anim -> last -> img -> gbottom,
481 view -> anim -> last -> img -> gtop,
482 view -> anim -> last -> img -> gnear,
483 view -> anim -> last -> img -> gfar);
484 }
485 else
486 {
487 view -> projection_matrix = m4_ortho (view -> anim -> last -> img -> gleft,
488 view -> anim -> last -> img -> gright,
489 view -> anim -> last -> img -> gbottom,
490 view -> anim -> last -> img -> gtop,
491 -view -> anim -> last -> img -> gfar,
492 view -> anim -> last -> img -> gfar);
493 }
494}

612void motion (glwin * view, gint x, gint y, GdkModifierType state)
613{
614 view -> mouseAction = MOTION;
615 int i;
616 if (view -> mouseStatus == CLICKED)
617 {
618 if (view -> mode == EDITION && view -> prepare_motion && view -> rebuild[0][0]) edit_for_motion (view);
619
620 if (state & GDK_BUTTON1_MASK)
621 {
622 arc_ball_rotation (view, x, y);
623 }
624 else if (state & GDK_BUTTON2_MASK)
625 {
626 if (view -> mode != EDITION)
627 {
628 view -> anim -> last -> img -> c_shift[0] -= (double) (x - view -> mouseX) / view -> pixels[0];
629 view -> anim -> last -> img -> c_shift[1] += (double) (y - view -> mouseY) / view -> pixels[1];
630 for (i=0; i<2; i++)
631 {
632 if (view -> camera_widg[i+5])
633 {
634 if (GTK_IS_WIDGET(view -> camera_widg[i+5]))
635 {
636 gtk_spin_button_set_value ((GtkSpinButton *)view -> camera_widg[i+5], - view -> anim -> last -> img -> c_shift[i]);
637 }
638 }
639 }
640 }
641 else
642 {
643 vec3_t pos_a = vec3(x, - y, 0.75);
644 vec3_t pos_b = vec3(view -> mouseX, - view -> mouseY, 0.75);
645 vec3_t trans_a = v3_un_project (pos_a, view -> view_port, view -> projection_matrix);
646 vec3_t trans_b = v3_un_project (pos_b, view -> view_port, view -> projection_matrix);
647 vec3_t trans;
648 trans.x = (trans_a.x - trans_b.x);
649 trans.y = (trans_b.y - trans_a.y);
650 if (view -> anim -> last -> img -> rep == PERSPECTIVE)
651 {
652 trans.x *= view -> anim -> last -> img -> p_depth;
653 trans.y *= view -> anim -> last -> img -> p_depth;
654 }
655 trans.z = 0.0;
656 translate (get_project_by_id(view -> proj), 1, 1, trans);
657 }
658 }
659 if (view -> mode == EDITION)
660 {
661 init_default_shaders (view);
662#ifdef GTK3
663 // GTK3 Menu Action To Check
664 set_advanced_bonding_menus (view);
665#endif
666 }
667 view -> mouseX = x;
668 view -> mouseY = y;
669 update (view);
670 }
