137int set_v_dummy (gchar * this_word)
138{
139 int i;
140 for (i=0; i<this_reader -> ndummy; i++)
141 {
142 if (g_strcmp0(this_reader -> dummy[i], this_word) == 0)
143 {
144 return i+1;
145 }
146 }
147 gchar ** dummy = NULL;
148 if (this_reader -> ndummy)
149 {
150 dummy = duplicate_strings (this_reader -> ndummy, this_reader -> dummy);
151 g_free (this_reader -> dummy);
152 }
153 this_reader -> ndummy ++;
154 this_reader -> dummy = g_malloc0(this_reader -> ndummy*sizeof*this_reader -> dummy);
155 if (dummy)
156 {
157 for (i=0; i<this_reader -> ndummy-1; i++)
158 {
159 this_reader -> dummy[i] = g_strdup_printf ("%s", dummy[i]);
160 }
161 }
162 this_reader -> dummy[this_reader -> ndummy-1] = g_strdup_printf ("%s", this_word);
163 // Dummy added, then do we use this dummy ?
164 gchar * str = g_strdup_printf ("Use dummy atom(s) for unknown %s species ?", this_word);
165 gboolean use_dummy = ask_yes_no ("Use dummy atom(s) ?", str, GTK_MESSAGE_QUESTION, MainWindow);
166 g_free (str);
167 if (use_dummy)
168 {
169 str = g_strdup_printf ("Using dummy atom(s) for unknown %s species", this_word);
170 add_reader_info (str, 1);
171 g_free (str);
172 return this_reader -> ndummy;
173 }
174 else
175 {
176 str = g_strdup_printf ("No dummy atom(s) for unknown %s species", this_word);
177 add_reader_info (str, 1);
178 g_free (str);
179 return 0;
180 }

191void check_for_species (double v, int ato)
192{
193 int i;
194 gboolean add_spec;
195 add_spec = TRUE;
196 if (this_reader -> nspec)
197 {
198 for (i=0; i<this_reader -> nspec; i++)
199 {
200 if (this_reader -> z[i] == v)
201 {
202 if (this_reader -> cartesian)
203 {
204 active_project -> atoms[0][ato].sp = i;
205 }
206 else
207 {
208 this_reader -> lot[ato] = i;
209 }
210 this_reader -> nsps[i] ++;
211 add_spec = FALSE;
212 break;
213 }
214 }
215 }
216 if (add_spec)
217 {
218 if (this_reader -> nspec)
219 {
220 this_reader -> z = g_realloc (this_reader -> z, (this_reader -> nspec+1)*sizeof*this_reader -> z);
221 this_reader -> nsps = g_realloc (this_reader -> nsps, (this_reader -> nspec+1)*sizeof*this_reader -> nsps);
222 }
223 if (this_reader -> cartesian)
224 {
225 active_project -> atoms[0][ato].sp = this_reader -> nspec;
226 }
227 else
228 {
229 this_reader -> lot[ato] = this_reader -> nspec;
230 }
231 this_reader -> nsps[this_reader -> nspec] = 1;
232 this_reader -> z[this_reader -> nspec] = v;
233 this_reader -> nspec ++;
234 }

245int open_coord_file (gchar * filename, int fti)
246{
247 int res;
248#ifdef OPENMP
249 struct stat status;
250 res = stat (filename, & status);
251 if (res == -1)
252 {
253 add_reader_info ("Error - cannot get file statistics !", 0);
254 return 1;
255 }
256 int fsize = status.st_size;
257#endif
258 coordf = fopen (filename, dfi[0]);
259 if (! coordf)
260 {
261 add_reader_info ("Error - cannot open coordinates file !", 0);
262 return 1;
263 }
264 int i, j, k, l;
265#ifdef OPENMP
266 gchar * coord_content = g_malloc0(fsize*sizeof*coord_content);
267 fread (coord_content, fsize, 1, coordf);
268 fclose (coordf);
269 int linecount = 0;
270 for (j=0; j<fsize; j++) if (coord_content[j] == '\n') linecount ++;
271 coord_line = g_malloc0 (linecount*sizeof*coord_line);
272 coord_line[0] = & coord_content[0];
273 i = 1;
274 for (j=0; j<fsize; j++)
275 {
276 if (coord_content[j] == '\n')
277 {
278 coord_content[j] = '\0';
279 if (i < linecount)
280 {
281 coord_line[i] = & coord_content[j+1];
282 i ++;
283 }
284 }
285 }
286#else
287 gchar * buf = g_malloc0(LINE_SIZE*sizeof*buf);
288 head = NULL;
289 tail = NULL;
290 i = 0;
291 while (fgets(buf, LINE_SIZE, coordf))
292 {
293 if (head == NULL)
294 {
295 head = g_malloc0 (sizeof*head);
296 tail = g_malloc0 (sizeof*tail);
297 tail = head;
298 }
299 else
300 {
301 tail -> next = g_malloc0 (sizeof*tail -> next);
302 if (fti == 9)
303 {
304 tail -> next -> prev = g_malloc0 (sizeof*tail -> next -> prev);
305 tail -> next -> prev = tail;
306 }
307 tail = tail -> next;
308 }
309 tail -> line = g_strdup_printf ("%s", buf);
310 tail -> line = substitute_string (tail -> line, "\n", "\0");
311 i ++;
312 }
313 g_free (buf);
314 fclose (coordf);
315#endif
316 if (i)
317 {
318 this_reader -> cartesian = TRUE;
319 if (fti < 2)
320 {
321 res = open_xyz_file (i);
322 }
323 else if (fti == 2)
324 {
325 res = open_c3d_file (i);
326 }
327 else if (fti < 5)
328 {
329 res = open_trj_file (i);
330 }
331 else if (fti < 7)
332 {
333 res = open_vas_file (i);
334 }
335 else if (fti > 6 && fti < 9)
336 {
337 res = open_pdb_file (i);
338 }
339 else if (fti == 9 || fti == 10)
340 {
341 if (fti == 10) cif_use_symmetry_positions = TRUE;
342 this_reader -> cartesian = FALSE;
343 res = open_cif_file (i);
344 }
345 else if (fti == 11)
346 {
347 res = open_hist_file (i);
348 }
349 }
350 else
351 {
352 res = 1;
353 }
354#ifndef OPENMP
355 if (tail) g_free (tail);
356#endif
357 if (! res)
358 {
359 if (fti == 9)
360 {
361 if (! this_reader -> cartesian)
362 {
363 // this_reader -> lattice.sp_group -> sid = 2;
364 // get_origin (this_reader -> lattice.sp_group);
365 if (! cif_use_symmetry_positions)
366 {
367 i = build_crystal (FALSE, active_project, TRUE, FALSE, & this_reader -> lattice, MainWindow);
368 if (! i)
369 {
370 add_reader_info ("Error trying to build crystal using the CIF file parameters !\n"
371 "This usually comes from: \n"
372 "\t - incorrect space group description\n"
373 "\t - incomplete space group description\n"
374 "\t - missing space group setting\n"
375 "\t - incorrect space group setting\n", 0);
376 res = 3;
377 }
378 else if (i > 1)
379 {
380 add_reader_info ("Potential issue(s) when building crystal !\n"
381 "This usually comes from: \n"
382 "\t - incorrect space group description\n"
383 "\t - incomplete space group description\n"
384 "\t - missing space group setting\n"
385 "\t - incorrect space group setting\n", 1);
386 this_reader -> setting = ! this_reader -> setting;
387 if (this_reader -> num_sym_pos)
388 {
389 add_reader_info ("\nAnother model will be built using included symmetry positions\n", 1);
390 cif_use_symmetry_positions = TRUE;
391 }
392 }
393 }
394 }
395 }
396 for (i=0; i<active_project -> steps; i++)
397 {
398 for (j=0; j<active_project -> natomes; j++)
399 {
400 active_project -> atoms[i][j].id = j;
401 active_project -> atoms[i][j].show[0] = TRUE;
402 active_project -> atoms[i][j].show[1] = TRUE;
403 active_project -> atoms[i][j].label[0] = FALSE;
404 active_project -> atoms[i][j].label[1] = FALSE;
405 active_project -> atoms[i][j].pick[0] = FALSE;
406 active_project -> atoms[i][j].cloned = FALSE;
407 }
408 }
409 if (fti != 9 || this_reader -> cartesian)
410 {
411 active_project -> nspec = this_reader -> nspec;
412 active_project -> chemistry = alloc_chem_data (active_project -> nspec);
413 active_project_changed (activep);
414 k = l = 0;
415 reader_info (coord_files_ext[fti], "Number of species", active_project -> nspec);
416 for (i=0; i<active_project -> nspec; i++)
417 {
418 j = (int)this_reader -> z[i];
419 if (this_reader -> z[i] < 1.0)
420 {
421 active_chem -> label[i] = g_strdup_printf ("%s", this_reader -> dummy[l]);
422 active_chem -> element[i] = g_strdup_printf ("Dummy %s", this_reader -> dummy[l]);
423 active_chem -> chem_prop[CHEM_M][i] = 1.0;
424 active_chem -> chem_prop[CHEM_R][i] = 0.5;
425 l ++;
426 }
427 else
428 {
429 active_chem -> label[i] = g_strdup_printf ("%s", periodic_table_info[j].lab);
430 active_chem -> element[i] = g_strdup_printf ("%s", periodic_table_info[j].name);
431 active_chem -> chem_prop[CHEM_M][i] = set_mass_ (& j);
432 active_chem -> chem_prop[CHEM_R][i] = set_radius_ (& j, & k);
433 if (! active_chem -> chem_prop[CHEM_R][i])
434 {
435 gchar * str = g_strdup_printf ("For species %s, radius is equal to 0.0 !", active_chem -> label[i]);
436 add_reader_info (str, 1);
437 g_free (str);
438 }
439 active_chem -> chem_prop[CHEM_N][i] = set_neutron_ (& j);
440 }
441 active_chem -> nsps[i] = this_reader -> nsps[i];
442 g_print ("Reading coordinates [%s]:\t %s, nsps[%d]= %d\n", coord_files_ext[fti], active_chem -> label[i], i+1, active_chem -> nsps[i]);
443 active_chem -> chem_prop[CHEM_Z][i] = this_reader -> z[i];
444 }
445 cif_use_symmetry_positions = FALSE;
446 }
447 else
448 {
449 reader_info (coord_files_ext[fti], "Number of species", active_project -> nspec);
450 for (i=0; i<active_project -> nspec; i++)
451 {
452 g_print ("Reading coordinates [%s]:\t %s, nsps[%d]= %d\n", coord_files_ext[fti], active_chem -> label[i], i+1, active_chem -> nsps[i]);
453 }
454 }
455 }
456 return res;
457}
