121 str_b = g_strdup_printf ("%s.%d.%d", str_a, item_id, mpop);

135 widget_add_action (view_pop_actions, (radio) ? (const gchar *)str_a : (const gchar *)str_b, handler, data,
136 check, status, radio, (const gchar *)str_b);

138 }
139 }

269void menu_bar_attach_color_palettes (glwin * view, GtkWidget * menu_bar)
270{
271 /* Here we need to attached color palettes for:
272 - Box
273 - Atoms
274 - Clones
275 - Total coordination(s)
276 - Partial coordination(s)
277 - Fragment(s) and molecule(s)
278 - Ring(s)
279 - Background
280 This is a work in progress since the 'gtk_popover_menu_bar_add_child' to use to insert widget is bugged:
281 https://gitlab.gnome.org/GNOME/gtk/-/issues/5955
282 */
283 int i, j, k, l, m;
284 gchar * str;
285 struct project * this_proj = get_project_by_id (view -> proj);
286 // Box
287 if (! gtk_popover_menu_bar_add_child ((GtkPopoverMenuBar *)menu_bar, color_palette (view, -1, -1, -1), "set-box-color.0"))
288 {
289 g_debug ("Color palette error: box - custom= set-box-color.0");
290 }
291 // Atom(s) and clone(s)
292 for (i=0; i<2; i++)
293 {
294 for (j=0; j<this_proj -> nspec; j++)
295 {
296 str = g_strdup_printf ("set-%s.%d", (! i) ? "atom-color" : "clone-color", j);
297 if (! gtk_popover_menu_bar_add_child ((GtkPopoverMenuBar *)menu_bar, color_palette (view, i*this_proj -> nspec+j, -1, -1), (const gchar *)str))
298 {
299 g_debug ("Color palette error: %s - %d - custom= %s", (! i) ? "atom-color" : "clone-color", j+1, str);
300 }
301 g_free (str);
302 }
303 }
304 // Coordinations
305 for (i=0; i<2; i++)
306 {
307 if (this_proj -> coord -> ntg[i])
308 {
309 for (j=0; j<this_proj -> nspec; j++)
310 {
311 for (k=0; k<this_proj -> coord -> ntg[i][j]; k++)
312 {
313 m = 0;
314 for (l=0; l<j; l++)
315 {
316 m += this_proj -> coord -> ntg[i][l];
317 }
318 if (i)
319 {
320 str = g_strdup_printf ("set-%s-c.%d", exact_name (env_name (this_proj, k, j, 1, NULL)), m);
321 }
322 else
323 {
324 str = g_strdup_printf ("set-%d-c.%d", this_proj -> coord -> geolist[i][j][k], m);
325 }
326 m += (i) ? this_proj -> coord -> totcoord[0] : 0;
327 if (! gtk_popover_menu_bar_add_child ((GtkPopoverMenuBar *)menu_bar, color_palette (view, 2*this_proj -> nspec+m, -1, -1), (const gchar *)str))
328 {
329 g_debug ("Color palette error: %s - spec= %d - coord= %d, custom= %s", (! i) ? "total-coord" : "partial-coord", j+1, k+1, str);
330 }
331 g_free (str);
332 }
333 }
334 }
335 }
336 // Fragment(s) and molecule(s)
337 for (i=2; i<4; i++)
338 {
339 for (j=0; j<this_proj -> coord -> totcoord[i]; j++)
340 {
341 str = g_strdup_printf ("set-%s-%d", (i == 2) ? "fcol" : "mcol", j);
342 k = 2*this_proj -> nspec + this_proj -> coord -> totcoord[0] + this_proj -> coord -> totcoord[1] + j;
343 if (i == 3) k += this_proj -> coord -> totcoord[2];
344 if (! gtk_popover_menu_bar_add_child ((GtkPopoverMenuBar *)menu_bar, color_palette (view, k, i, 0), (const gchar *)str))
345 {
346 g_debug ("Color palette error: %s - %d, custom= %s", (i == 2) ? "fragment" : "molecule", j+1, str);
347 }
348 g_free (str);
349 }
350 }
351 // Rings
352 for (i=4; i<9; i++)
353 {
354 for (j=0; j<this_proj -> coord -> totcoord[i]; j++)
355 {
356 str = g_strdup_printf ("set-rcol-%d-%d", i, j);
357 if (! gtk_popover_menu_bar_add_child ((GtkPopoverMenuBar *)menu_bar, color_palette (view, -3, i-4, 0), (const gchar *)str))
358 {
359 g_debug ("Color palette error: rings - %d - %d, custom= %s", i, j+1, str);
360 }
361 g_free (str);
362 }
363 }
364 // Background
365 if (! gtk_popover_menu_bar_add_child ((GtkPopoverMenuBar *)menu_bar, color_palette (view, -2, -1, -1), "set-back-color.0"))
366 {
367 g_debug ("Color palette error: background - custom= set-back-color.0");
368 }
369}

378GtkWidget * opengl_window_create_menu_bar (glwin * view)
379{
380 view -> menu_bar = destroy_this_widget (view -> menu_bar);
381 if (view -> action_group) g_object_unref (view -> action_group);
382 view -> action_group = g_simple_action_group_new ();
383 opengl_project = NULL;
384
385 GtkWidget * menu_bar = gtk_popover_menu_bar_new_from_model ((GMenuModel *)opengl_menu_bar(view));
386 menu_bar_attach_color_palettes (view, menu_bar);
387 opengl_project_changed (activev);
388 gchar * str = g_strdup_printf ("gl-%d", view -> action_id);
389 gtk_widget_insert_action_group (menu_bar, str, G_ACTION_GROUP(view -> action_group));
390 g_free (str);
391
392
393 add_box_child_start (GTK_ORIENTATION_HORIZONTAL, view -> menu_box, menu_bar, FALSE, FALSE, 0);
394
395 show_the_widgets (menu_bar);
396 return menu_bar;
397}
