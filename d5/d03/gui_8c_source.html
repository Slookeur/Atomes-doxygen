146void view_buffer (GtkTextBuffer * buffer)

189void atomes_key_pressed (guint keyval, GdkModifierType state)
190{
191 if (state & GDK_CONTROL_MASK)
192 {
193 switch (keyval)
194 {
195 case GDK_KEY_a:
196 create_about_dialog (NULL, NULL);
197 break;
198 case GDK_KEY_c:
199 on_close_workspace (NULL, GINT_TO_POINTER(1));
200 break;
201 case GDK_KEY_n:
202 on_create_new_project (NULL, NULL);
203 break;
204 case GDK_KEY_o:
205 on_open_save_activate (NULL, GINT_TO_POINTER(0));
206 break;
207 case GDK_KEY_p:
208 get_atom_id_from_periodic_table (NULL);
209 break;
210 case GDK_KEY_q:
211#ifdef GTK4
212 leaving_question (NULL, NULL);
213#else
214 leaving_question (NULL, NULL, NULL);
215#endif
216 break;
217 case GDK_KEY_s:
218 on_save_as_activate (NULL, GINT_TO_POINTER(3));
219 break;
220 case GDK_KEY_t:
221 on_show_curve_toolbox (NULL, NULL);
222 break;
223 case GDK_KEY_w:
224 on_open_save_activate (NULL, GINT_TO_POINTER(2));
225 break;
226 }
227 }
228}

479void widget_add_action (GSimpleActionGroup * action_group, const gchar * act, GCallback handler, gpointer data,
480 gboolean check, gboolean status, gboolean radio, const gchar * stat)
481{
482 GSimpleAction * action;
483 GVariant * action_state;
484 if (radio)
485 {
486 action_state = g_variant_new_string (stat);
487 action = g_simple_action_new_stateful (act, G_VARIANT_TYPE_STRING, action_state);
488 }
489 else if (check)
490 {
491 action_state = g_variant_new_boolean (status);
492 action = g_simple_action_new_stateful (act, NULL, action_state);
493 }
494 else
495 {
496 action = g_simple_action_new (act, NULL);
497 }
498 g_signal_connect (action, "activate", handler, data);
499 g_action_map_add_action (G_ACTION_MAP(action_group), G_ACTION(action));
500}

520GMenuItem * create_gmenu_item (const gchar * label, const gchar * action, const gchar * accel,
521 const gchar * custom, int format, const gchar * icon,
522 gboolean check, gboolean status, gboolean radio, const gchar * rstatus)
523{
524 GMenuItem * item;
525 item = g_menu_item_new (label, action);
526 /* GKT4 bug, normally mark-up should be provided using boolean:
527 g_menu_item_set_attribute (item, "use-markup", "b", TRUE, NULL);
528 But it does not work, however it does using a string: */
529 g_menu_item_set_attribute (item, "use-markup", "s", "TRUE", NULL);
530 if (custom)
531 {
532 g_menu_item_set_attribute (item, "custom", "s", custom, NULL);
533 // g_menu_item_set_attribute_value (item, "custom", g_variant_new_string (custom));
534 GVariant * cust = g_menu_item_get_attribute_value (item, "custom", g_variant_type_new("s"));
535 if (cust) g_print ("item :: %s, custom :: %s\n", label, g_variant_get_string (cust, NULL));
536 }
537 else
538 {
539 if (accel) g_menu_item_set_attribute (item, "accel", "s", accel, NULL);
540 // if (check) g_menu_item_set_attribute (item, "target", "b", status, NULL);
541 if (radio) g_menu_item_set_attribute (item, "target", "s", rstatus, NULL);
542#ifdef MENU_ICONS
543 if (icon)
544 {
545 GIcon * gicon = get_gicon_from_data (format, icon);
546 if (gicon)
547 {
548 g_menu_item_set_icon (item, gicon);
549 g_object_unref (gicon);
550 }
551 }
552#endif
553 }
554 return item;
555}
