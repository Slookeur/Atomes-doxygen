154void view_buffer (GtkTextBuffer * buffer)

197void atomes_key_pressed (guint keyval, GdkModifierType state)
198{
199 if (state & GDK_CONTROL_MASK)
200 {
201 switch (keyval)

217 break;

219#ifdef GTK4

222 leaving_question (NULL, NULL, NULL);
223#endif
224 break;

230 break;
231 case GDK_KEY_w:

233 break;
234 }
235 }

487void widget_add_action (GSimpleActionGroup * action_group, const gchar * act, GCallback handler, gpointer data,
488 gboolean check, gboolean status, gboolean radio, const gchar * stat)
489{
490 GSimpleAction * action;
491 GVariant * action_state;
492 if (radio)
493 {
494 action_state = g_variant_new_string (stat);
495 action = g_simple_action_new_stateful (act, G_VARIANT_TYPE_STRING, action_state);
496 }
497 else if (check)
498 {
499 action_state = g_variant_new_boolean (status);
500 action = g_simple_action_new_stateful (act, NULL, action_state);
501 }
502 else
503 {
504 action = g_simple_action_new (act, NULL);
505 }
506 g_signal_connect (action, "activate", handler, data);
507 g_action_map_add_action (G_ACTION_MAP(action_group), G_ACTION(action));
508}

528GMenuItem * create_gmenu_item (const gchar * label, const gchar * action, const gchar * accel,
529 const gchar * custom, int format, const gchar * icon,
530 gboolean check, gboolean status, gboolean radio, const gchar * rstatus)
531{
532 GMenuItem * item;
533 item = g_menu_item_new (label, action);
534 /* GKT4 bug, normally mark-up should be provided using boolean:
535 g_menu_item_set_attribute (item, "use-markup", "b", TRUE, NULL);
536 But it does not work, however it does using a string: */
537 g_menu_item_set_attribute (item, "use-markup", "s", "TRUE", NULL);
538 if (custom)
539 {
540 g_menu_item_set_attribute (item, "custom", "s", custom, NULL);
541 // g_menu_item_set_attribute_value (item, "custom", g_variant_new_string (custom));
542 GVariant * cust = g_menu_item_get_attribute_value (item, "custom", g_variant_type_new("s"));
543 if (cust) g_print ("item :: %s, custom :: %s\n", label, g_variant_get_string (cust, NULL));
544 }
545 else
546 {
547 if (accel) g_menu_item_set_attribute (item, "accel", "s", accel, NULL);
548 // if (check) g_menu_item_set_attribute (item, "target", "b", status, NULL);
549 if (radio) g_menu_item_set_attribute (item, "target", "s", rstatus, NULL);
550#ifdef MENU_ICONS
551 if (icon)
552 {
553 GIcon * gicon = get_gicon_from_data (format, icon);
554 if (gicon)
555 {
556 g_menu_item_set_icon (item, gicon);
557 g_object_unref (gicon);
558 }
559 }
560#endif
561 }
562 return item;
563}
