316Light init_light_source (int type, float val, float vbl)
317{
318 Light new_light;
319 new_light.type = type;
320 new_light.fix = (type != 1) ? 0 : 1;
321 new_light.show = 0;
322 new_light.direction = vec3(0.0, 0.0, 0.0);
323 double intensity = (type == 1) ? 100.0*DEFAULT_INTENSITY : DEFAULT_INTENSITY;
324 // double intensity = DEFAULT_INTENSITY;
325 if (val != vbl)
326 {
327 intensity *= exp (val/vbl);
328 }
329 if (vbl <= 50.0) intensity *= vbl / 100.0;
330
331 new_light.intensity = vec3 (intensity, intensity, intensity);
332 new_light.attenuation = vec3 (1.0, 0.14, 0.07);
333 new_light.spot_data = vec3 (20.0, 20.0, 20.0);
334 if (type == 0)
335 {
336 new_light.position = vec3 (0.0, 0.0, 0.0);
337 new_light.direction = vec3 (0.0, 0.0, -1.0);
338 }
339 else
340 {
341 new_light.position = vec3 (vbl*1.5, 0.0, 0.0);
342 if (type == 2)
343 {
344 new_light.intensity = v3_muls (new_light.intensity, 100.0);
345 float tan = (val * sqrt(2.0) / 2.0) / (vbl - val);
346 float tetha = fabs(atanf (tan)) * 90.0 / pi;
347 new_light.spot_data = vec3 (tetha, tetha, tetha);
348 }
349 }
350 return new_light;
351}

522void add_remove_lights (int val, gpointer data)
523{
524 int i, j, k, m;
525 glwin * view = (glwin *)data;
526 struct project * this_proj = get_project_by_id(view -> proj);
527 image * this_image = view -> anim -> last -> img;
528 gboolean delete_ligth;
529 i = this_image -> lights;
530 Light * old_spots;
531 if (val > i)
532 {
533#ifdef DEBUG
534 g_debug ("ADDING_LIGHT_SOURCE:: val= %d, i= %d", val, i);
535#endif
536 // Adding light source(s)
537 old_spots = copy_light_sources (i, i, this_image -> l_ght);
538 g_free (this_image -> l_ght);
539 this_image -> l_ght = copy_light_sources (val, i, old_spots);
540 this_image -> lights = val;
541 float pos = (this_proj -> cell.box[0].param[0][0] == 0.0) ? 20.0 : this_proj -> cell.box[0].param[0][0];
542 for (j=i; j<val; j++)
543 {
544 this_image -> l_ght[j] = init_light_source (0, pos, this_proj -> modelgl -> p_moy); // Init directional by default
545 }
546 //free (old_spots);
547 }
548 else if (val < i)
549 {
550 // We need to remove a light
551#ifdef DEBUG
552 g_debug ("REMOVING_LIGHT_SOURCE:: val= %d, i= %d", val, i);
553#endif
554 int * ltr = light_source_to_be_removed (i-val, this_image, view -> opengl_win);
555 if (ltr != NULL)
556 {
557 old_spots = copy_light_sources (i, i, this_image -> l_ght);
558 for (k=0; k < i-val; k++)
559 {
560#ifdef DEBUG
561 g_debug ("REMOVING_LIGHT_SOURCES:: k= %d, ltr[%d]= %d", k, k, ltr[k]);
562#endif
563 }
564 g_free (this_image -> l_ght);
565 this_image -> l_ght = g_malloc (val*sizeof*this_image -> l_ght);
566 m = -1;
567 for (j=0; j<i; j++)
568 {
569 delete_ligth = FALSE;
570 for (k=0; k< i-val; k++)
571 {
572 if (j == ltr[k]) delete_ligth = TRUE;
573 }
574 if (! delete_ligth)
575 {
576 m ++;
577 this_image -> l_ght[m] = copy_light_source (old_spots[j]);
578 }
579 }
580 g_free (old_spots);
581 this_image -> lights = val;
582#ifdef DEBUG
583 g_debug ("LIGHT(s) HAVE BEEN REMOVED:: NEW_LIGHTS_NUM= %d", val);
584#endif
585 }
586 }
587 //this_light = this_image -> l_ght;
588 view -> create_shaders[LIGHT] = TRUE;
589 view -> opengl_win -> lights = destroy_this_widget (view -> opengl_win -> lights);
590 create_lights_combo (this_image, view -> opengl_win);
591 gtk_widget_show (view -> opengl_win -> lights);
592 gtk_combo_box_set_active (GTK_COMBO_BOX(view -> opengl_win -> lights), 0);
593 update_light_data (0, view -> opengl_win);
594 update (view);
595}
