185int is_in_table (int i, int j)
186{
187 if (i == 7) return 0;
188 if (i == 0 && (j > 0 && j < 17)) return 0;
189 if ((i == 1 || i == 2) && (j > 1 && j < 12)) return 0;
190 if ((i == 5 || i == 6) && j ==2) return 0;
191 if (i > 7 && j< 3) return 0;
192 if (i > 2 && i < 5) return 19 + (i-3)*18 + j;
193 if (i == 0)
194 {
195 if (j==0)
196 {
197 return 1;
198 }
199 else
200 {
201 return 2;
202 }
203 }
204 if (i > 0 && i < 3) return 2 + j+1 - 10 * (j/11) + 8 * (i/2);
205 if (i == 5 && j < 2) return 55 + j;
206 if (i == 6 && j < 2) return 87 + j;
207 if (i == 5 && j > 2) return 72 + j-3;
208 if (i == 6 && j > 2) return 104 + j-3;
209 if (i == 8) return 57 + j-3;
210 if (i == 9) return 89 + j-3;
211 return -1;
212}
