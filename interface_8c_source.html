478G_MODULE_EXPORT void run_iask (GtkDialog * iask, gint response_id, gpointer data)
479{
480 int i = GPOINTER_TO_INT(data);
481 gboolean done = FALSE;
482 const gchar * riask;
483 if (response_id == GTK_RESPONSE_OK)
484 {
485 if (i == 0 || i > 3)
486 {
487 riask = entry_get_text (GTK_ENTRY(answer));
488 res_int = atof(riask);
489 if (i > 4)
490 {
491 if (res_int > 0 && res_int < i+1)
492 {
493 done = TRUE;
494 res_int --;
495 }
496 }
497 else if (res_int > 0)
498 {
499 done = TRUE;
500 }
501 }
502 else
503 {
504 done = TRUE;
505 res_int = gtk_combo_box_get_active (GTK_COMBO_BOX(answer));
506 }
507 }
508 if (done) destroy_this_dialog (iask);
509}
