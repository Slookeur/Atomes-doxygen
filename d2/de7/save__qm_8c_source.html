114int save_thermo (FILE * fp, thermostat * thermo)
115{
116 if (fwrite (& thermo -> id, sizeof(int), 1, fp) != 1) return ERROR_RW;
117 if (fwrite (& thermo -> type, sizeof(int), 1, fp) != 1) return ERROR_RW;
118 if (fwrite (& thermo -> sys, sizeof(int), 1, fp) != 1) return ERROR_RW;
119 if (fwrite (& thermo -> show, sizeof(gboolean), 1, fp) != 1) return ERROR_RW;
120 if (fwrite (thermo -> params, sizeof(double), 4, fp) != 4) return ERROR_RW;
121 if (fwrite (& thermo -> natoms, sizeof(int), 1, fp) != 1) return ERROR_RW;
122 return OK;
123}

135int save_fixed_atoms (FILE * fp, int fixatoms, int * fixlist, int ** fixcoord)
136{
137 int i;
138 if (fwrite (& fixatoms, sizeof(int), 1, fp) != 1) return ERROR_RW;
139 if (fixatoms)
140 {
141 if (fixlist)
142 {
143 if (fwrite (fixlist, sizeof(int), fixatoms, fp) != fixatoms) return ERROR_RW;
144 }
145 else
146 {
147 i = 0;
148 if (fwrite (& i, sizeof(int), 1, fp) != 1) return ERROR_RW;
149 }
150 if (fixcoord)
151 {
152 for (i=0; i<fixatoms; i++) if (fwrite (fixcoord[i], sizeof(int), 3, fp) != 3) return ERROR_RW;
153 }
154 else
155 {
156 i = 0;
157 if (fwrite (& i, sizeof(int), 1, fp) != 1) return ERROR_RW;
158 }
159 }
160 return OK;
161}

172int save_cpmd_data (FILE * fp, int cid, project * this_proj)
173{
174 int i;
175 if (this_proj -> cpmd_input[cid] == NULL)
176 {
177 i = 0;
178 if (fwrite (& i, sizeof(int), 1, fp) != 1) return ERROR_RW;
179 return OK;
180 }
181 i = 1;
182 if (fwrite (& i, sizeof(int), 1, fp) != 1) return ERROR_RW;
183 if (fwrite (& this_proj -> cpmd_input[cid] -> calc_type, sizeof(int), 1, fp) != 1) return ERROR_RW;
184 if (fwrite (this_proj -> cpmd_input[cid] -> default_opts, sizeof(double), 17, fp) != 17) return ERROR_RW;
185 if (fwrite (this_proj -> cpmd_input[cid] -> calc_opts, sizeof(double), 24, fp) != 24) return ERROR_RW;
186 if (fwrite (& this_proj -> cpmd_input[cid] -> thermostats, sizeof(int), 1, fp) != 1) return ERROR_RW;
187 if (this_proj -> cpmd_input[cid] -> thermostats)
188 {
189 thermostat * thermo = this_proj -> cpmd_input[cid] -> ions_thermostat;
190 i = 0;
191 while (thermo)
192 {
193 if (save_thermo (fp, thermo) != OK) return ERROR_RW;
194 i ++;
195 thermo = thermo -> next;
196 }
197 i = (this_proj -> cpmd_input[cid] -> elec_thermostat) ? 1 : 0;
198 if (fwrite (& i, sizeof(int), 1, fp) != 1) return ERROR_RW;
199 if (this_proj -> cpmd_input[cid] -> elec_thermostat)
200 {
201 if (save_thermo (fp, this_proj -> cpmd_input[cid] -> elec_thermostat) != OK) return ERROR_RW;
202 }
203 }
204 if (save_fixed_atoms (fp, this_proj -> cpmd_input[cid] -> fixat, this_proj -> cpmd_input[cid] -> fixlist, this_proj -> cpmd_input[cid] -> fixcoord) != OK)
205 {
206 return ERROR_RW;
207 }
208 if (fwrite (& this_proj -> cpmd_input[cid] -> dummies, sizeof(int), 1, fp) != 1) return ERROR_RW;
209 if (this_proj -> cpmd_input[cid] -> dummies)
210 {
211 dummy_atom * dummy = this_proj -> cpmd_input[cid] -> dummy;
212 while (dummy)
213 {
214 if (fwrite (& dummy -> id, sizeof(int), 1, fp) != 1) return ERROR_RW;
215 if (fwrite (& dummy -> type, sizeof(int), 1, fp) != 1) return ERROR_RW;
216 if (fwrite (& dummy -> show, sizeof(gboolean), 1, fp) != 1) return ERROR_RW;
217 if (fwrite (dummy -> xyz, sizeof(double), 3, fp) != 3) return ERROR_RW;
218 if (fwrite (dummy -> coord, sizeof(int), 4, fp) != 4) return ERROR_RW;
219 if (fwrite (& dummy -> natoms, sizeof(int), 1, fp) != 1) return ERROR_RW;
220 if (dummy -> natoms)
221 {
222 if (fwrite (dummy -> list, sizeof(int), dummy -> natoms, fp) != dummy -> natoms) return ERROR_RW;
223 }
224 dummy = dummy -> next;
225 }
226 }
227 for (i=0; i<this_proj -> nspec; i++)
228 {
229 if (fwrite (this_proj -> cpmd_input[cid] -> pp[i], sizeof(int), 2, fp) != 2) return ERROR_RW;
230 }
231 return save_this_string (fp, this_proj -> cpmd_input[cid] -> info);
232}

243int save_cp2k_data (FILE * fp, int cid, project * this_proj)
244{
245 int i, j;
246 if (this_proj -> cp2k_input[cid] == NULL)
247 {
248 i = 0;
249 if (fwrite (& i, sizeof(int), 1, fp) != 1) return ERROR_RW;
250 return OK;
251 }
252 i = 1;
253 if (fwrite (& i, sizeof(int), 1, fp) != 1) return ERROR_RW;
254 if (fwrite (& this_proj -> cp2k_input[cid] -> input_type, sizeof(int), 1, fp) != 1) return ERROR_RW;
255 if (fwrite (this_proj -> cp2k_input[cid] -> opts, sizeof(double), 42, fp) != 42) return ERROR_RW;
256 for (i=0; i<3; i++)
257 {
258 if (fwrite (this_proj -> cp2k_input[cid] -> extra_opts[i], sizeof(double), 4, fp) != 4) return ERROR_RW;
259 }
260 if (fwrite (& this_proj -> cp2k_input[cid] -> thermostats, sizeof(int), 1, fp) != 1) return ERROR_RW;
261 if (this_proj -> cp2k_input[cid] -> thermostats)
262 {
263 thermostat * thermo = this_proj -> cp2k_input[cid] -> ions_thermostat;
264 while (thermo)
265 {
266 if (save_thermo (fp, thermo) != OK) return ERROR_RW;
267 thermo = thermo -> next;
268 }
269 }
270 for (i=0; i<2; i++)
271 {
272 if (save_fixed_atoms (fp, this_proj -> cp2k_input[cid] -> fixat[i], this_proj -> cp2k_input[cid] -> fixlist[i], this_proj -> cp2k_input[cid] -> fixcoord[i]) != OK)
273 {
274 return ERROR_RW;
275 }
276 }
277 for (i=0; i<this_proj -> nspec; i++)
278 {
279 if (fwrite (this_proj -> cp2k_input[cid] -> spec_data[i], sizeof(int), 2, fp) != 2) return ERROR_RW;
280 for (j=0; j<2; j++)
281 {
282 if (save_this_string (fp, this_proj -> cp2k_input[cid] -> spec_files[i][j]) != OK) return ERROR_RW;

284 }
285 for (i=0; i<5; i++)
286 {
287 if (save_this_string (fp, this_proj -> cp2k_input[cid] -> files[i]) != OK) return ERROR_RW;
