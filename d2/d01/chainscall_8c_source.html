100void update_chains_view (struct project * this_proj)
101{

103 gchar * nelt;

105 gchar * tab;
106 gchar * cid;
107 gchar * str;
108
109 if (this_proj -> text_buffer[CH+OT] == NULL) this_proj -> text_buffer[CH+OT] = add_buffer (NULL, NULL, NULL);
110 view_buffer (this_proj -> text_buffer[CH+OT]);
111 j = this_proj -> csparam[0];
112 if (! j)
113 {
114 nelt = g_strdup_printf ("All");
115 col = NULL;
116 }
117 else
118 {
119 nelt = g_strdup_printf ("%s", this_proj -> chemistry -> label[j-1]);

121 }
122 print_info ("\n\nChain statistics\n\n", "heading", this_proj -> text_buffer[CH+OT]);

124
125 if (this_proj -> csparam[1])

128 }
129 if (this_proj -> csparam[2])

131 print_info (" * only ABAB chains have been considered\n", "italic", this_proj -> text_buffer[CH+OT]);
132 }
133 if (this_proj -> csparam[3])

135 print_info (" * homopolar bonds can not shorten the chains\n", "italic", this_proj -> text_buffer[CH+OT]);

143 print_info ("\n Atom(s) used to start the search: ", NULL, this_proj -> text_buffer[CH+OT]);

159 print_info ("\n Total number of chains: ", NULL, this_proj -> text_buffer[CH+OT]);
160 str = g_strdup_printf ("%f\n", this_proj -> csdata[0]);
161 print_info (str, "bold", this_proj -> text_buffer[CH+OT]);
162 g_free (str);
163 }
164
165 if (this_proj -> steps > 1)
166 {
167 print_info ("\n\t n Av. by step \tCc(n)[", "bold", this_proj -> text_buffer[CH+OT]);
168 print_info (nelt, col, this_proj -> text_buffer[CH+OT]);
169 if (j == this_proj -> nspec)
170 {
171 print_info ("]\t +/-\n", "bold", this_proj -> text_buffer[CH+OT]);
172 }
173 else
174 {
175 print_info ("]\t +/-\n", "bold", this_proj -> text_buffer[CH+OT]);
176 }
177 }
178 else
179 {
180 print_info ("\n\t n\tNumber\t\tCc(n)[", "bold", this_proj -> text_buffer[CH+OT]);
181 print_info (nelt, col, this_proj -> text_buffer[CH+OT]);
182 print_info ("]\n", NULL, this_proj -> text_buffer[CH+OT]);
183 }
184 tab = NULL;
185 cid = NULL;

187 l = (! k) ? this_proj -> natomes : this_proj -> chemistry -> nsps[k-1];
188 j = 1;

191 if (this_proj -> curves[CH][k] -> data[1][i] != 0.0)

193 j ++;
194 if (j - 2*(j/2) == 0)
195 {
196 tab = g_strdup_printf ("grey_back");
197 cid = g_strdup_printf ("bold_grey_back");
198 }

200 {

203 }
204 print_info ("\t", NULL, this_proj -> text_buffer[CH+OT]);
205 if (i < 9)
206 {
207 print_info (" ",cid, this_proj -> text_buffer[CH+OT]);
208 }
209 str = g_strdup_printf("%d", i+1);
210 print_info (str, cid, this_proj -> text_buffer[CH+OT]);
211 g_free (str);
212 str = g_strdup_printf("\t%f\t", l*this_proj -> curves[CH][k] -> data[1][i]);
213 print_info (str, tab, this_proj -> text_buffer[CH+OT]);
214 g_free (str);
215 str = g_strdup_printf("%f\t", this_proj -> curves[CH][k] -> data[1][i]);
216 print_info (str, tab, this_proj -> text_buffer[CH+OT]);
217 g_free (str);
218 if (this_proj -> steps > 1)
219 {
220 str = g_strdup_printf("%f\t", this_proj -> curves[CH][k] -> err[i]);
221 print_info (str, tab, this_proj -> text_buffer[CH+OT]);
222 g_free (str);
223 }
224 print_info ("\n", NULL, this_proj -> text_buffer[CH+OT]);
225 if (tab != NULL)
226 {
227 g_free (tab);
228 }
229 if (cid != NULL)
230 {
231 g_free (cid);
232 }
233 }
234 }
235 print_info (calculation_time(TRUE, this_proj -> calc_time[CH]), NULL, this_proj -> text_buffer[CH+OT]);
236 g_free (nelt);
237 if (col != NULL)
238 {
239 g_free (col);
240 }

250void clean_chains_data (glwin * view)
251{
252 struct project * this_proj = get_project_by_id(view -> proj);
253 if (this_proj -> coord -> totcoord[9])
254 {
255 if (this_proj -> coord -> ntg[9]) g_free (this_proj -> coord -> ntg[9]);
256 this_proj -> coord -> ntg[9] = NULL;
257 if (this_proj -> coord -> geolist[9][0]) g_free (this_proj -> coord -> geolist[9][0]);
258 this_proj -> coord -> geolist[9][0] = NULL;
259 this_proj -> coord -> totcoord[9] = 0;
260 }
261 if (view -> all_chains) g_free (view -> all_chains);
262 view -> all_chains = NULL;
263 if (view -> num_chains) g_free (view -> num_chains);
264 view -> num_chains = NULL;
265 if (view -> gcid[9]) g_free (view -> gcid[9]);
266 view -> gcid[9] = NULL;
267 view -> chains = FALSE;
268 view -> chain_max = 0;
269 int i, j;
270 for (i=0; i<this_proj -> steps; i++)
271 {
272 for (j=0; j<this_proj -> natomes; j++)
273 {
274 if (this_proj -> atoms[i][j].chain)
275 {
276 g_free (this_proj -> atoms[i][j].chain);
277 this_proj -> atoms[i][j].chain = NULL;
278 }
279 }
280 }

291G_MODULE_EXPORT void on_calc_chains_released (GtkWidget * widg, gpointer data)
292{
293 int j, k;
294
295 cutoffsend ();
296 //if (active_project -> steps > 1) statusb = 1;
297 if (! active_project -> initok[CH]) initchn ();
298 active_project -> csparam[6] = 0;
299 if (! active_project -> dmtx) active_project -> dmtx = run_distance_matrix (widg, 6, 0);
300
301 if (active_project -> dmtx)
302 {
303 clean_curves_data (CH, 0, active_project -> numc[CH]);
304 clean_chains_data (active_glwin);
305 active_glwin -> all_chains = g_malloc0 (active_project -> steps*sizeof*active_glwin -> all_chains);
306 active_glwin -> num_chains = g_malloc0 (active_project -> steps*sizeof*active_glwin -> num_chains);
307 for (j=0; j<active_project -> steps; j++)
308 {
309 active_glwin -> all_chains[j] = g_malloc0 (active_project -> csparam[5]*sizeof*active_glwin -> all_chains[j]);
310 active_glwin -> num_chains[j] = allocint (active_project -> csparam[5]);
311 for (k=0; k < active_project -> natomes; k++)
312 {
313 if (active_project -> atoms[j][k].chain) g_free (active_project -> atoms[j][k].chain);
314 active_project -> atoms[j][k].chain = NULL;
315 active_project -> atoms[j][k].chain = g_malloc0 (active_project -> csparam[5]*sizeof*active_project -> atoms[j][k].chain);
316 }
317 }
318 k = 1;
319 prepostcalc (widg, FALSE, CH, 0, opac);
320 clock_gettime (CLOCK_MONOTONIC, & start_time);
321 j = initchains_ (& active_project -> csparam[0],
322 & active_project -> csparam[1],
323 & active_project -> csparam[2],
324 & active_project -> csparam[3],
325 & active_project -> csparam[4],
326 & active_project -> csparam[5],
327 & active_project -> csearch);
328 clock_gettime (CLOCK_MONOTONIC, & stop_time);
329 active_project -> calc_time[CH] = get_calc_time (start_time, stop_time);
330 if (j == 0)
331 {
332 show_error ("The chain statistics calculation has failed", 0, widg);
333 }
334 else if (j == 2)
335 {
336 gchar * str = g_strdup_printf ("\t<b>The chain statistics have failed !</b>\n\n"
337 "The number of chain per MD step appears\n"
338 "to be bigger than the initial value of <b>%d</b>\n"
339 "used to allocate memory to store the results.\n\n"
340 "Increase the value and start again !",
341 active_project -> csearch);
342 show_error (str, 0, widg);
343 g_free (str);
344 j = 0;
345 }
346 prepostcalc (widg, TRUE, CH, j, 1.0);
347 }
348 else
349 {
350 show_error ("The nearest neighbors table calculation has failed", 0, widg);
351 }
352 if (active_coord -> totcoord[9])
353 {
354 active_project -> csparam[6] = 1;
355 active_glwin -> chains = TRUE;
356 active_glwin -> chain_max = active_project -> csparam[5];
357 update_chains_view (active_project);
358 }
359 gtk_widget_show (curvetoolbox);
360 clean_coord_window (active_project);
361 fill_tool_model ();
362#ifdef GTK3
363 update_chains_menus (active_glwin);
364#endif
365}
