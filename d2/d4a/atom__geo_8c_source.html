52void sort_partial_geo (int ** geom, int num_a)
53{
54 int i, j, k, l;
55
56 for(i=0;i<num_a;i++)
57 {
58 for(j=i+1;j<num_a;j++)
59 {
60 if(geom[i][0] > geom[j][0])
61 {
62 for (k=0; k<2; k++)
63 {
64 l = geom[i][k];

87int is_this_a_new_geo (int gid, coord_info * obj, int * old_z, int old_geo, int old_sp, int new_sp, coord_info * coord, double * new_z)
88{
89 int i, j, k, l, m, n, o;
90 int ** n_part, ** o_part;
91
92 // Number of coord of type id for spec new_sp
93 i = coord -> ntg[gid][new_sp];
94 // Using j to store the total number of neighbor(s) for the coordination to test
95 // Using k to store the number of type(s) of neighbor(s) for the coordination to test
96 j = k = 0;
97 for (l=0; l<obj -> species; l++)
98 {
99 if (obj -> partial_geo[old_sp][old_geo][l])
100 {
101 k ++;
102 j += obj -> partial_geo[old_sp][old_geo][l];
103 }
104 }
105 // Then comparing with already stored data in coord
106 for (l=0; l<i; l++)
107 {
108 switch (gid)
109 {
110 case 0:
111 if (coord -> geolist[gid][new_sp][l] == obj -> geolist[gid][old_sp][old_geo]) return l;
112 break;
113 case 1:
114 m = 0;
115 n = 0;
116 for (o=0; o<coord -> species; o++)
117 {
118 if (coord -> partial_geo[new_sp][l][o])
119 {
120 m ++;
121 n += coord -> partial_geo[new_sp][l][o];
122 }
123 }
124 if (j == n && k == m)
125 {
126 // Same number of atoms in the coordination for the spec
127 // 2 structures of size l: [0]= Z, [1] = nato
128 // Sort by Z, then comparing
129 n_part = allocdint (m, 2);
130 o_part = allocdint (m, 2);
131 m = 0;
132 for (n=0; n<coord -> species; n++)
133 {
134 if (coord -> partial_geo[new_sp][l][n])
135 {
136 n_part[m][0] = (int)new_z[n];
137 n_part[m][1] = coord -> partial_geo[new_sp][l][n];
138 m ++;
139 }
140 }
141 if (m > 1) sort_partial_geo (n_part, m);
142 m = 0;
143 for (n=0; n<obj -> species; n++)
144 {
145 if (obj -> partial_geo[old_sp][old_geo][n])
146 {
147 o_part[m][0] = old_z[n];
148 o_part[m][1] = obj -> partial_geo[old_sp][old_geo][n];
149 m ++;
150 }
151 }
152 if (m > 1) sort_partial_geo (o_part, m);
153 n = 1;
154 for (o=0; o<m; o++)
155 {
156 if (o_part[o][0] != n_part[o][0] || o_part[o][1] != n_part[o][1])
157 {
158 n = 0;
159 break;
160 }
161 }
162 g_free (n_part);
163 g_free (o_part);
164 if (n) return l;
165 }
166 break;
167 }
168 }
169
170 // If we keep going then this is a new type of coordination sphere
171 int * tmpgeol = NULL;
172 if (! gid)
173 {
174 tmpgeol = allocint (coord -> ntg[gid][new_sp]+1);
175 for (j=0; j<coord -> ntg[gid][new_sp]; j++)
176 {
177 tmpgeol[j] = coord -> geolist[gid][new_sp][j];
178 }
179 tmpgeol[j] = obj -> geolist[gid][old_sp][old_geo];
180 }
181 else
182 {
183 int ** tmp_part = NULL;
184 tmp_part = g_malloc0 (coord -> ntg[1][new_sp]*sizeof*tmp_part);
185 for (j=0; j<coord -> ntg[1][new_sp]; j++)
186 {
187 tmp_part[j] = duplicate_int (coord -> species, coord -> partial_geo[new_sp][j]);
188 }
189 g_free (coord -> partial_geo[new_sp]);
190 coord -> partial_geo[new_sp] = g_malloc0 ((coord -> ntg[1][new_sp]+1)*sizeof*coord -> partial_geo[new_sp]);
191 if (tmp_part)
192 {
193 for (j=0; j<coord -> ntg[1][new_sp]; j++)
194 {
195 coord -> partial_geo[new_sp][j] = duplicate_int (coord -> species, tmp_part[j]);
196 }
197 g_free (tmp_part);
198 }
199 coord -> partial_geo[new_sp][j] = allocint (coord -> species);
200 for (k=0; k<obj -> species; k++)
201 {
202 if (old_z[k])
203 {
204 l = find_spec_id (coord -> species, old_z[k], new_z);
205 coord -> partial_geo[new_sp][j][l] = obj -> partial_geo[old_sp][old_geo][k];
206 }
207 }
208 }
209 coord -> ntg[gid][new_sp] ++;
210 coord -> totcoord[gid] ++;
211 if (! gid)
212 {
213 if (coord -> geolist[gid][new_sp] != NULL) g_free (coord -> geolist[gid][new_sp]);
214 coord -> geolist[gid][new_sp] = duplicate_int (coord -> ntg[gid][new_sp], tmpgeol);
215 g_free (tmpgeol);
216 }
217 return i;
218}

379void check_coord_modification (struct project * this_proj, int * old_id, struct atom * new_list,
380 struct insert_object * this_object, gboolean movtion, gboolean passivating)
381{
382 struct atom * tmp_new;
383 int g, h, i, j, k, l, m, n;
384 gboolean correct_it;
385 int * new_z = allocint (this_proj -> nspec);
386 double * old_z;
387 if (this_object) old_z = duplicate_double (this_proj -> nspec, this_proj -> chemistry -> chem_prop[CHEM_Z]);
388 int * new_old_id;
389 int * nvois = allocint (this_proj -> nspec);
390 if (passivating) new_old_id = duplicate_int (this_proj -> natomes, old_id);
391 for (i=0; i<this_proj -> nspec; i++)
392 {
393 new_z[i] = (int)this_proj -> chemistry -> chem_prop[CHEM_Z][i];
394 }
395
396 // first create a dummy coord structure to store an atom individual data
397 coord_info * new_coord = g_malloc0 (sizeof*new_coord);
398 for (i=0; i<2; i++)
399 {
400 new_coord -> totcoord[i] = 1;
401 new_coord -> ntg[i] = allocint (this_proj -> nspec);
402 for (j=0; j<this_proj -> nspec; j++) new_coord -> ntg[i][j] = 1;
403 new_coord -> geolist[i] = allocdint (this_proj -> nspec, 1);
404 }
405 new_coord -> species = this_proj -> nspec;
406 new_coord -> partial_geo = alloctint (this_proj -> nspec, 1, this_proj -> nspec);
407 g = (passivating) ? 2 : 1;
408 for (h=0; h<g; h++)
409 {
410 tmp_new = new_list;
411 while (tmp_new)
412 {
413 i = tmp_new -> sp;
414 // Fill the dummy coord with the new atom information
415 for (j=0; j<2; j++) new_coord -> geolist[j][i][0] = tmp_new -> numv;
416 k = tmp_new -> coord[1];
417 for (j=0; j<this_proj -> nspec; j++)
418 {
419 new_coord -> partial_geo[i][0][j] = this_proj -> coord -> partial_geo[i][k][j];
420 }
421 j = tmp_new -> id;
422 correct_it = FALSE;
423 if (movtion)
424 {
425 for (k=0; k<tmp_new -> numv; k++)
426 {
427 l = tmp_new -> vois[k];
428 if ((old_id[j] > 0 && old_id[l] < 0) || (old_id[j] < 0 && old_id[l] > 0))
429 {
430 correct_it = TRUE;
431 // This neighbor will be moved / removed
432 if (! passivating || h)
433 {
434 m = this_proj -> atoms[0][l].sp;
435 // For the atom studied reduce the number of neighbors of that species:
436 new_coord -> partial_geo[i][0][m] --;
437 for (n=0; n<2; n++) new_coord -> geolist[n][i][0] --;
438 }
439 }
440 }
441 }
442 else
443 {
444 for (k=0; k<this_proj -> nspec; k++) nvois[k]=0;
445 for (k=0; k<tmp_new -> numv; k++)
446 {
447 l = tmp_new -> vois[k];
448 m = this_object -> at_list[l].sp;
449 nvois[m] ++;
450 }
451 l = 0;
452 for (k=0; k<this_proj -> nspec; k++)
453 {
454 if (nvois[k] != new_coord -> partial_geo[i][0][k])
455 {
456 correct_it = TRUE;
457 new_coord -> partial_geo[i][0][k] = nvois[k];
458 }
459 l += nvois[k];
460 }
461 for (k=0; k<2; k++) new_coord -> geolist[k][i][0] = l;
462 }
463
464 if (correct_it)
465 {
466 if (passivating && ! h)
467 {
468 if (old_id[j])
469 {
470 switch (this_proj -> modelgl -> search_widg[8] -> filter)
471 {
472 case 0:
473 l = i;
474 break;
475 case 1:
476 l = tmp_new -> numv;
477 break;
478 case 2:
479 l = tmp_new -> coord[1];
480 for (m=0; m<i; m++) l += this_proj -> coord -> ntg[1][m];
481 break;
482 default:
483 l = tmp_new -> coord[this_proj -> modelgl -> search_widg[8] -> filter-1];
484 break;
485 }
486 if (get_insert_object_by_origin (this_proj -> modelgl -> atom_win -> to_be_inserted[3], l, 0)) new_old_id[j] = abs(old_id[j]);
487 }
488 }
489 else
490 {
491 for (j=0; j<2; j++)
492 {
493 if (this_object)
494 {
495 tmp_new -> coord[j] = find_this_geo_id (j, new_coord, new_z, 0, i, i, this_object -> coord, old_z);
496 }
497 else
498 {
499 tmp_new -> coord[j] = find_this_geo_id (j, new_coord, new_z, 0, i, i, this_proj -> modelgl -> atom_win -> coord, this_proj -> modelgl -> atom_win -> new_z);
500 }
501 }
502 }
503 correct_it = FALSE;
504 }
505 tmp_new = tmp_new -> next;
506 }
507 if (passivating && ! h)
508 {
509 for (i=0; i<this_proj -> natomes; i++) old_id[i] = new_old_id[i];
510 g_free (new_old_id);
511 }
512 }
513 if (this_object) g_free (old_z);
514 g_free (new_coord);
515 g_free (nvois);
516}
