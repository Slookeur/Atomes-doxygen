210void send_mol_neighbors_ (int * stp, int * mol, int * aid, int * nvs, int neigh[* nvs])
211{
212 int i, j, k, l, m, n, o, p, q, r, s, t, u;
213 struct search_molecule * tmp_mol = & in_calc_mol[* stp - 1][* mol - 1];
214 i = active_project -> atoms[* stp - 1][* aid - 1].sp;
215 j = active_project -> atoms[* stp - 1][* aid - 1].coord[1];
216 k = pgeo[i] + j;
217 for (l=0; l < * nvs; l++)
218 {
219 m = neigh[l]-1;
220 n = active_project -> atoms[tmp_mol -> md][m].sp;
221 o = active_project -> atoms[tmp_mol -> md][m].coord[1];
222 p = pgeo[n] + o;
223 tmp_mol -> pbonds[k][p] ++;
224 }
225 if (* nvs > 1)
226 {
227 for (l=0; l<*nvs-1; l++)
228 {
229 m = neigh[l]-1;
230 n = active_project -> atoms[tmp_mol -> md][m].sp;
231 o = active_project -> atoms[tmp_mol -> md][m].coord[1];
232 p = pgeo[n] + o;
233 for (q= l+1; q<*nvs; q++)
234 {
235 r = neigh[q]-1;
236 s = active_project -> atoms[tmp_mol -> md][r].sp;
237 t = active_project -> atoms[tmp_mol -> md][r].coord[1];
238 u = pgeo[s] + t;
239 tmp_mol -> pangles[p][k][u] ++;
240 tmp_mol -> pangles[u][k][p] ++;
241 }
242 }
243 }

255void update_mol_details (struct search_molecule * mol, int sp, int cp)
256{
257 int i, j, k, l, m;
258 for (i=0; i<active_project -> nspec; i++)
259 {
260 j = active_coord -> partial_geo[sp][cp][i];
261 mol -> nbonds += j;
262 if (j)
263 {
264 for (k=0; k<active_project -> nspec; k++)
265 {
266 l = active_coord -> partial_geo[sp][cp][k];
267 if (k != i || l > 1)
268 {
269 m = (j+l)*((j+l)-1)/2;
270 mol -> nangles += m;
271 }
272 }
273 }
274 }

289void send_mol_details_ (int * stp, int * mol, int * ats, int * sps, int spec_in_mol[* sps], int atom_in_mol[* ats])
290{
291 int i, j, k, l;
292 struct search_molecule * tmp_mol = & in_calc_mol[* stp - 1][* mol - 1];
293 tmp_mol -> id = * mol - 1;
294 tmp_mol -> md = * stp - 1;
295 tmp_mol -> multiplicity = 1;
296 tmp_mol -> fragments = allocint(1);
297 tmp_mol -> fragments[0] = * mol - 1;
298 tmp_mol -> natoms = * ats;
299 tmp_mol -> lgeo = allocdint (active_project -> nspec, active_coord -> totcoord[1]);
300 tmp_mol -> pbonds = allocdint (active_coord -> totcoord[1], active_coord -> totcoord[1]);
301 tmp_mol -> pangles = alloctint (active_coord -> totcoord[1], active_coord -> totcoord[1], active_coord -> totcoord[1]);
302 tmp_mol -> atoms = duplicate_int (* ats, atom_in_mol);
303 for (i=0; i< * ats; i++)
304 {
305 j = atom_in_mol[i]-1;
306 k = active_project -> atoms[0][j].sp;
307 l = active_project -> atoms[* stp - 1][j].coord[1];
308 tmp_mol -> lgeo[k][pgeo[k]+l] ++;
309 update_mol_details (tmp_mol, k, l);
310 }
311 j = 0;
312 for (i=0; i<active_project -> nspec; i++)
313 {
314 if (spec_in_mol[i] > 0) j++;
315 }
316 tmp_mol -> nspec = j;
317 tmp_mol -> species = duplicate_int (active_project -> nspec, spec_in_mol);
318 tmp_mol -> nbonds /= 2;
319 tmp_mol -> nangles /= 2;
320}

330gboolean are_identical_molecules (struct search_molecule * mol_a, struct search_molecule * mol_b)
331{
332 int i, j, k;
333/*#ifdef DEBUG
334 g_debug ("AID:: mol_a -> id= %d, mol_b-> id= %d", mol_a -> id, mol_b -> id);
335 g_debug ("AID:: mol_a -> natoms= %d, mol_b-> natoms= %d", mol_a -> natoms, mol_b -> natoms);
336 g_debug ("AID:: mol_a -> nspec= %d, mol_b-> nspec= %d", mol_a -> nspec, mol_b -> nspec);
337 g_debug ("AID:: mol_a -> nbonds= %d, mol_b-> nbonds= %d", mol_a -> nbonds, mol_b -> nbonds);
338 g_debug ("AID:: mol_a -> nangles= %d, mol_b-> nangles= %d", mol_a -> nangles, mol_b -> nangles);
339#endif*/
340 if (mol_a -> md != mol_b -> md) return FALSE;
341 if (mol_a -> natoms != mol_b -> natoms) return FALSE;
342 if (mol_a -> nspec != mol_b -> nspec) return FALSE;
343 for (i=0; i < active_project -> nspec; i++)
344 {
345 if (mol_a -> species[i] != mol_b -> species[i]) return FALSE;
346 }
347
348 if (mol_a -> nbonds != mol_b -> nbonds) return FALSE;
349 if (mol_a -> nangles != mol_b -> nangles) return FALSE;
350
351 for (i=0; i < active_project -> nspec; i++)
352 {
353 for (j=0; j< active_coord -> totcoord[1]; j++)
354 {
355 if (mol_a -> lgeo[i][j] != mol_b -> lgeo[i][j]) return FALSE;
356 }
357 }
358
359 if (mol_a -> nbonds > 1)
360 {
361 for (i=0; i < active_coord -> totcoord[1]; i++)
362 {
363 for (j=0; j < active_coord -> totcoord[1]; j++)
364 {
365 if (mol_a -> pbonds[i][j] != mol_b -> pbonds[i][j]) return FALSE;
366 }
367 }
368 if (mol_a -> nangles > 1)
369 {
370 for (i=0; i < active_coord -> totcoord[1]; i++)
371 {
372 for (j=0; j < active_coord -> totcoord[1]; j++)
373 {
374 for (k=0; k < active_coord -> totcoord[1]; k++)
375 {
376 if (mol_a -> pangles[i][j][k] != mol_b -> pangles[i][j][k]) return FALSE;
377 }
378 }
379 }
380 }
381 }
382 return TRUE;
383}

450void setup_molecules_ (int * stepid)
451{
452 int i, j, k, l, m, n;
453 struct search_molecule * mtmp_at, * mtmp_bt;
454 struct search_molecule * first_mol = NULL;
455 struct search_molecule * tmp_mol;
456 gboolean add_it;
457 i = * stepid -1;
458 j = 0;
459 for (k=0; k<active_project -> modelfc -> mol_by_step[i]; k++)
460 {
461 mtmp_bt = & in_calc_mol[i][k];
462 add_it = TRUE;
463 mtmp_at = first_mol;
464 while (mtmp_at)
465 {
466 if (are_identical_molecules (mtmp_at, mtmp_bt))
467 {
468 mtmp_at -> fragments = merge_mol_data (mtmp_at -> multiplicity, mtmp_bt -> multiplicity,
469 mtmp_at -> fragments, mtmp_bt -> fragments);
470 mtmp_at -> atoms = merge_mol_data (mtmp_at -> natoms*mtmp_at -> multiplicity, mtmp_bt -> natoms*mtmp_bt -> multiplicity,
471 mtmp_at -> atoms, mtmp_bt -> atoms);
472 mtmp_at -> multiplicity ++;
473 add_it = FALSE;
474 break;
475 }
476 mtmp_at = mtmp_at -> next;
477 }
478 if (add_it)
479 {
480 if (first_mol)
481 {
482 mtmp_at = first_mol;
483 while (mtmp_at -> next) mtmp_at = mtmp_at -> next;
484 mtmp_at -> next = duplicate_search_molecule (mtmp_bt);
485 mtmp_at -> next -> prev = mtmp_at;
486 }
487 else
488 {
489 first_mol = duplicate_search_molecule (mtmp_bt);
490 }
491 j ++;
492 }
493 free_search_molecule_data (mtmp_bt);
494 }
495 g_free (in_calc_mol[i]);
496 in_calc_mol[i] = NULL;
497 active_project -> modelfc -> mol_by_step[i] = j;
498 active_project -> modelfc -> mols[i] = g_malloc0(j*sizeof*active_project -> modelfc -> mols[i]);
499 tmp_mol = first_mol;
500 for (k=0; k<j; k++)
501 {
502 l = tmp_mol -> natoms*tmp_mol -> multiplicity;
503 for (m=0; m<l; m++)
504 {
505 n = tmp_mol -> atoms[m] - 1;
506 active_project -> atoms[i][n].coord[3] = k;
507 }
508 duplicate_molecule (& active_project -> modelfc -> mols[i][k], tmp_mol);
509 active_project -> modelfc -> mols[i][k].id = k;
510 free_search_molecule_data (tmp_mol);
511 if (k < j-1) tmp_mol = tmp_mol -> next;
512 }
513 while (tmp_mol -> prev)
514 {
515 tmp_mol = tmp_mol -> prev;
516 g_free (tmp_mol -> next);
517 }
518 g_free (first_mol);
519}
