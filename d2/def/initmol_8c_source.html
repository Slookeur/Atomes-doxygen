215void send_mol_neighbors_ (int * stp, int * mol, int * aid, int * nvs, int neigh[* nvs])
216{
217 int i, j, k, l, m, n, o, p, q, r, s, t, u;
218 struct search_molecule * tmp_mol = & in_calc_mol[* stp - 1][* mol - 1];
219 i = active_project -> atoms[* stp - 1][* aid - 1].sp;
220 j = active_project -> atoms[* stp - 1][* aid - 1].coord[1];
221 k = pgeo[i] + j;
222 for (l=0; l < * nvs; l++)
223 {
224 m = neigh[l]-1;
225 n = active_project -> atoms[tmp_mol -> md][m].sp;
226 o = active_project -> atoms[tmp_mol -> md][m].coord[1];
227 p = pgeo[n] + o;
228 tmp_mol -> pbonds[k][p] ++;
229 }
230 if (* nvs > 1)
231 {
232 for (l=0; l<*nvs-1; l++)
233 {
234 m = neigh[l]-1;
235 n = active_project -> atoms[tmp_mol -> md][m].sp;
236 o = active_project -> atoms[tmp_mol -> md][m].coord[1];
237 p = pgeo[n] + o;
238 for (q= l+1; q<*nvs; q++)
239 {
240 r = neigh[q]-1;
241 s = active_project -> atoms[tmp_mol -> md][r].sp;
242 t = active_project -> atoms[tmp_mol -> md][r].coord[1];
243 u = pgeo[s] + t;
244 tmp_mol -> pangles[p][k][u] ++;
245 tmp_mol -> pangles[u][k][p] ++;
246 }
247 }
248 }

260void update_mol_details (struct search_molecule * mol, int sp, int cp)
261{
262 int i, j, k, l, m;
263 for (i=0; i<active_project -> nspec; i++)
264 {
265 j = active_coord -> partial_geo[sp][cp][i];
266 mol -> nbonds += j;
267 if (j)
268 {
269 for (k=0; k<active_project -> nspec; k++)
270 {
271 l = active_coord -> partial_geo[sp][cp][k];
272 if (k != i || l > 1)
273 {
274 m = (j+l)*((j+l)-1)/2;
275 mol -> nangles += m;
276 }
277 }
278 }
279 }

294void send_mol_details_ (int * stp, int * mol, int * ats, int * sps, int spec_in_mol[* sps], int atom_in_mol[* ats])
295{
296 int i, j, k, l;
297 struct search_molecule * tmp_mol = & in_calc_mol[* stp - 1][* mol - 1];
298 tmp_mol -> id = * mol - 1;
299 tmp_mol -> md = * stp - 1;
300 tmp_mol -> multiplicity = 1;
301 tmp_mol -> fragments = allocint(1);
302 tmp_mol -> fragments[0] = * mol - 1;
303 tmp_mol -> natoms = * ats;
304 tmp_mol -> lgeo = allocdint (active_project -> nspec, active_coord -> totcoord[1]);
305 tmp_mol -> pbonds = allocdint (active_coord -> totcoord[1], active_coord -> totcoord[1]);
306 tmp_mol -> pangles = alloctint (active_coord -> totcoord[1], active_coord -> totcoord[1], active_coord -> totcoord[1]);
307 tmp_mol -> atoms = duplicate_int (* ats, atom_in_mol);
308 for (i=0; i< * ats; i++)
309 {
310 j = atom_in_mol[i]-1;
311 k = active_project -> atoms[0][j].sp;
312 l = active_project -> atoms[* stp - 1][j].coord[1];
313 tmp_mol -> lgeo[k][pgeo[k]+l] ++;
314 update_mol_details (tmp_mol, k, l);
315 }
316 j = 0;
317 for (i=0; i<active_project -> nspec; i++)
318 {
319 if (spec_in_mol[i] > 0) j++;
320 }
321 tmp_mol -> nspec = j;
322 tmp_mol -> species = duplicate_int (active_project -> nspec, spec_in_mol);
323 tmp_mol -> nbonds /= 2;
324 tmp_mol -> nangles /= 2;
325}

335gboolean are_identical_molecules (struct search_molecule * mol_a, struct search_molecule * mol_b)
336{
337 int i, j, k;
338/*#ifdef DEBUG
339 g_debug ("AID:: mol_a -> id= %d, mol_b-> id= %d", mol_a -> id, mol_b -> id);
340 g_debug ("AID:: mol_a -> natoms= %d, mol_b-> natoms= %d", mol_a -> natoms, mol_b -> natoms);
341 g_debug ("AID:: mol_a -> nspec= %d, mol_b-> nspec= %d", mol_a -> nspec, mol_b -> nspec);
342 g_debug ("AID:: mol_a -> nbonds= %d, mol_b-> nbonds= %d", mol_a -> nbonds, mol_b -> nbonds);
343 g_debug ("AID:: mol_a -> nangles= %d, mol_b-> nangles= %d", mol_a -> nangles, mol_b -> nangles);
344#endif*/
345 if (mol_a -> md != mol_b -> md) return FALSE;
346 if (mol_a -> natoms != mol_b -> natoms) return FALSE;
347 if (mol_a -> nspec != mol_b -> nspec) return FALSE;
348 for (i=0; i < active_project -> nspec; i++)
349 {
350 if (mol_a -> species[i] != mol_b -> species[i]) return FALSE;
351 }
352
353 if (mol_a -> nbonds != mol_b -> nbonds) return FALSE;
354 if (mol_a -> nangles != mol_b -> nangles) return FALSE;
355
356 for (i=0; i < active_project -> nspec; i++)
357 {
358 for (j=0; j< active_coord -> totcoord[1]; j++)
359 {
360 if (mol_a -> lgeo[i][j] != mol_b -> lgeo[i][j]) return FALSE;
361 }
362 }
363
364 if (mol_a -> nbonds > 1)
365 {
366 for (i=0; i < active_coord -> totcoord[1]; i++)
367 {
368 for (j=0; j < active_coord -> totcoord[1]; j++)
369 {
370 if (mol_a -> pbonds[i][j] != mol_b -> pbonds[i][j]) return FALSE;
371 }
372 }
373 if (mol_a -> nangles > 1)
374 {
375 for (i=0; i < active_coord -> totcoord[1]; i++)
376 {
377 for (j=0; j < active_coord -> totcoord[1]; j++)
378 {
379 for (k=0; k < active_coord -> totcoord[1]; k++)
380 {
381 if (mol_a -> pangles[i][j][k] != mol_b -> pangles[i][j][k]) return FALSE;
382 }
383 }
384 }
385 }
386 }
387 return TRUE;
388}

455void setup_molecules_ (int * stepid)
456{
457 int i, j, k, l, m, n;
458 struct search_molecule * mtmp_at, * mtmp_bt;
459 struct search_molecule * first_mol = NULL;
460 struct search_molecule * tmp_mol;
461 gboolean add_it;
462 i = * stepid -1;
463 j = 0;
464 for (k=0; k<active_project -> modelfc -> mol_by_step[i]; k++)
465 {
466 mtmp_bt = & in_calc_mol[i][k];
467 add_it = TRUE;
468 mtmp_at = first_mol;
469 while (mtmp_at)
470 {
471 if (are_identical_molecules (mtmp_at, mtmp_bt))
472 {
473 mtmp_at -> fragments = merge_mol_data (mtmp_at -> multiplicity, mtmp_bt -> multiplicity,
474 mtmp_at -> fragments, mtmp_bt -> fragments);
475 mtmp_at -> atoms = merge_mol_data (mtmp_at -> natoms*mtmp_at -> multiplicity, mtmp_bt -> natoms*mtmp_bt -> multiplicity,
476 mtmp_at -> atoms, mtmp_bt -> atoms);
477 mtmp_at -> multiplicity ++;
478 add_it = FALSE;
479 break;
480 }
481 mtmp_at = mtmp_at -> next;
482 }
483 if (add_it)
484 {
485 if (first_mol)
486 {
487 mtmp_at = first_mol;
488 while (mtmp_at -> next) mtmp_at = mtmp_at -> next;
489 mtmp_at -> next = duplicate_search_molecule (mtmp_bt);
490 mtmp_at -> next -> prev = mtmp_at;
491 }
492 else
493 {
494 first_mol = duplicate_search_molecule (mtmp_bt);
495 }
496 j ++;
497 }
498 free_search_molecule_data (mtmp_bt);
499 }
500 g_free (in_calc_mol[i]);
501 in_calc_mol[i] = NULL;
502 active_project -> modelfc -> mol_by_step[i] = j;
503 active_project -> modelfc -> mols[i] = g_malloc0(j*sizeof*active_project -> modelfc -> mols[i]);
504 tmp_mol = first_mol;
505 for (k=0; k<j; k++)
506 {
507 l = tmp_mol -> natoms*tmp_mol -> multiplicity;
508 for (m=0; m<l; m++)
509 {
510 n = tmp_mol -> atoms[m] - 1;
511 active_project -> atoms[i][n].coord[3] = k;
512 }
513 duplicate_molecule (& active_project -> modelfc -> mols[i][k], tmp_mol);
514 active_project -> modelfc -> mols[i][k].id = k;
515 free_search_molecule_data (tmp_mol);
516 if (k < j-1) tmp_mol = tmp_mol -> next;
517 }
518 while (tmp_mol -> prev)
519 {
520 tmp_mol = tmp_mol -> prev;
521 g_free (tmp_mol -> next);
522 }
523 g_free (first_mol);
524}
