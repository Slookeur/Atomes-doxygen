103object_3d * draw_cylinder (int quality, float ra, float rb)
104{
105 int i, j;
106 object_3d * new_cylinder = g_malloc0 (sizeof*new_cylinder);
107 new_cylinder -> quality = quality;
108 new_cylinder -> num_vertices = cylinder_vertices (quality);
109 new_cylinder -> vert_buffer_size = 3;
110 new_cylinder -> vertices = allocfloat (3*new_cylinder -> num_vertices);
111 new_cylinder -> num_indices = cylinder_indices (quality);
112 new_cylinder -> ind_buffer_size = 1;
113 new_cylinder -> indices = allocint (cylinder_indices (quality));
114
115 float step = 2.0 * pi / (quality-1);
116 float x, y;
117 j = 0;
118 // Vertex
119 for(i = 0; i < quality; i++)
120 {
121 x = cos (i*step);
122 y = sin (i*step);
123 new_cylinder -> vertices[j] = ra*x;
124 new_cylinder -> vertices[j+1] = ra*y;
125 new_cylinder -> vertices[j+2] = 0.5;
126 j += 3;
127 new_cylinder -> vertices[j] = rb*x;
128 new_cylinder -> vertices[j+1] = rb*y;
129 new_cylinder -> vertices[j+2] = -0.5;
130 j += 3;
131 }
132
133 for (i = 0; i < 2*quality; i++)
134 {
135 new_cylinder -> indices[i] = i % (2*quality-2);
136 }
137
138 return new_cylinder;
139}

174object_3d * draw_cylinder_cap (int quality, float rad, gboolean picked)
175{
176 int i, j;
177
178 object_3d * new_cap = g_malloc0 (sizeof*new_cap);
179 new_cap -> quality = quality;
180 new_cap -> num_vertices = cap_vertices(quality);
181 new_cap -> vert_buffer_size = 3;
182 new_cap -> vertices = allocfloat (3*new_cap -> num_vertices);
183 new_cap -> num_indices = cap_indices(quality);
184 new_cap -> ind_buffer_size = 1;
185 new_cap -> indices = allocint (new_cap -> num_indices);
186
187 float delta = (picked) ? 0.05 : 0.0;
188 float step = 2.0 * pi / (quality - 1);
189 float x, y;
190 j = 0;
191 // Center
192 new_cap -> vertices[j] = 0.0;
193 new_cap -> vertices[j+1] = 0.0;
194 new_cap -> vertices[j+2] = 0.0 - delta;
195 j += 3;
196 for(i = 0; i < quality; i++)
197 {
198 x = cos (i*step);
199 y = sin (i*step);
200 new_cap -> vertices[j] = rad*x;
201 new_cap -> vertices[j+1] = rad*y;
202 new_cap -> vertices[j+2] = 0.0 - delta;
203 j += 3;
204 }
205
206 for (i=0; i <= quality; i++)
207 {
208 new_cap -> indices[i] = i % (quality + 1);
209 }
210 return new_cap;
211}

221vec4_t rotate_bond (vec3_t a, vec3_t b)
222{
223 vec3_t c, f, raxis;
224 c = v3_sub (vec3(b.x, b.y, b.z), vec3(a.x, a.y, a.z));
225 f = vec3 (0.0, 0.0, 1.0);
226 double rangle;
227 raxis = v3_cross (f, c);
228 //rangle = 180.0 / pi * acos(v3_dot(f, c) / v3_length(c));
229 rangle = acos(v3_dot(f, c) / v3_length(c));
230 return axis_to_quat (raxis, rangle);
231}

296void setup_cylinder_vertice (float * vertices, vec3_t pos_a, vec3_t pos_b, ColRGBA col, float rad, float alpha, float delta)
297{
298 int s = nbs*CYLI_BUFF_SIZE;
299 vertices[s] = (pos_a.x + pos_b.x)/2.0;
300 vertices[s+1] = (pos_a.y + pos_b.y)/2.0;
301 vertices[s+2] = (pos_a.z + pos_b.z)/2.0;
302 vertices[s+3] = v3_length(v3_sub(pos_a, pos_b)) + delta;
303 vertices[s+4] = rad;
304 vec4_t quat = rotate_bond (pos_b, pos_a);
305 vertices[s+5] = quat.w;
306 vertices[s+6] = quat.x;
307 vertices[s+7] = quat.y;
308 vertices[s+8] = quat.z;
309 vertices[s+9] = col.red;
310 vertices[s+10] = col.green;
311 vertices[s+11] = col.blue;
312 vertices[s+12] = col.alpha * alpha;
313 nbs ++;
314}

365void setup_this_bond (int sty, gboolean to_pick, gboolean picked, int cap, int bi, int pi, struct atom * at, struct atom * bt, float al, float * vertices)
366{
367 float alpha = 1.0;
368 float delta = 0.0;
369 float shift[3];
370 int p, q, r;
371 vec3_t pos_a, pos_b;
372 ColRGBA col = get_atom_color (at -> sp, at -> id, 1.0, (picked) ? pi + 1 : 0, to_pick);
373 float rad = get_bond_radius ((to_pick) ? BALL_AND_STICK : (sty == NONE) ? plot -> style : sty, bi, at -> sp, bt -> sp, (picked) ? 1.0 : 0.0);
374 if (! cap && picked)
375 {
376 gboolean show_a, show_b;
377 int sta, stb;
378 if (in_movie_encoding && plot -> at_data != NULL)
379 {
380 show_a = plot -> at_data[bt -> id].show[bi];
381 show_b = plot -> at_data[bt -> id].pick[pi];
382 sta = plot -> at_data[at -> id].style;
383 stb = plot -> at_data[bt -> id].style;
384 }
385 else
386 {
387 show_a = bt -> show[bi];
388 show_b = bt -> pick[pi];
389 sta = at -> style;
390 stb = bt -> style;
391 }
392 delta = ((show_a && show_b) && (sta == stb)) ? 0.0 : 0.1;
393 }
394 for (p=0; p<plot -> extra_cell[0]+1;p++)
395 {
396 for (q=0; q<plot -> extra_cell[1]+1; q++)
397 {
398 for (r=0; r<plot -> extra_cell[2]+1; r++)
399 {
400 shift[0]=p*box_gl -> vect[0][0]+q*box_gl -> vect[1][0]+r*box_gl -> vect[2][0];
401 shift[1]=p*box_gl -> vect[0][1]+q*box_gl -> vect[1][1]+r*box_gl -> vect[2][1];
402 shift[2]=p*box_gl -> vect[0][2]+q*box_gl -> vect[1][2]+r*box_gl -> vect[2][2];
403 at_shift (at, shift);
404 at_shift (bt, shift);
405 pos_a = vec3(at -> x, at -> y, at -> z);
406 pos_b = vec3((at -> x + bt -> x)/2.0, (at -> y + bt -> y)/2.0, (at -> z + bt -> z)/2.0);
407 if (to_pick || ((sty == NONE && (plot -> style == BALL_AND_STICK || plot -> style == CYLINDERS)) || sty == BALL_AND_STICK || sty == CYLINDERS))
408 {
409 if (cap)
410 {
411 setup_cap_vertice (vertices, pos_a, pos_b, col, rad, alpha * al);
412 }
413 else
414 {
415 setup_cylinder_vertice (vertices, pos_a, pos_b, col, rad, (to_pick) ? 1.0 : alpha*al, delta);
416 }
417 }
418 else
419 {
420 setup_line_vertice (vertices, pos_a, col, alpha*al);
421 setup_line_vertice (vertices, pos_b, col, alpha*al);
422 }
423 at_unshift (at, shift);
424 at_unshift (bt, shift);
425 alpha = 0.5;
426 }
427 }
428 }

447void prepare_bond (int sty, gboolean to_pick, gboolean picked, int cap, int bi, int pi, int bid, struct atom * at, struct atom * bt, float * vertices)
448{
449 if (bi == 0)
450 {
451 setup_this_bond (sty, to_pick, picked, cap, bi, pi, at, bt, 1.0, vertices);
452 }
453 else
454 {
455 struct atom * tmp_a, * tmp_b;
456 float x, y, z;
457 int sign;
458 sign = 1;
459 if (wingl -> bondid[step][1][bid][0] == at -> id) sign = -1;
460 x = wingl -> clones[step][bid].x;
461 y = wingl -> clones[step][bid].y;
462 z = wingl -> clones[step][bid].z;
463 tmp_a = duplicate_atom (at);
464 tmp_b = duplicate_atom (at);
465 tmp_a -> pick[pi] = bt -> pick[pi];
466 tmp_a -> style = bt -> style;
467 tmp_a -> sp = bt -> sp + proj_sp;
468 tmp_b -> sp += proj_sp;
469 tmp_a -> x += sign * x;
470 tmp_a -> y += sign * y;
471 tmp_a -> z += sign * z;
472 setup_this_bond (sty, to_pick, picked, cap, bi, pi, tmp_b, tmp_a, 0.5, vertices);
473 g_free (tmp_a);
474 g_free (tmp_b);
475
476 tmp_a = duplicate_atom (bt);
477 tmp_b = duplicate_atom (bt);
478 tmp_a -> pick[pi] = at -> pick[pi];
479 tmp_a -> style = at -> style;
480 tmp_a -> sp = at -> sp + proj_sp;
481 tmp_b -> sp += proj_sp;
482 tmp_a -> id = at -> id;
483 tmp_a -> x -= sign * x;
484 tmp_a -> y -= sign * y;
485 tmp_a -> z -= sign * z;
486 setup_this_bond (sty, to_pick, picked, cap, bi, pi, tmp_a, tmp_b, 0.5, vertices);
487 g_free (tmp_a);
488 g_free (tmp_b);
489 }

504int find_bond_vertices (gboolean to_pick, int sty, int sa, int sb, int bi, int cap)
505{
506 int i, j, k, l, m, n;
507 gboolean show_a, show_b;
508 l = 0;
509 for (i=0; i < wingl -> bonds[step][bi]; i++)
510 {
511 j = wingl -> bondid[step][bi][i][0];
512 k = wingl -> bondid[step][bi][i][1];
513 if (in_movie_encoding && plot -> at_data != NULL)
514 {
515 show_a = plot -> at_data[j].show[bi];
516 m = plot -> at_data[j].style;
517 show_b = plot -> at_data[k].show[bi];
518 n = plot -> at_data[k].style;
519 }
520 else
521 {
522 show_a = proj_gl -> atoms[step][j].show[bi];
523 m = proj_gl -> atoms[step][j].style;
524 show_b = proj_gl -> atoms[step][k].show[bi];
525 n = proj_gl -> atoms[step][k].style;
526 }
527 if (proj_gl -> atoms[step][j].sp == sa && proj_gl -> atoms[step][k].sp == sb)
528 {
529 if (show_a && (m == sty || to_pick))
530 {
531 if (cap)
532 {
533 if (! show_b || n != sty) l += 1 + bi;
534 }
535 else
536 {
537 l += 1 + bi;
538 }
539 }
540 }
541 if (proj_gl -> atoms[step][j].sp == sb && proj_gl -> atoms[step][k].sp == sa)
542 {
543 if (show_b && (n == sty || to_pick))
544 {
545 if (cap)
546 {
547 if (! show_a || m != sty) l += 1 + bi;
548 }
549 else
550 {
551 l += 1 + bi;
552 }
553 }
554 }
555 }
556 return 2*l;
557}

675int create_bond_lists (gboolean to_pick)
676{
677 int nshaders = 0;
678 int **** nbonds;
679 int **** ncaps;
680 int nbds[7];
681 int ncap[7];
682 object_3d * cyl, * cap;
683 int f, g, h, i, j, k, l, m;
684
685 if (! to_pick)
686 {
687 cleaning_shaders (wingl, BONDS);
688 wingl -> create_shaders[BONDS] = FALSE;
689 }
690
691 g = (plot -> draw_clones) ? 2 : 1;
692
693 nbonds = allocqint (NUM_STYLES, g, proj_sp, proj_sp);
694 if (! to_pick) ncaps = allocqint (NUM_STYLES, g, proj_sp, proj_sp);
695
696 for (f=0; f<NUM_STYLES; f++)
697 {
698 nbds[f] = ncap[f] = 0;
699 k = l = m = 0;
700 if (to_pick || (! f && (plot -> style != SPHERES && plot -> style != PUNT)) || (f && f-1 != SPHERES && f-1 != PUNT))
701 {
702 for (h=0; h<g; h++)
703 {
704 for (i=0; i<proj_sp; i++)
705 {
706 for (j=0; j<proj_sp; j++)
707 {
708 nbonds[f][h][i][j] = find_bond_vertices (to_pick, f-1, i, j, h, 0);
709 k += nbonds[f][h][i][j];
710 if (nbonds[f][h][i][j] > 0) l ++;
711 if (! to_pick && ((! f && (plot -> style == BALL_AND_STICK || plot -> style == CYLINDERS)) || (f && (f-1 == BALL_AND_STICK || f-1 == CYLINDERS))))
712 {
713 ncaps[f][h][i][j] = find_bond_vertices (to_pick, f-1, i, j, h, 1);
714 m += ncaps[f][h][i][j];
715 }
716 }
717 }
718 }
719 nbds[f] = k;
720 ncap[f] = m;
721 if (to_pick || (((! f && (plot -> style == BALL_AND_STICK || plot -> style == CYLINDERS)) || (f && (f-1 == BALL_AND_STICK || f-1 == CYLINDERS)))))
722 {
723 if (k > 0)
724 {
725 nshaders ++;
726 if (m > 0 && ! to_pick) nshaders ++;
727 }
728 }
729 else
730 {
731 nshaders += l;
732 }
733 }
734 if (to_pick) break;
735 }
736#ifdef DEBUG
737 g_debug ("Bond LIST:: to_pick= %s, shaders= %d", (to_pick) ? "true" : "false", nshaders);
738#endif
739 if (nshaders == 0) return nshaders;
740 if (! to_pick) wingl -> ogl_glsl[BONDS][step] = g_malloc0 (nshaders*sizeof*wingl -> ogl_glsl[BONDS][step]);
741 l = 0;
742 for (f=0; f<NUM_STYLES; f++)
743 {
744 if (nbds[f])
745 {
746 if (to_pick || (! f && (plot -> style == BALL_AND_STICK || plot -> style == CYLINDERS)) || (f && (f-1 == BALL_AND_STICK || f-1 == CYLINDERS)))
747 {
748 cyl = draw_cylinder (plot -> quality, 1.0, 1.0);
749 cyl -> num_instances = (nbds[f]/2) * (plot -> extra_cell[0]+1)*(plot -> extra_cell[1]+1)*(plot -> extra_cell[2]+1);
750 cyl -> inst_buffer_size = CYLI_BUFF_SIZE;
751 cyl -> instances = allocfloat (CYLI_BUFF_SIZE*cyl -> num_instances);
752 nbs = 0;
753 for (h=0; h<g; h++)
754 {
755 setup_all_cylinder_vertices (f-1, to_pick, 0, h, cyl -> instances);
756 if (to_pick && h==0) wingl -> bonds_to_be_picked = wingl -> to_be_picked;
757 }
758 if (! to_pick)
759 {
760 wingl -> ogl_glsl[BONDS][step][l] = init_shader_program (BONDS, GLSL_CYLINDERS, cylinder_vertex, NULL, full_color, GL_TRIANGLE_STRIP, 6, 1, TRUE, cyl);
761 g_free (cyl);
762 l ++;
763 if (ncap[f] > 0)
764 {
765 cap = draw_cylinder_cap (plot -> quality, 1.0, FALSE);
766 cap -> num_instances = (ncap[f]/2) * (plot -> extra_cell[0]+1)*(plot -> extra_cell[1]+1)*(plot -> extra_cell[2]+1);
767 cap -> inst_buffer_size = CAPS_BUFF_SIZE;
768 cap -> instances = allocfloat (CAPS_BUFF_SIZE*cap -> num_instances);
769 nbs = 0;
770 for (h=0; h<g; h++)
771 {
