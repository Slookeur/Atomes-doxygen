75double get_object_dim (struct insert_object * object)
76{
77 double dmax = 0.0;
78 vec3_t at, bt;
79 vec3_t dist;
80 int i, j;
81 for (i=0; i<object -> atoms-1; i++)
82 {
83 at = vec3 (object -> at_list[i].x, object -> at_list[i].y, object -> at_list[i].z);
84 for (j=i+1; j<object -> atoms; j++)
85 {
86 bt = vec3 (object -> at_list[j].x, object -> at_list[j].y, object -> at_list[j].z);
87 dist = v3_sub(at, bt);
88 dmax = max (dmax, v3_length(dist));
89 }
90 }
91 return dmax+1.0;
92}

102void correct_pos_and_get_dim (struct insert_object * object, gboolean adjust)
103{
104 int i;
105 if (object -> baryc) g_free(object -> baryc);
106 object -> baryc = allocdouble(3);
107 for (i=0; i<object -> atoms; i++)
108 {
109 object -> baryc[0] += object -> at_list[i].x;
110 object -> baryc[1] += object -> at_list[i].y;
111 object -> baryc[2] += object -> at_list[i].z;
112 }
113 for (i=0; i<3; i++) object -> baryc[i] /= object -> atoms;
114 if (adjust)
115 {
116 for (i=0; i<object -> atoms; i++)
117 {
118 object -> at_list[i].x -= object -> baryc[0];
119 object -> at_list[i].y -= object -> baryc[1];
120 object -> at_list[i].z -= object -> baryc[2];
121 }
122 }
123 object -> dim = get_object_dim (object);
124}

139gboolean rebuild_atom_neighbors (struct project * this_proj, int step, struct insert_object * object, int target, int aid, struct atom * at, gboolean * checked_at)
140{
141 int i, j;
142 struct distance dist;
143 for (i=0; i<at -> numv; i++)
144 {
145 j = at -> vois[i];
146 dist = distance_3d (& this_proj -> cell, step, at, & object -> at_list[j]);
147 if (dist.pbc && ! checked_at[j])
148 {
149 object -> at_list[j].x = at -> x - dist.x;
150 object -> at_list[j].y = at -> y - dist.y;
151 object -> at_list[j].z = at -> z - dist.z;
152 }
153 }
154 checked_at[aid] = TRUE;
155 i = 0;
156 for (j=0; j<object -> atoms; j++) if (checked_at[j]) i ++;
157 if (i == target) return TRUE;
158 for (i=0; i<at -> numv; i++)
159 {
160 j = at -> vois[i];
161 if (! checked_at[j])
162 {
163 if (rebuild_atom_neighbors (this_proj, step, object, target, j, & object -> at_list[j], checked_at)) return TRUE;
164 }
165 }
166 return FALSE;
167}

178void reconstruct_bonds (struct project * this_proj, int ifcl, int * bcid)
179{
180 int i, j, k, l;
181 int o_step = this_proj -> modelgl -> anim -> last -> img -> step;
182 int ** old_bid = NULL;
183 old_bid = allocdint (this_proj -> modelgl -> bonds[o_step][0], 2);
184 for (i=0; i<this_proj -> modelgl -> bonds[o_step][0]; i++)
185 {
186 for (j=0; j<2; j++) old_bid[i][j] = this_proj -> modelgl -> bondid[o_step][0][i][j];
187 }
188 g_free (this_proj -> modelgl -> bondid[o_step][0]);
189 this_proj -> modelgl -> bondid[o_step][0] = allocdint (this_proj -> modelgl -> bonds[o_step][0]+ ifcl, 2);
190 for (i=0; i<this_proj -> modelgl -> bonds[o_step][0]; i++)
191 {
192 for (j=0; j<2; j++) this_proj -> modelgl -> bondid[o_step][0][i][j] = old_bid[i][j];
193 }
194 if (old_bid) g_free (old_bid);
195 old_bid = NULL;
196
197 for (i=0; i<ifcl; i++)
198 {
199 j = bcid[i];
200 k = this_proj -> modelgl -> bonds[o_step][0];
201 for (l=0; l<2; l++)
202 {
203 this_proj -> modelgl -> bondid[o_step][0][k][l] = this_proj -> modelgl -> bondid[o_step][1][j][l];
204 }
205 this_proj -> modelgl -> bonds[o_step][0] ++;
206 this_proj -> modelgl -> allbonds[0] ++;
207 this_proj -> modelgl -> bondid[o_step][1][j][0] = -1;
208 }
209
210 vec3_t * old_clo = NULL;
211 i = this_proj -> modelgl -> bonds[o_step][1] - ifcl;
212 l = 0;
213 if (i)
214 {
215 old_clo = g_malloc0 (i*sizeof*old_clo);
216 old_bid = allocdint (i, 2);
217 for (i=0; i<this_proj -> modelgl -> bonds[o_step][1]; i++)
218 {
219 if (this_proj -> modelgl -> bondid[o_step][1][i][0] > -1)
220 {
221 for (k=0; k<2; k++) old_bid[l][k] = this_proj -> modelgl -> bondid[o_step][1][i][k];
222 old_clo[l].x = this_proj -> modelgl -> clones[o_step][i].x;
223 old_clo[l].y = this_proj -> modelgl -> clones[o_step][i].y;
224 old_clo[l].z = this_proj -> modelgl -> clones[o_step][i].z;
225 l ++;
226 }
227 }
228 }
229 g_free (this_proj -> modelgl -> bondid[o_step][1]);
230 this_proj -> modelgl -> bondid[o_step][1] = NULL;
231 g_free (this_proj -> modelgl -> clones[o_step]);
232 this_proj -> modelgl -> clones[o_step] = NULL;
233 this_proj -> modelgl -> bonds[o_step][1] = this_proj -> modelgl -> allbonds[1] = l;
234 if (l)
235 {
236 this_proj -> modelgl -> bondid[o_step][1] = allocdint (l, 2);
237 this_proj -> modelgl -> clones[o_step] = g_malloc0 (l*sizeof*this_proj -> modelgl -> clones[o_step]);
238 for (i=0; i<l; i++)
239 {
240 for (j=0; j<2; j++) this_proj -> modelgl -> bondid[o_step][1][i][j] = old_bid[i][j];
241 this_proj -> modelgl -> clones[o_step][i].x = old_clo[i].x;
242 this_proj -> modelgl -> clones[o_step][i].y = old_clo[i].y;
243 this_proj -> modelgl -> clones[o_step][i].z = old_clo[i].z;
244 }
245 if (old_clo) g_free (old_clo);
246 old_clo = NULL;
247 if (old_bid) g_free (old_bid);
248 old_bid = NULL;
249 }
250}

261void reconstruct_coordinates_for_object (struct project * this_proj, struct insert_object * this_object, gboolean upcoord)
262{
263 int h, i, j, k;
264 if (this_proj -> coord -> totcoord[2])
265 {
266 int * tmp_multi = allocint (this_proj -> coord -> totcoord[2]);
267 for (i=0; i<this_object -> atoms; i++)
268 {
269 j = this_object -> at_list[i].coord[2];
270 tmp_multi[j] = 1;
271 }
272 gboolean * checked_at = allocbool (this_object -> atoms);
273 h = (this_proj -> cell.npt) ? this_proj -> modelgl -> anim -> last -> img -> step : 0;
274 for (i=0; i<this_proj -> coord -> totcoord[2]; i++)
275 {
276 if (tmp_multi[i])
277 {
278 k = 0;
279 for (j=0; j<this_object -> atoms; j++)
280 {
281 checked_at[j] = FALSE;
282 if (this_object -> at_list[j].coord[2] == i)
283 {
284 k ++;
285 }
286 }
287 for (j=0; j<this_object -> atoms; j++)
288 {
289 if (this_object -> at_list[j].coord[2] == i)
290 {
291 rebuild_atom_neighbors (this_proj, h, this_object, k, j, & this_object -> at_list[j], checked_at);
292 }
293 }
294 }
295 }
296 g_free (checked_at);
297 }
298 correct_pos_and_get_dim (this_object, TRUE);
299 if (upcoord)
300 {
301 reconstruct_bonds (this_proj, this_object -> ifcl, this_object -> bcid);
302 for (i=0; i<this_object -> atoms; i++)
303 {
304 j = this_object -> at_list[i].id;
305 this_proj -> atoms[0][j].x = this_object -> at_list[i].x + this_object -> baryc[0];
306 this_proj -> atoms[0][j].y = this_object -> at_list[i].y + this_object -> baryc[1];
307 this_proj -> atoms[0][j].z = this_object -> at_list[i].z + this_object -> baryc[2];
308 this_proj -> atoms[0][j].cloned = FALSE;
309 }
310 for (i=0; i<3; i++) this_proj -> modelgl -> saved_coord[i] = save_coordinates (this_proj, i);
311 }
312}

352tint ulam_coord (glwin * view)
353{
354 tint pos;
355 int m;
356 pos.a = pos.b = pos.c = 0;
357 if (view -> builder_win) return pos;
358 int p = view -> nth_copy;
359 if (p > 0)
360 {
361 pos.c = p/9;
362 m = p - pos.c*9;
363 pos.b = m/3;
364 pos.a = p - pos.c*9 - pos.b*3;
365 }
366 return pos;
367}

379struct insert_object * duplicate_insert_object (struct insert_object * old_obj)
380{
381 struct insert_object * new_obj = g_malloc0 (sizeof*new_obj);
382 new_obj -> origin = old_obj -> origin;
383 new_obj -> type = old_obj -> type;
384 new_obj -> dim = old_obj -> dim;
385 new_obj -> name = g_strdup_printf ("%s", old_obj -> name);
386 new_obj -> atoms = old_obj -> atoms;
387 new_obj -> at_list = g_malloc0 (new_obj -> atoms*sizeof*new_obj -> at_list);
388 int i;
389 for (i=0; i<new_obj -> atoms; i++)
390 {
391 new_obj -> at_list[i] = * duplicate_atom (& old_obj -> at_list[i]);
392 }
393 new_obj -> old_z = duplicate_int (old_obj -> species, old_obj -> old_z);
394 new_obj -> coord = duplicate_coord_info (old_obj -> coord);
395 new_obj -> baryc = duplicate_double (3, old_obj -> baryc);
396
397 new_obj -> occ = old_obj -> occ;
398 new_obj -> species = old_obj -> species;
399 if (old_obj -> bonds)
400 {
401 new_obj -> ibonds = g_malloc0 (old_obj -> bonds*sizeof*new_obj -> ibonds);
402 int i;
403 for (i=0; i<old_obj -> bonds; i++)
404 {
405 new_obj -> ibonds[i] = duplicate_int (2, old_obj -> ibonds[i]);
406 }
407 }
408 new_obj -> bonds = old_obj -> bonds;
409 return new_obj;
410}

486{

489

504 }
505 }
506 if (j)
507 {
508 object -> at_list[i].vois = allocint (j);
509 j = 0;
510 for (k=0; k<object -> at_list[i].numv; k++)
511 {
512 l = tmpv[k];
513 if (new_id[l])
514 {

516 {
517 object -> at_list[i].vois[j] = new_id[l]-1;
518 j ++;
519 }
520 }
521 }
522 }
523 object -> at_list[i].numv = j;
524 g_free (tmpv);
525 }

540void clean_object_bonds (struct project * proj, int o_step, struct insert_object * object, int * new_id, gboolean movtion)
541{
542 int h, i, j, k, l, m;
543 gboolean doit;
544 int ** tmpibonds = allocdint (proj -> modelgl -> bonds[o_step][0]+proj -> modelgl -> bonds[o_step][1], 2);
545 if (proj -> modelgl -> bonds[o_step][1])
546 {
547 h = (movtion) ? 2 : 1;
548 }
549 else
550 {
551 h = 1;
552 }
553 i = 0;
554 if (new_id) object -> bcid = allocint (proj -> modelgl -> bonds[o_step][1]);
555 for (j=0; j<h; j++)
556 {
557 for (k=0; k<proj -> modelgl -> bonds[o_step][j]; k++)
558 {
559 l = proj -> modelgl -> bondid[o_step][j][k][0];
560 m = proj -> modelgl -> bondid[o_step][j][k][1];
561 doit = FALSE;
562 if (new_id)
563 {
564 if (new_id[l] && new_id[m]) doit = TRUE;
565 }
566 else
567 {
568 doit = TRUE;
569 }
570 if (doit)
571 {
572 if (new_id)
573 {
574 tmpibonds[i][0] = new_id[l] - 1;
575 tmpibonds[i][1] = new_id[m] - 1;
576 }
577 else
578 {
579 tmpibonds[i][0] = l;
580 tmpibonds[i][1] = m;
581 }
582 i ++;
583 if (j && new_id)
584 {
585 object -> bcid[object -> ifcl] = k;
586 object -> ifcl ++;
587 }
588 }
589 }
590 }
591 if (i)
592 {
593 object -> ibonds = allocdint (i, 2);
594 for (j=0; j<i; j++)
595 {
596 object -> ibonds[j][0] = tmpibonds[j][0];
597 object -> ibonds[j][1] = tmpibonds[j][1];
598 }
599 }
600 g_free (tmpibonds);
601 object -> bonds = i;
602 if (new_id) clean_object_vois (proj, object, new_id, movtion);
603}

619void add_object_atoms (struct insert_object * this_object, struct project * this_proj,
620 int o_step, int numa, int * old_id, gboolean check_bonding, atom_search * remove)
621{
622 int i, j;
623 this_object -> atoms = numa;
624 this_object -> at_list = g_malloc0 (this_object -> atoms*sizeof*this_object -> at_list);
625 int * new_id = allocint (this_proj -> natomes);
626 for (i=0; i<this_object -> atoms; i++)
627 {
628 j = old_id[i]-1;
629 new_id[j] = i+1;
630 if (remove) remove -> todo[j] = 1;
631 this_object -> at_list[i] = * duplicate_atom (& this_proj -> atoms[o_step][j]);
632 if (i)
633 {
634 this_object -> at_list[i].prev = & this_object -> at_list[i-1];
635 this_object -> at_list[i-1].next = & this_object -> at_list[i];
636 }
637 }
638 gboolean movtion = ((object_motion && this_proj -> modelgl -> rebuild[0][0]) || (! object_motion && this_proj -> modelgl -> rebuild[1][0]));
639 if (check_bonding)
640 {
641 clean_object_bonds (this_proj, o_step, this_object, new_id, movtion);
642 }
643 correct_coordinates_for_object (this_proj, this_object, movtion);
644 if (check_bonding)
645 {
646 check_coord_modification (this_proj, NULL, & this_object -> at_list[0], this_object, FALSE, FALSE);
647 }
648 g_free (old_id);
649 g_free (new_id);
650}
