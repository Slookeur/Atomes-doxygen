41int save_this_string (FILE * fp, gchar * string)
42{
43 int i;
44 if (string)
45 {
46 i = strlen (string);
47 if (i > 0)
48 {
49 i ++;
50 if (fwrite (& i, sizeof(int), 1, fp) != 1) return ERROR_RW;
51 if (fwrite (string, sizeof(char), i, fp) != i) return ERROR_RW;
52 }
53 else
54 {
55 i = 0;
56 if (fwrite (& i, sizeof(int), 1, fp) != 1) return ERROR_RW;
57 }
58 }
59 else
60 {
61 i = 0;
62 if (fwrite (& i, sizeof(int), 1, fp) != 1) return ERROR_RW;
63 }
64 return OK;
