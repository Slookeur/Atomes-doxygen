81double ** save_coordinates (struct project * this_proj, int status)
82{
83 int i, j;
84 i = 0;
85 if (status > 1)
86 {
87 i = this_proj -> natomes;
88 }
89 else
90 {
91 for (j=0; j<this_proj -> natomes; j++)
92 {
93 if (this_proj -> atoms[0][j].pick[0] == status) i++;
94 }
95 }
96 if (i == 0) return NULL;
97
98 double ** coords = allocddouble (i, 3);
99 i = 0;
100 if (status > 1)
101 {
102 for (i=0; i<this_proj -> natomes; i++)
103 {
104 coords[i][0] = this_proj -> atoms[0][i].x;
105 coords[i][1] = this_proj -> atoms[0][i].y;
106 coords[i][2] = this_proj -> atoms[0][i].z;
107 }
108 }
109 else
110 {
111 for (j=0; j<this_proj -> natomes; j++)
112 {
113 if (this_proj -> atoms[0][j].pick[0] == status)
114 {
115 coords[i][0] = this_proj -> atoms[0][j].x;
116 coords[i][1] = this_proj -> atoms[0][j].y;
117 coords[i][2] = this_proj -> atoms[0][j].z;
118 i ++;
119 }
120 }
121 }
122 return coords;
123}

161{

164 float i = 0.0;

259void rotate_quat (struct project * this_proj, vec4_t q, int status, int axis)
260{
261 int j;
262 mat4_t rot = m4_quat_rotation (q);
263 vec3_t c_old, c_new;
264 for (j=0; j<this_proj -> natomes; j++)
265 {
266 if (this_proj -> atoms[0][j].pick[0] == status)
267 {
268 c_old = vec3(this_proj -> atoms[0][j].x, this_proj -> atoms[0][j].y, this_proj -> atoms[0][j].z);
269 c_new = v3_sub(c_old, this_proj -> modelgl -> baryc[status]);
270 if (axis)
271 {
272 c_old = m4_mul_pos (this_proj -> modelgl -> view_matrix, c_new);
273 c_new = m4_mul_pos (rot, c_old);
274 c_old = m4_mul_pos (this_proj -> modelgl -> un_view_matrix, c_new);
275 }
276 else
277 {
278 c_old = m4_mul_pos (rot, c_new);
279 }
280 c_new = v3_add (c_old, this_proj -> modelgl -> baryc[status]);
281 this_proj -> atoms[0][j].x = c_new.x;
282 this_proj -> atoms[0][j].y = c_new.y;
283 this_proj -> atoms[0][j].z = c_new.z;
284 }
285 }

318void random_move_this_atom (struct project * this_proj, int aid)
319{
320 int i, j, k, l;
321 // Using CPU time to randomize
322 clock_t begin = clock();
323 double prob;
324 i = (int)begin;
325 for (j=0; j<3; j++)
326 {
327 k = (j+1)*i*aid*(3*this_proj -> numwid);
328 prob= random3_(& k);
329 k *= k;
330 l = (prob <= 0.5) ? 1 : -1;
331 prob = random3_(& k);
332 switch (j)
333 {
334 case 0:
335 this_proj -> atoms[0][aid].x += l*prob*sqrt(this_proj -> modelgl -> atom_win -> msd[aid]/3.0);
336 break;
337 case 1:
338 this_proj -> atoms[0][aid].y += l*prob*sqrt(this_proj -> modelgl -> atom_win -> msd[aid]/3.0);
339 break;
340 case 2:
341 this_proj -> atoms[0][aid].z += l*prob*sqrt(this_proj -> modelgl -> atom_win -> msd[aid]/3.0);
342 break;
343 }
344 }

357void random_rotate_this_object (struct project * this_proj, struct insert_object * object, double ratio, double msd)
358{
359 int i, j, k, l, m, n;
360 vec3_t c_new, c_old;
361 vec3_t ax[3];
362 ax[0] = vec3(1.0,0.0,0.0);
363 ax[1] = vec3(0.0,1.0,0.0);
364 ax[2] = vec3(0.0,0.0,1.0);
365 vec3_t baryc = vec3(object -> baryc[0], object -> baryc[1], object -> baryc[2]);
366 vec4_t qr;
367 mat4_t rot;
368 // Using CPU time to randomize
369 clock_t begin = clock();
370 double prob;
371 i = (int)begin;
372 for (j=0; j<3; j++)
373 {
374 k = (j+1)*i*(3*this_proj -> numwid);
375 prob=random3_(& k);
376 k *= k;
377 l = (prob <= 0.5) ? 1 : -1;
378 prob = random3_(& k);
379 qr = axis_to_quat(ax[j], l*prob*sqrt(ratio*msd/3.0)*pi/90.0);
380 rot = m4_quat_rotation (qr);
381 for (m=0; m<object -> atoms; m++)
382 {
383 n = object -> at_list[m].id;
384 if (! was_moved_atom[n])
385 {
386 c_old = vec3(object -> at_list[m].x, object -> at_list[m].y, object -> at_list[m].z);
387 c_new = m4_mul_pos (rot, c_old);
388 object -> at_list[m].x = c_new.x;
389 object -> at_list[m].y = c_new.y;
390 object -> at_list[m].z = c_new.z;
391 c_old = v3_add (c_new, baryc);
392 this_proj -> atoms[0][n].x = c_old.x;
393 this_proj -> atoms[0][n].y = c_old.y;
394 this_proj -> atoms[0][n].z = c_old.z;
395 }
396 }
397 }

410void random_translate_this_object (struct project * this_proj, struct insert_object * object, double ratio, double msd)
411{
412 int i, j, k, l, m, n;
413 // Using CPU time to randomize
414 clock_t begin = clock();
415 double prob;
416 i = (int)begin;
417 for (j=0; j<3; j++)
418 {
419 k = (j+1)*i*(3*this_proj -> numwid);
420 prob=random3_(& k);
421 k *= k;
422 l = (prob <= 0.5) ? 1 : -1;
423 prob = l*random3_(& k)*sqrt(ratio*msd/3.0);
424 switch (j)
425 {
426 case 0:
427 for (m=0; m<object -> atoms; m++)
428 {
429 n = object -> at_list[m].id;
430 if (! was_moved_atom[n]) this_proj -> atoms[0][n].x += prob;
431 }
432 break;
433 case 1:
434 for (m=0; m<object -> atoms; m++)
435 {
436 n = object -> at_list[m].id;
437 if (! was_moved_atom[n]) this_proj -> atoms[0][n].y += prob;
438 }
439 break;
440 case 2:
441 for (m=0; m<object -> atoms; m++)
442 {
443 n = object -> at_list[m].id;
444 if (! was_moved_atom[n]) this_proj -> atoms[0][n].z += prob;
445 }
446 break;
447 }
448 object -> baryc[j] += prob;
449 }
450}

462void random_move_this_object (struct project * this_proj, struct insert_object * object, int move, double msd)
463{
464 int i, j;
465 // Using CPU time to randomize
466 clock_t begin = clock();
467 double prob;
468 i = (int)begin;
469 switch (move)
470 {
471 case 1:
472 random_translate_this_object (this_proj, object, 1.0, msd);
473 break;
474 case 2:
475 random_rotate_this_object (this_proj, object, 1.0, msd);
476 break;
477 case 3:
478 j = object -> species*i*object -> atoms*(3*this_proj -> numwid);
479 prob = random3_(& j);
480 random_translate_this_object (this_proj, object, prob, msd);
481 random_rotate_this_object (this_proj, object, 1.0-prob, msd);
482 break;
483 }
484}

511gboolean rebuild_selection (struct project * this_proj, atom_search * asearch, int filter)
512{
513 int i, j, k, l;
514 gboolean was_frag = this_proj -> modelgl -> adv_bonding[0];
515 gboolean was_mol = this_proj -> modelgl -> adv_bonding[1];
516 gboolean recons = FALSE;
517 int old_frag = this_proj -> coord -> totcoord[2];
518 i = (asearch -> action == DISPL) ? 0 : 1;
519 if (asearch -> object && this_proj -> modelgl -> rebuild[0][i])
520 {
521 int * saved_todo = duplicate_int (asearch -> todo_size, asearch -> todo);
522 int old_tds = asearch -> todo_size;
523 g_free (asearch -> todo);
524 allocate_todo (asearch, this_proj -> natomes);
525 j = 0;
526 int * oifcl = NULL;
527 struct insert_object * object = this_proj -> modelgl -> atom_win -> to_be_moved[i];
528 while (object)
529 {
530 j += object -> ifcl;
531 object = object -> next;
532 }
533 if (j)
534 {
535 object = this_proj -> modelgl -> atom_win -> to_be_moved[i];
536 oifcl = allocint (j);
537 l = 0;
538 while (object)
539 {
540 for (k=0; k<object -> ifcl; k++)
541 {
542 oifcl[l] = object -> bcid[k];
543 l ++;
544 }
545 object = object -> next;
546 }
547 }
548 if (j)
549 {
550 reconstruct_bonds (this_proj, j, oifcl);
551 g_free (oifcl);
552 oifcl = NULL;
553 }
554 object = this_proj -> modelgl -> atom_win -> to_be_moved[i];
555 while (object)
556 {
557 if (object -> ifcl) reconstruct_coordinates_for_object (this_proj, object, TRUE);
558 if (filter < 3)
559 {
560 recons = TRUE;
561 for (j=0; j<object -> atoms; j++)
562 {
563 k = object -> at_list[j].id;
564 asearch -> todo[k] = 1;
565 }
566 }
567 object = object -> next;
568 }
569 if (recons)
570 {
571 apply_action (this_proj, asearch);
572 trigger_refresh (this_proj, asearch);
573 }
574 g_free (asearch -> todo);
575 asearch -> todo = duplicate_int (old_tds, saved_todo);
576 asearch -> todo_size = old_tds;
577 g_free (saved_todo);
578 }
579 if ((was_frag && ! this_proj -> modelgl -> adv_bonding[0]) || (was_mol && ! this_proj -> modelgl -> adv_bonding[1]))
580 {
581 return TRUE;
582 }
583 else if (old_frag != this_proj -> coord -> totcoord[2])
584 {
585 return TRUE;
586 }
587 else
588 {
589 return FALSE;
590 }
591}
