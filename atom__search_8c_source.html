177G_MODULE_EXPORT void set_atoms_for_action (GtkComboBox * box, gpointer data)
178{
179 tint * id = (tint *)data;

182 i = gtk_combo_box_get_active (box);

185 {
186 for (j=0; j<2; j++)
187 {
188 gtk_combo_box_set_active (GTK_COMBO_BOX(this_proj -> modelgl -> atom_win -> axis_combo[j]), this_proj -> modelgl -> atom_win -> axis[j]);
189 for (k=0; k<6; k++)
190 {
191 gtk_range_set_value (GTK_RANGE(this_proj -> modelgl -> atom_win -> edit_scale[k]), this_proj -> modelgl -> atom_win -> new_param[i][j][k]);
192 update_range_and_entry (this_proj, i, this_proj -> modelgl -> atom_win -> axis[j], k);
193 }
194 }
195 }
196 clean_picked_and_labelled (this_proj -> modelgl -> search_widg[id -> c], TRUE);
197 update_search_tree (this_proj -> modelgl -> search_widg[id -> c]);
