88G_MODULE_EXPORT void show_hide_coord (GSimpleAction * action, GVariant * parameter, gpointer data)
89#else
98G_MODULE_EXPORT void show_hide_coord (GtkWidget * widg, gpointer data)
99#endif
100{
101 qint * cid = (qint *)data;
102 int i, j, k;
103 int c, g, s;
104 gboolean doit = TRUE;
105 gboolean show;
106 struct project * this_proj = get_project_by_id(cid -> a);
107 s = cid -> b;
108 c = cid -> c;
109 g = cid -> d;
110 j = 0;
111 if (g < 2)
112 {
113 for (i=0; i<s; i++)
114 {
115 j += this_proj -> coord -> ntg[g][i];
116 }
117 }
118 j += c;
119#ifdef GTK3
120 show = (widg) ? gtk_check_menu_item_get_active ((GtkCheckMenuItem *)widg) : this_proj -> modelgl -> anim -> last -> img -> show_coord[g][j];
121#else
122 GVariant * state;
123 if (action)
124 {
125 state = g_action_get_state (G_ACTION (action));
126 show = ! g_variant_get_boolean (state);
127 const gchar * coord = g_action_get_name ((GAction *)action);
128 int lgt = strlen (coord);
129 gchar * name = g_strdup_printf ("%c%c", coord[lgt-2], coord[lgt-1]);
130 if (g_strcmp0(name, ".1") == 0)
131 {
132 name = g_strdup_printf ("%.*s.0", lgt-2, coord);
133 g_action_group_activate_action ((GActionGroup *)this_proj -> modelgl -> action_group, (const gchar *)name, NULL);
134 g_free (name);
135 doit = FALSE;
136 }
137 }
138 else
139 {
140 show = this_proj -> modelgl -> anim -> last -> img -> show_coord[g][j];
141 }
142#endif
143#ifdef DEBUG
144 g_debug ("SHOW_HIDE_COORD:: p= %d, s= %d, c= %d, g= %d, show= %d", this_proj -> id, s, c, g, show);
145#endif
146
147#ifdef GTK3
148 // GTK3 Menu Action To Check
149 if (is_coord_in_menu(g, this_proj))
150 {
151 for (i=0; i<2; i++)
152 {
153 if (widg != this_proj -> modelgl -> ogl_geom[i][g][j])
154 {
155 if (GTK_IS_WIDGET(this_proj -> modelgl -> ogl_geom[i][g][j]))
156 {
157 if (gtk_check_menu_item_get_active ((GtkCheckMenuItem *)this_proj -> modelgl -> ogl_geom[i][g][j]) != show)
158 {
159 gtk_check_menu_item_set_active ((GtkCheckMenuItem *)this_proj -> modelgl -> ogl_geom[i][g][j], show);
160 }
161 }
162 }
163 }
164 }
165#endif
166 if (doit)
167 {
168 this_proj -> modelgl -> anim -> last -> img -> show_coord[g][j] = show;
169 for (i=0; i<this_proj -> steps; i++)
170 {
171 for (j=0; j<this_proj -> natomes; j++)
172 {
173 if (g < 2)
174 {
175 if (this_proj -> atoms[i][j].sp == s && this_proj -> atoms[i][j].coord[g] == c)
176 {
177 this_proj -> atoms[i][j].show[0] = this_proj -> atoms[i][j].show[1] = show;
178 }
179 }
180 else if (g < 4)
181 {
182 if (this_proj -> atoms[i][j].coord[g] == c)
183 {
184 this_proj -> atoms[i][j].show[0] = this_proj -> atoms[i][j].show[1] = show;
185 }
186 }
187 else if (g < 9)
188 {
189 k = this_proj -> coord -> geolist[g][0][c] - 1;
190 if (this_proj -> atoms[i][j].rings[s][k] != NULL)
191 {
192 if (this_proj -> atoms[i][j].rings[s][k][0])
193 {
194 this_proj -> atoms[i][j].show[0] = this_proj -> atoms[i][j].show[1] = show;
195 }
196 }
197 }
198 else
199 {
200 k = this_proj -> coord -> geolist[g][0][c] - 1;
201 if (this_proj -> atoms[i][j].chain[k] != NULL)
202 {
203 if (this_proj -> atoms[i][j].chain[k][0])
204 {
205 this_proj -> atoms[i][j].show[0] = this_proj -> atoms[i][j].show[1] = show;
206 }
207 }
208 }
209 }
210 }
211 init_default_shaders (this_proj -> modelgl);
212#ifdef GTK4
213 if (action)
214 {
215 g_action_change_state (G_ACTION (action), g_variant_new_boolean (show));
216 g_variant_unref (state);
217 }
218#endif
219 }
220}

490GMenu * menu_show_coord (glwin * view, int popm, int id, int mid)
491{
492 GMenu * menu = g_menu_new ();
493 GMenu * menus;
494 struct project * this_proj = get_project_by_id (view -> proj);
495 gchar * stra, * strb;
496 int i, j, k;
497 for (i=0; i<this_proj -> nspec; i++)
498 {
499 menus = g_menu_new ();
500 if (this_proj -> coord)
501 {
502 if (this_proj -> coord -> ntg[id])
503 {
504 j = 0;
505 for (k=0; k<i; k++)
506 {
507 j += this_proj -> coord -> ntg[id][k];
508 }
509 for (k=0; k < this_proj -> coord -> ntg[id][i]; k++)
510 {
511 if (id)
512 {
513 stra = exact_name (env_name (this_proj, k, i, 1, NULL));
514 }
515 else
516 {
517 stra = g_strdup_printf ("%d", this_proj -> coord -> geolist[id][i][k]);
518 }
519 if (! mid)
520 {
521 strb = g_strdup_printf ("%s-s", stra);
522 append_opengl_item (view, menus, stra, strb, popm, k+j, NULL, IMG_NONE, NULL, FALSE, G_CALLBACK(show_hide_coord), & view -> gcid[id][k+j][id],
523 TRUE, view -> anim -> last -> img -> show_coord[id][k+j], FALSE, TRUE);
524 }
525 else
526 {
527 strb = g_strdup_printf ("%s-c", stra);
528 append_submenu (menus, stra, color_item(view, strb, popm, k+j, G_CALLBACK(window_color_coord), & view -> gcid[id][k+j][id]));
529 }
530 g_free (stra);
531 g_free (strb);
532 }
533 }
534 }
535 append_submenu (menu, this_proj -> chemistry -> label[i], menus);
536 g_object_unref (menus);
537 }
538 return menu;
