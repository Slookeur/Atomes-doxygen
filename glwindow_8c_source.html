491 if (view -> anim -> last -> img -> color_map[col] < i-1)

496 }

502 {

507 j = col*ATOM_MAPS + view -> anim -> last -> img -> color_map[col] + 1;
508 }
509 else
510 {
511 j = col*ATOM_MAPS + ((view -> custom_map) ? 6 : 0);
512 }
513 }
514 else
515 {
516 j = col*ATOM_MAPS;
517 }
518 gboolean was_input = reading_input;

520#ifdef GTK3

522 set_color_map (view -> color_styles[j], & view -> colorp[j][0]);

528 reading_input = was_input;

543void set_motion (glwin * view, int axis, int da, int db, gboolean UpDown, GdkModifierType state)
544{
545 if (state & GDK_CONTROL_MASK)
546 {
547 if (view -> mode == EDITION)
548 {
549 vec3_t trans;
550 if (axis)
551 {
552 trans = vec3(view -> anim -> last -> img -> p_depth*(double) da / view -> pixels[axis], 0.0, 0.0);
553 }
554 else
555 {
556 trans = vec3(0.0, view -> anim -> last -> img -> p_depth*(double) da / view -> pixels[axis], 0.0);
557 }
558 translate (get_project_by_id(view -> proj), 1, 1, trans);
559 init_default_shaders (view);
560 update (view);
561 }
562 else if (state & GDK_SHIFT_MASK)
563 {
564 spin_go (NULL, & view -> colorp[db][axis]);
565 }
566 else
567 {
568 view -> anim -> last -> img -> c_shift[! axis] += (double) da / view -> pixels[axis];
569 if (view -> camera_widg[! axis + 5])
570 {
571 if (GTK_IS_WIDGET(view -> camera_widg[! axis + 5]))
572 {
573 gtk_spin_button_set_value ((GtkSpinButton *)view -> camera_widg[! axis + 5], - view -> anim -> last -> img -> c_shift[! axis]);
574 }
575 }
576 update (view);
577 }
578 }
579 else if ((state & GDK_SHIFT_MASK) && UpDown)
580 {
581 if (db == 1 && view -> anim -> last -> img -> zoom > ZOOM_MAX) zoom (view, -1);
582 if (db == 3) zoom (view, 1);
583 update (view);
584 }
585 else if (state != GDK_SHIFT_MASK)
586 {
587 save_rotation_quaternion (view);
588 double cameraAngle[2] = {0.0, 0.0};
589 cameraAngle[axis] = 0.5*da;

602mat4_t insert_projection (glwin * view)
603{
604 GLdouble w, h;
605 GLdouble dw, dh;
606 double gleft, gright, gbottom, gtop;
607 double zoom = view -> anim -> last -> img -> zoom;
608 zoom *= (view -> anim -> last -> img -> p_depth / view -> anim -> last -> img -> gnear);
609 dw = view -> anim -> last -> img -> c_shift[0]*2.0*zoom;
610 dh = view -> anim -> last -> img -> c_shift[1]*2.0*zoom;
611 double aspect = (double)view -> pixels[0] / (double)view -> pixels[1];
612 if (aspect > 1.0)
613 {
614 w = zoom * aspect;
615 h = zoom;
616 }
617 else
618 {
619 w = zoom;
620 h = zoom / aspect;
621 }
622 gleft = -w + dw;
623 gright = w + dw;
624 gbottom = -h + dh;
625 gtop = h + dh;
626 return m4_ortho (gleft, gright, gbottom, gtop, -view -> anim -> last -> img -> gfar, view -> anim -> last -> img -> gfar);
627}
