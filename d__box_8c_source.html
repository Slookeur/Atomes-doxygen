94gboolean not_in_already (vec3_t a, vec3_t b, float * vertices)
95{
96 int i, j, k;
97 vec3_t tma, tmb;
98 for (i=0; i<nbs/2; i=i+2)
99 {
100 j = i*BOX_BUFF_SIZE;
101 k = j+BOX_BUFF_SIZE;
102 tma = vec3(vertices[j], vertices[j+1], vertices[j+2]);
103 tmb = vec3(vertices[k], vertices[k+1], vertices[k+2]);
104 if (are_identical_vec3(a, tma) && are_identical_vec3(b, tmb))
105 {
106 return FALSE;
107 }
108 else if (are_identical_vec3(b, tma) && are_identical_vec3(a, tmb))
109 {
110 return FALSE;
111 }
112 }
113 return TRUE;
114}

281{

292
293 if (plot -> box_axis[BOX] == WIREFRAME)

423void prepare_cuboid (vec3_t position, int id)
424{
425 float lgt = 1.0;

427 light -> vert_buffer_size = POLY_BUFF_SIZE;
428 light -> num_vertices = 36;
429 light -> vertices = allocfloat (light -> vert_buffer_size*light -> num_vertices);
430 int i, j, k, l;
431 l = 0;
432 for (i=0; i<36; i++)
433 {
434 j = 10*l;
435 k = 6*l;
436 light -> vertices[j] = lgt*cuboid_vertices[k]*0.5 + position.x;
437 light -> vertices[j+1] = lgt*cuboid_vertices[k+1]*0.5 + position.y;

439 light -> vertices[j+3] = cuboid_vertices[k+3];
440 light -> vertices[j+4] = cuboid_vertices[k+4];
441 light -> vertices[j+5] = cuboid_vertices[k+5];
442 light -> vertices[j+6] = 1.0;
443 light -> vertices[j+7] = 1.0;
444 light -> vertices[j+8] = 1.0;
445 light -> vertices[j+9] = 1.0;
446 l ++;

448 wingl -> ogl_glsl[LIGHT][0][id] = init_shader_program (LIGHT, GLSL_POLYEDRA, full_vertex, NULL, full_color, GL_TRIANGLES, 3, 1, FALSE, light);

513double draw_cuboid (gboolean draw, int SHADID, int shadnum, mat4_t rot, vec3_t cpos, double paral[3][3], ColRGBA col, double slab_alpha)
514{
515 int i, j, k, l, m, n, o;
516 vec3_t pos;
517 double cvol;
518 double shift[3];
519 cvol = (paral[0][1]*paral[1][2]-paral[0][2]*paral[1][1])*paral[2][0];
520 cvol += (paral[0][2]*paral[1][0]-paral[0][0]*paral[1][2])*paral[2][1];
521 cvol += (paral[0][0]*paral[1][1]-paral[0][1]*paral[1][0])*paral[2][2];
522 cvol = fabs(cvol);
523 if (draw)
524 {
525 object_3d * slab = g_malloc0 (sizeof*slab);
526 slab -> vert_buffer_size = POLY_BUFF_SIZE;
527 slab -> num_vertices = 36*(plot -> extra_cell[0]+1)*(plot -> extra_cell[1]+1)*(plot -> extra_cell[2]+1);
528 slab -> vertices = allocfloat (slab -> vert_buffer_size*slab -> num_vertices);
529 for (i=0; i<36; i++)
530 {
531 j = i*POLY_BUFF_SIZE;
532 k = 6*i;
533 for (l=0; l<3; l++)
534 {
535 slab -> vertices[j+l] += 0.5*(cuboid_vertices[k]*paral[0][l] + cuboid_vertices[k+1]*paral[1][l] + cuboid_vertices[k+2]*paral[2][l]);
536 }
537 pos = vec3 (slab -> vertices[j], slab -> vertices[j+1], slab -> vertices[j+2]);
538 pos = m4_mul_pos (rot, pos);
539 slab -> vertices[j] = pos.x + cpos.x;
540 slab -> vertices[j+1] = pos.y + cpos.y;
541 slab -> vertices[j+2] = pos.z + cpos.z;
542 }
543 n = 0;
544 for (i=0; i<plot -> extra_cell[0]+1; i++)
545 {
