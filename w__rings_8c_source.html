55G_MODULE_EXPORT void on_select_rings (GtkCellRendererToggle * cell_renderer, gchar * string_path, gpointer data)
56{
57 tint * dat = (tint * )data;
58 gboolean saved_label[2];
59 int i, j, g, c, s, v, u, a, b;
60 opengl_project_changed(dat -> a);
61 coord_edition * coord = opengl_project -> modelgl -> coord_win;
62 b = (opengl_project -> steps > 1) ? 1: 0;
63 for (i=0; i<5; i++)
64 {
65 if (i*10 > dat -> b) break;
66 }
67 g = i - 1;
68 for (i=2+b; i<5+b; i++)
69 {
70 if (10*g+i == dat -> b) break;
71 }
72 c = i;
73 if (gtk_cell_renderer_toggle_get_active(cell_renderer))
74 {
75 v = 0;
76 }
77 else
78 {
79 v = 1;
80 }
81 GtkTreeIter iter;
82 GtkTreePath * path = gtk_tree_path_new_from_string (string_path);
83 gtk_tree_model_get_iter (GTK_TREE_MODEL(coord -> rings_model[g]), & iter, path);
84 gtk_tree_store_set (coord -> rings_model[g], & iter, c, v, -1);
85 if (b)
86 {
87 gtk_tree_model_get (GTK_TREE_MODEL(coord -> rings_model[g]), & iter, 0, & s, -1);
88 s = - s - 1;
89 }
90 else
91 {
92 s = 0;
93 }
94 gtk_tree_model_get (GTK_TREE_MODEL(coord -> rings_model[g]), & iter, b, & i, -1);
95 i = -i;
96 gtk_tree_model_get (GTK_TREE_MODEL(coord -> rings_model[g]), & iter, b+1, & j, -1);
97 switch (c-b)
98 {
99 case 2:
100 // Viz
101 for (u=0; u<i; u++)
102 {
103 a = opengl_project -> modelgl -> all_rings[g][s][i-1][j-1][u];
104#ifdef GTK4
105 if (opengl_project -> atoms[s][a].show[0] != v) show_hide_this_atom (NULL, NULL, GINT_TO_POINTER(a));
106#else
107 if (opengl_project -> atoms[s][a].show[0] != v) show_hide_this_atom (NULL, GINT_TO_POINTER(a));
108#endif // GTK4
109 }
110 break;
111 case 3:
112 // Show poly
113 opengl_project -> modelgl -> show_rpoly[g][s][i-1][j-1] = v;
114 int shaders[1] = {RINGS};
115 re_create_md_shaders (1, shaders, opengl_project);
116 break;
117 case 4:
118 // Label
119 for (u=0; u<i; u++)
120 {
121 a = opengl_project -> modelgl -> all_rings[g][s][i-1][j-1][u];
122 opengl_project -> atoms[s][a].label[0] = opengl_project -> atoms[s][a].label[1] = v;
123 }
124 opengl_project -> modelgl -> create_shaders[LABEL] = TRUE;
125 break;
126 case 5:
127 // Pick
128 for (u=0; u<i; u++)
129 {
130 a = opengl_project -> modelgl -> all_rings[g][s][i-1][j-1][u];
131 saved_label[0] = opengl_project -> atoms[s][a].label[0];
132 saved_label[1] = opengl_project -> atoms[s][a].label[1];
133#ifdef GTK4
134 if (opengl_project -> atoms[s][a].pick[0] != v) select_unselect_this_atom (NULL, NULL, GINT_TO_POINTER(a));
135#else
136 if (opengl_project -> atoms[s][a].pick[0] != v) select_unselect_this_atom (NULL, GINT_TO_POINTER(a));
137#endif // GTK4
138 opengl_project -> atoms[s][a].label[0] = saved_label[0];
139 opengl_project -> atoms[s][a].label[1] = saved_label[1];
140 }
141 break;
142 }
143 update (opengl_project -> modelgl);
144}

284 1, 0,

300 5, FALSE,
301 6, FALSE, -1);
302 }
303 else
304 {
305 gtk_tree_store_set (store, & ring_level, 0, -j,
306 1, l+1,
307 2, FALSE,
308 3, this_proj -> modelgl -> show_rpoly[rid][h][j-1][l],
309 4, FALSE,
310 5, FALSE, -1);
311 }
312 }

338 coord -> rings_model[rid] = gtk_tree_store_newv (6+j, col_type);
339 if (fill_this) fill_rings_model (coord -> rings_model[rid], this_proj, rid);
340 GtkWidget * rings_tree = gtk_tree_view_new_with_model (GTK_TREE_MODEL(coord -> rings_model[rid]));
341 for (i=0; i<6+j; i++)
342 {
343 if (i < 2+j)
344 {
345 rings_cell[i] = gtk_cell_renderer_text_new ();
346 }
347 else
348 {
349 rings_cell[i] = gtk_cell_renderer_toggle_new ();
350 g_signal_connect (G_OBJECT(rings_cell[i]), "toggled", G_CALLBACK(on_select_rings), & this_proj -> modelgl -> colorp[rid*10+i][0]);
351 }
352 gtk_cell_renderer_set_fixed_size (rings_cell[i], -1, 25);
353 rings_col[i] = gtk_tree_view_column_new_with_attributes (ctitle[i+k], rings_cell[i], ctype[i+k], i, NULL);
354 gtk_tree_view_append_column(GTK_TREE_VIEW(rings_tree), rings_col[i]);
355 gtk_tree_view_column_set_alignment (rings_col[i], 0.5);
356 gtk_tree_view_column_set_cell_data_func (rings_col[i], rings_cell[i], rings_set_visible, & this_proj -> modelgl -> colorp[i][0], NULL);
357 }
358 return rings_tree;
