258G_MODULE_EXPORT void update_cp2k_option (GtkEntry * res, gpointer data)
259{
260 int i;
261 i = GPOINTER_TO_INT(data);
262 const gchar * m = entry_get_text (res);
263 double v = atof(m);
264 switch (i)
265 {
266 case CP2VDW:
267 if (v > 0.0) tmp_cp2k -> extra_opts[0][2] = v;
268 update_entry_double (res, tmp_cp2k -> extra_opts[0][2]);
269 break;
270 case CP2ROK:
271 if (v > 0.0) tmp_cp2k -> extra_opts[1][0] = v;
272 update_entry_double (res, tmp_cp2k -> extra_opts[1][0]);
273 break;
274 default:
275 if ((int)v == -1 || (int)v > 0) tmp_cp2k -> extra_opts[2][i-CP2POR] = v;
276 update_entry_int (res, tmp_cp2k -> extra_opts[2][i-CP2POR]);
277 break;
278 }
279 print_start_buffer ();
280}
