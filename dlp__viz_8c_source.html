83void field_selection (int i, int viz, int lab, int aid)
84{
85 if (viz && tmp_proj -> atoms[0][i].pick[0] != viz)
86 {
87 tmp_view -> picked ++;
88 }
89 else if (! viz && tmp_proj -> atoms[0][i].pick[0] != viz)
90 {
91 tmp_view -> picked --;
92 }
93 tmp_proj -> atoms[0][i].pick[0] = viz;
94 tmp_proj -> atoms[0][i].label[0] = lab;
95 tmp_proj -> atoms[0][i].coord[4] = aid;
96}

359int get_field_objects (int id, int jd)
360{
361 int kd;
362 int i;
363 switch (id)
364 {
365 case -1:
366 kd = 0;
367 for (i=0; i<tmp_field -> molecules; i++) kd = max(kd, get_active_field_molecule (i) -> mol -> natoms);
368 break;
369 case 0:
370 kd = tmp_field -> molecules;
371 break;
372 case 1:
373 kd = 0;
374 for (i=0; i<tmp_field -> molecules; i++) kd = max(kd, get_active_field_molecule (i) -> atoms);
375 break;
376 case 2:
377 kd = get_active_field_molecule (jd) -> shells;
378 break;
379 case 3:
380 kd = get_active_field_molecule (jd) -> constraints;
381 break;
382 case 4:
383 kd = get_active_field_molecule (jd) -> pmfs;
384 break;
385 case 5:
386 kd = get_active_field_molecule (jd) -> rigids;
387 break;
388 case 6:
389 kd = get_active_field_molecule (jd) -> tethered;
390 break;
391 case SEXTERN:
392 kd = tmp_field -> extern_fields;
393 break;
394 default:
395 if (id < MOLIMIT)
396 {
397 kd = get_active_field_molecule (jd) -> nstruct[id-7];
398 }
399 else
400 {
401 kd = tmp_field -> nbody[id - MOLIMIT];
402 }
403 break;
404 }
405 return kd;

417void select_object (int id, int jd, int kd)
418{
419 int i;
420 field_object = id;
421 num_field_objects = get_field_objects (id, kd);
422 switch (id)
423 {
424 case 2:
425 // Select a core-shell unit in a molecule
426 tmp_fshell = get_active_shell (kd, jd);
427 if (tmp_fshell -> ia[0] < 0 || tmp_fshell -> ia[1] < 0) toviz.c = FALSE;
428 tmp_fshell -> use = toviz.c;
429 break;
430 case 3:
431 // Select a contraint bond in a molecule
432 tmp_fcons = get_active_constraint (kd, jd);
433 if (! tmp_fcons -> ia[0] || ! tmp_fcons -> ia[1]) toviz.c = FALSE;
434 tmp_fcons -> use = toviz.c;
435 break;
436 case 4:
437 // Select a PMF in a molecule
438 tmp_fpmf = get_active_field_molecule (kd) -> first_pmf;
439 for (i=0; i < get_active_field_molecule(kd) -> pmfs; i++)
440 {
441 tmp_fpmf -> use = FALSE;
442 if (tmp_fpmf -> next != NULL) tmp_fpmf = tmp_fpmf -> next;
443 }
444 tmp_fpmf = get_active_pmf (kd, jd);
445 if (tmp_fpmf -> num[0] > 0 && tmp_fpmf -> num[1] > 0) tmp_fpmf -> use = toviz.c;
446 break;
447 case SEXTERN:
448 tmp_fext = tmp_field -> first_external;
449 for (i=0; i < tmp_field -> extern_fields; i++)
450 {
451 tmp_fext -> use = FALSE;
452 if (tmp_fext -> next != NULL) tmp_fext = tmp_fext -> next;
453 }
454 tmp_fext = get_active_external (jd);
455 tmp_fext -> use = toviz.c;
456 break;
457 default:
458 if (id < MOLIMIT)
459 {
460 get_active_struct (id-7, kd, jd) -> def -> use = toviz.c;
461 }
462 else
463 {
464 get_active_body (jd, id - MOLIMIT) -> use = toviz.c;
465 }
466 break;
467 }

480void visualize_single_struct (int id, int jd, int kd, int * ids)
481{
482 int i, j, k, l, m, n, o, p;
483 int * idat;
484 i = struct_id(id);
485 idat = allocint (i);
486 for (j=0; j<i; j++)
487 {
488 k = ids[j];
489 idat[j] = get_active_atom (kd, tmp_fmol -> atoms_id[k][0].a) -> id;
490 }
491 for (j=0; j<i; j++)
492 {
493 k = ids[j];
494 for (l=0; l< tmp_fmol -> multi; l++)
495 {
496 m = tmp_fmol -> atoms_id[k][l].b;
497 n = get_active_atom (kd, tmp_fmol -> atoms_id[k][l].a) -> list[m];
498 field_selection (n, 0, 0, k);
499 }
500 }
501 g_free (idat);
502 i = ids[0];
503 j = ids[1];
504 for (l=0; l< tmp_fmol -> multi; l++)
505 {
506 m = tmp_fmol -> atoms_id[i][l].b;
507 n = get_active_atom (kd, tmp_fmol -> atoms_id[i][l].a) -> list[m];
508 field_selection (n, toviz.c, 0, jd);
509 o = tmp_fmol -> atoms_id[j][l].b;
510 p = get_active_atom (kd, tmp_fmol -> atoms_id[j][l].a) -> list[o];
511 field_selection (p, toviz.c, 0, jd);
512 }
513 if (id > 8 && id < 15)
514 {
515 k = ids[2];
516 for (l=0; l< tmp_fmol -> multi; l++)
517 {
518 o = tmp_fmol -> atoms_id[k][l].b;
519 p = get_active_atom (kd, tmp_fmol -> atoms_id[k][l].a) -> list[o];
520 field_selection (p, toviz.c, 0, jd);
521 }
522 if (id > 10)
523 {
524 i = ids[3];
525 for (l=0; l< tmp_fmol -> multi; l++)
526 {
527 o = tmp_fmol -> atoms_id[i][l].b;
528 p = get_active_atom (kd, tmp_fmol -> atoms_id[i][l].a) -> list[o];
529 field_selection (p, toviz.c, 0, jd);
530 }
531 }

544void visualize_object (int id, int jd, int kd)
545{
546 int l, m, n, o, p, q, r, s;
547 struct field_atom * tshell;
548 field_object = id;
549 num_field_objects = get_field_objects (id, kd);
550 switch (id)
551 {
552 case -1:
553 // View molecule atoms id
554 tmp_fmol = get_active_field_molecule (jd);
555 tmp_fmol -> show_id = toviz.c;
556 tmp_fat = tmp_fmol -> first_atom;
557 for (m=0; m < tmp_fmol -> atoms; m++)
558 {
559 for (n=0; n < tmp_fat -> num; n++)
560 {
561 field_selection (tmp_fat -> list[n], toviz.c, toviz.c, tmp_fat -> list_id[n]);
562 }
563 if (tmp_fat -> next != NULL) tmp_fat = tmp_fat -> next;
564 }
565 break;
566 case 0:
567 // View an entire molecule
568 tmp_fmol = get_active_field_molecule (jd);
569 tmp_fmol -> show = toviz.c;
570 for (m=0; m<tmp_fmol -> multi; m++)
571 {
572 toviz.a = 2;
573 toviz.b = tmp_fmol -> fragments[m];
574 for (n=0; n<tmp_proj -> natomes; n++)
575 {
576 if (tmp_proj -> atoms[0][n].coord[2] == toviz.b) field_selection (n, toviz.c, toviz.c, jd);
577 }
578 }
579 break;
580 case 1:
581 // View a type of atoms in a molecule
582 tmp_fat = get_active_atom (kd, jd);
583 tmp_fat -> show = toviz.c;
584 for (m=0; m<tmp_fat -> num; m++)
585 {
586 n = tmp_fat -> list[m];
587 field_selection (n, toviz.c, toviz.c, jd);
588 }
589 break;
590 case 2:
591 // View a type of Core-shell unit in a molecule
592 tmp_fshell = get_active_shell (kd, jd);
593 tmp_fshell -> show = toviz.c;
594 for (m=0; m<2; m++)
595 {
596 if (tmp_fshell -> ia[m])
597 {
598 n = tmp_fshell -> ia[0]-1;
599 tshell = get_active_atom (tmp_fmol -> id, tmp_fmol -> atoms_id[0][n].a);
600 for (o=0; o<tmp_fmol -> multi; o++)
601 {
602 p = tmp_fmol -> atoms_id[n][o].b;
603 field_selection (tshell -> list[p], toviz.c, toviz.c, jd);
604 }
605 }
606 }
607 break;
608 case 3:
609 // View a type of bond constraint in a molecule
610 tmp_fcons = get_active_constraint (kd, jd);
611 tmp_fcons -> show = toviz.c;
612 for (l=0; l<2; l++)
613 {
614 if (tmp_fcons -> ia[l])
615 {
616 for (m=0; m<tmp_fmol -> multi; m++)
617 {
618 n = tmp_fmol -> atoms_id[tmp_fcons -> ia[l]-1][m].a;
619 o = tmp_fmol -> atoms_id[tmp_fcons -> ia[l]-1][m].b;
620 p = get_active_atom (tmp_fmol -> id, n) -> list[o];
621 field_selection (p, toviz.c, toviz.c, jd);
622 }
623 }
624 }
625 break;
626 case 4:
627 tmp_fpmf = get_active_pmf (kd, jd);
628 tmp_fpmf -> show = toviz.c;
629 for (m=0; m<2; m++)
630 {
631 if (tmp_fpmf -> num[m] > 0)
632 {
633 for (n=0; n<tmp_fpmf -> num[m]; n++)
634 {
635 o = tmp_fpmf -> list[m][n];
636 for (p=0; p<tmp_fmol -> multi;p++)
637 {
638 q = tmp_fmol -> atoms_id[o][p].a;
639 r = tmp_fmol -> atoms_id[o][p].b;
640 s = get_active_atom (tmp_fmol -> id, q) -> list[r];
641 field_selection (s, toviz.c, toviz.c, jd);
642 }
643 }
644 }
645 }
646 break;
647 case 5:
648 tmp_frig = get_active_rigid (kd, jd);
649 tmp_frig -> show = toviz.c;
650 if (tmp_frig -> num > 0)
651 {
652 for (n=0; n<tmp_frig -> num; n++)
653 {
654 o = tmp_frig -> list[n];
655 for (p=0; p<tmp_fmol -> multi;p++)
656 {
657 q = tmp_fmol -> atoms_id[o][p].a;
658 r = tmp_fmol -> atoms_id[o][p].b;
659 s = get_active_atom (tmp_fmol -> id, q) -> list[r];
660 field_selection (s, toviz.c, toviz.c, jd);
661 }
662 }
663 }
664 break;
665 case 6:
666 tmp_ftet = get_active_tethered (kd, jd);
667 tmp_ftet -> show = toviz.c;
668 if (tmp_ftet -> num)
669 {
670 o = tmp_ftet -> num-1;
671 p = tmp_fmol -> atoms_id[o][0].a;
672 for (q=0; q<tmp_fmol -> multi;q++)
673 {
674 r = tmp_fmol -> atoms_id[o][q].b;
675 s = get_active_atom (tmp_fmol -> id, p) -> list[r];
676 field_selection (s, toviz.c, toviz.c, jd);
677 }
678 }
679 break;
680 case SEXTERN:
681
682 break;
683 default:
684 if (id < MOLIMIT)
685 {
686 // View a type of bond / angles / dihedral / improper / inversion in a molecule
687 tmp_fstr = get_active_struct (id-7, kd, jd);
688 tmp_fat = get_active_atom (kd, tmp_fstr -> aid[0]);
689 tmp_fbt = get_active_atom (kd, tmp_fstr -> aid[1]);
690 if (id > 8) tmp_fct = get_active_atom (kd, tmp_fstr -> aid[2]);
691 if (id > 10) tmp_fdt = get_active_atom (kd, tmp_fstr -> aid[3]);
692 tmp_fstr -> def -> show = toviz.c;
693 if (id < 9)
694 {
695 visualize_bonds (tmp_fstr -> def -> show, jd, tmp_fat, tmp_fbt);
696 }
697 else if (id < 11)
698 {
699 visualize_angles (tmp_fstr -> def -> show, jd, tmp_fat, tmp_fbt, tmp_fct);
700 }
701 else if (id < 13)
702 {
703 visualize_dihedrals (tmp_fstr -> def -> show, jd, tmp_fat, tmp_fbt, tmp_fct, tmp_fdt);
704 }
705 else if (id < 15)
706 {
707 visualize_imp_inv (tmp_fstr -> def -> show, id-7, jd, tmp_fat, tmp_fbt, tmp_fct, tmp_fdt);
708 }
709 }
710 else
711 {
712 // View a non-bonded interaction
713 tmp_fbody = get_active_body (jd, id - MOLIMIT);
714 tmp_fbody -> show = toviz.c;
715 visualize_body (tmp_fbody -> show, jd, tmp_fbody);
716 }
717 break;
718 }

729void check_to_visualize_properties_for_this_field_mol (int pid, int mol)
730{
731 int h, i, j;
732 struct field_prop * propv;
733 h = toviz.c;
734 toviz.c = 1;
735 tmp_fmol = get_active_field_molecule (mol);
736 i = get_field_objects (pid, mol);
737 for (j=0; j<i; j++)
738 {
739 if (show_field_object (pid, j, mol)) visualize_object (pid, j, mol);
740 if (pid > 6 && pid < MOLIMIT)
741 {
742 propv = get_active_struct (pid-7, mol, j) -> other;
743 while (propv)
744 {
745 if (propv -> show) visualize_single_struct (pid, j, mol, propv -> aid);
746 propv = propv -> next;
747 }
748 }
749 }
750 toviz.c = h;
751}

779void update_mol_tree (int a, int b)
780{
781 int i, j;
782 j = toviz.c;
783 for (i=0; i<tmp_field -> molecules; i++)
784 {
785 if (get_active_field_molecule (i) -> show_id && ! a)
786 {
787 toviz.c = 0;
788 visualize_object (-1, i, 0);
789 check_to_visualize_properties (-1);
790 toviz.c = j;
791 }
792 else if (get_active_field_molecule (i) -> show && a)
793 {
794 toviz.c = 0;
795 visualize_object (0, i, 0);
796 check_to_visualize_properties (0);
797 toviz.c = j;
798 }
799 }
800 if (a)
801 {
802 get_active_field_molecule (b) -> show_id = j;
803 }
804 else
805 {
806 get_active_field_molecule (b) -> show = j;
807 }
808 gtk_tree_store_clear (field_model[0]);
809 init_default_shaders (tmp_view);
810 fill_field_model (field_model[0], 0, -1);
811}

822G_MODULE_EXPORT void on_toggle_visualize_or_select_object (GtkCellRendererToggle * cell_renderer, gchar * string_path, gpointer data)
