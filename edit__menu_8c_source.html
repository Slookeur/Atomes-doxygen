490gboolean test_vol (double box[2][3], double vect[3][3])
491{
492 int i, j;
493 gboolean val = TRUE;
494 tmp_lat = 0;
495 for (i=0; i<2; i++)
496 {
497 for (j=0; j<3; j++)
498 {
499 if (box[i][j] == 0.0) val = FALSE;
500 }
501 }
502 if (val) tmp_lat = 1;
503 val = TRUE;
504 for (i=0; i<3; i++)
505 {
506 if (vect[i][0] == 0.0 && vect[i][1] == 0.0 && vect[i][2] == 0.0)
507 {
508 val = FALSE;
509 }
510 }
511 if (val) tmp_lat = 2;
512 return (tmp_lat) ? TRUE : FALSE;
513}

734G_MODULE_EXPORT void run_on_edit_activate (GtkDialog * win, gint response_id, gpointer data)
735{
736 int id = GPOINTER_TO_INT(data);
737 int i = (id < 2) ? id : (id > 2 && id < 5) ? 1 : 2;
738
739 gboolean done = FALSE;
740 char * errpbc ="To apply PBC or use fractional coordinates\n"
741 "describe properly the simulation box parameters";
742 switch (response_id)
743 {
744 case GTK_RESPONSE_APPLY:
745 if (i== 0)
746 {
747 test_chem ();
748 done = TRUE;
749 }
750 else if (i == 1)
751 {
752 done = test_pbc (tmp_pbc, tmp_frac, tmp_box, tmp_vect);
753 if (done)
754 {
755 prep_box (i);
756 if (id == 1)
757 {
758 if (! active_project -> run)
759 {
760 run_project ();
761 if (active_glwin) active_glwin -> create_shaders[MDBOX] = TRUE;
762 bonds_update = 1;
763 frag_update = (active_project -> natomes > ATOM_LIMIT) ? 0 : 1;
764 mol_update = (frag_update) ? ((active_project -> steps > STEP_LIMIT) ? 0 : 1) : 0;
765 active_project -> runc[0] = FALSE;
766 on_calc_bonds_released (NULL, NULL);
767 }
768 }
769 }
770 else
771 {
772 show_warning (errpbc, GTK_WIDGET(win));
773 }
774 }
