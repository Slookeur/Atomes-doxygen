301void to_remove_this_object (int type, gpointer data)
302{
303 int i, j;
304 struct selatom * selat;
305 tint * sel;
306 prepare_atom_edition (& opengl_project -> modelgl -> colorp[0][0], FALSE);
307 remove_search = allocate_atom_search (opengl_project -> id, REMOVE, REMOVE, opengl_project -> natomes);
308 switch (type)
309 {
310 case 0:
311 i = GPOINTER_TO_INT(data);
312 remove_search -> todo[i] = 1;
313 break;
314 case 1:
315 sel = (tint * )data;
316 i = opengl_project -> modelgl -> anim -> last -> img -> step;
317 for (j=0; j<opengl_project -> natomes; j++)
318 {
319 if (sel -> a < 2)
320 {
321 if (opengl_project -> atoms[i][j].sp == selected_aspec && opengl_project -> atoms[i][j].coord[sel -> a] == sel -> b)
322 {
323 remove_search -> todo[j] = 1;
324 }
325 }
326 else
327 {
328 if (opengl_project -> atoms[i][j].coord[sel -> a] == sel -> b)
329 {
330 remove_search -> todo[j] = 1;
331 }
332 }
333 }
334 break;
335 case 2:
336 sel = (tint * )data;
337 i = opengl_project -> modelgl -> anim -> last -> img -> step;
338 gboolean doit;
339 for (j=0; j<opengl_project -> natomes; j++)
340 {
341 doit = FALSE;
342 if (opengl_project -> atoms[i][j].sp == selected_aspec || selected_aspec == -1)
343 {
344 if (! sel -> a)
345 {
346 if (opengl_project -> atoms[i][j].pick[0] == is_selected || is_selected == -1) doit = TRUE;
347 }
348 else if (sel -> a == 2)
349 {
350 doit = TRUE;
351 }
352 else if (opengl_project -> atoms[i][j].label[0] == is_labelled)
353 {
354 if (opengl_project -> atoms[i][j].pick[0] == is_selected || is_selected == -1) doit = TRUE;
355 }
356 if (doit)
357 {
358 remove_search -> todo[j] = 1;
359 }
360 }
361 }
362 break;
363 case 3:
364 selat = bond_selection -> first;
365 for (i=0; i<bond_selection -> selected ; i++)
366 {
367 j = selat -> id;
368 remove_search -> todo[j] = 1;
369 selat = selat -> next;
370 }
371 break;
372 }

382gchar * get_object_from_action (GSimpleAction * action)
383{
384 int i, j, k;
385 const gchar * act = g_action_get_name(G_ACTION(action));
386 gchar * act_end;
387 gchar * str;
388 k = strlen (act);
389 if (strstr(act, "set-mol-ins"))
390 {
391 act_end = g_strdup_printf ("%c%c", act[k-4], act[k-3]);
392 for (i=0; mol[i].type || mol[i].object; i++)
393 {
394 if (mol[i].object)
395 {
396 str = (i < 10) ? g_strdup_printf (".%d", i) : g_strdup_printf ("%d", i);
397 if (g_strcmp0(str, act_end) == 0)
398 {
399 g_free (str);
400 g_free (act_end);
401 return g_strdup_printf ("%s", mol[i].object);
402 break;
403 }
404 g_free (str);
405 }
406 }
407 g_free (act_end);
408 }
409 if (strstr(act, "set-ifp"))
410 {
411 act_end = g_strdup_printf ("%c%c%c%c", act[k-6], act[k-5], act[k-4], act[k-3]);
412 for (i=0; i<nprojects; i++)
413 {
414 if (get_project_by_id(i) -> steps == 1 && get_project_by_id(i) -> natomes)
415 {
416 for (j=0; j<3; j++)
417 {
418 str = g_strdup_printf ("-%d.%d", j, i);
419 if (g_strcmp0(str, act_end) == 0)
420 {
421 g_free (str);
422 g_free (act_end);
423 return g_strdup_printf ("%s in %s", action_atoms[j], g_strdup_printf ("%s (%d)", get_project_by_id(i) -> name, i+1));
424 }
425 g_free (str);
426 }
427 }
428 }
429 g_free (act_end);
430 }
431 return NULL;
432}
