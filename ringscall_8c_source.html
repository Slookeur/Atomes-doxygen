431G_MODULE_EXPORT void on_calc_rings_released (GtkWidget * widg, gpointer data)
432{
433 int search = active_project -> rsearch[0];
434 int i, j, k;
435
436 if (toggled_rings) active_project -> dmtx = FALSE;
437
438#ifdef DEBUG
439 g_debug ("Calc rings !");
440 g_debug (" - rings definition: %d", search);
441 if (active_project -> rsparam[search][2]) g_debug (" - only ABAB rings !");
442 if (active_project -> rsparam[search][3]) g_debug (" - no homopolar bonds in rings !");
443 if (active_project -> rsparam[search][4]) g_debug (" - no homopolar bonds at all !");
444 g_debug (" - dmtx= %d", active_project -> dmtx);
445#endif
446
447 cutoffsend ();
448 //if (active_project -> steps > 1) statusb = 1;
449 if (! active_project -> initok[RI])
450 {
451 initrng ();
452 }
453 active_project -> rsparam[search][5] = 0;
454 if (! active_project -> dmtx || active_project -> rsparam[search][4] || (search > 2 && active_cell -> pbc))
455 {
456 active_project -> dmtx = run_distance_matrix (widg, search+1, 0);
457 }
458 if (active_project -> dmtx)
459 {
460 i = search;
461 j = 4*(active_project -> nspec + 1) * i;
462 clean_curves_data (RI, j+4*active_project -> rsparam[i][0], j+4*(active_project -> rsparam[i][0]+1));
463 clean_rings_data (i, active_glwin);
464 active_glwin -> all_rings[i] = g_malloc0 (active_project -> steps*sizeof*active_glwin -> all_rings[i]);
465 active_glwin -> num_rings[i] = g_malloc0 (active_project -> steps*sizeof*active_glwin -> num_rings[i]);
466 active_glwin -> show_rpoly[i] = g_malloc0 (active_project -> steps*sizeof*active_glwin -> show_rpoly[i]);
467 active_glwin -> ring_max[i] = active_project -> rsparam[i][1];
468 active_glwin -> rings = TRUE;
469 for (j=0; j<active_project -> steps; j++)
470 {
471 active_glwin -> all_rings[i][j] = g_malloc0 (active_project -> rsparam[i][1]*sizeof*active_glwin -> all_rings[i][j]);
472 active_glwin -> num_rings[i][j] = allocint (active_project -> rsparam[i][1]);
473 active_glwin -> show_rpoly[i][j] = g_malloc (active_project -> rsparam[i][1]*sizeof*active_glwin -> show_rpoly[i][j]);
474 for (k=0; k < active_project -> natomes; k++)
475 {
476 active_project -> atoms[j][k].rings[i] = g_malloc0 (active_project -> rsparam[i][1]*sizeof*active_project -> atoms[j][k].rings[i]);
477 }
478 }
479 prepostcalc (widg, FALSE, RI, 0, opac);
480 clock_gettime (CLOCK_MONOTONIC, & start_time);
481 j = initrings_ (& search,
482 & active_project -> rsparam[i][1],
483 & active_project -> rsparam[i][0],
484 & active_project -> rsearch[1],
485 & active_project -> rsparam[i][2],
486 & active_project -> rsparam[i][3]);
487 clock_gettime (CLOCK_MONOTONIC, & stop_time);
488 active_project -> rsdata[i][4] = get_calc_time (start_time, stop_time);
489 if (j == 0)
490 {
491 show_error ("The ring statistics calculation has failed", 0, widg);
492 active_glwin -> ring_max[i] = 0;
493 active_project -> rsdata[i][4] = 0.0;
494 }
495 else if (j == 2)
496 {
497 gchar * str = g_strdup_printf ("\t<b>The ring statistics have failed !</b>\n\n"
498 "The number of ring per MD step appears\n"
499 "to be bigger than the initial value of <b>%d</b>\n"
500 "used to allocate memory to store the results.\n\n"
501 "Increase the value and start again !",
502 active_project -> rsearch[1]);
503 show_error (str, 0, widg);
504 g_free (str);
505 active_glwin -> ring_max[i] = 0;
506 active_project -> rsdata[i][4] = 0.0;
507 j = 0;
508 }
509 else
510 {
511 if (active_coord -> totcoord[4+search])
512 {
513 active_project -> rsparam[search][5] = 1;
514 update_rings_view (active_project, search);
515 }
516 else
517 {
518 active_glwin -> ring_max[i] = 0;
519 }
520 }
521 prepostcalc (widg, TRUE, RI, j, 1.0);
522 }
523 else
524 {
525 show_error ("The nearest neighbors table calculation has failed", 0, widg);
526 }
527 active_glwin -> rings = FALSE;
528 for (i=0; i<5; i++)
529 {
530 if (active_coord -> totcoord[4+i])
531 {
532 active_glwin -> rings = TRUE;
533 break;
534 }
535 }
536 gtk_widget_show (curvetoolbox);
537 clean_coord_window (active_project);
538#ifdef GTK3
539 update_rings_menus (active_glwin);
540#else
541 update_menu_bar (active_glwin);
542#endif
543 fill_tool_model ();
544 if (search > 2 && active_cell -> pbc) active_project -> dmtx = FALSE;
545}
