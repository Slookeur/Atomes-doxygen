600int prepare_measure_shaders (int type, int shaders)
601{
602 int nshaders = 0;
603 if (plot -> selected[type] -> selected > 1 && plot -> selected[type] -> selected < MAX_IN_SELECTION)
604 {
605 if (plot -> mpattern > -1)
606 {
607 // First the bond distances
608 measure = g_malloc0 (sizeof*measure);
609 measure -> vert_buffer_size = LINE_BUFF_SIZE;
610 measure -> num_vertices = 2 * num_bonds (plot -> selected[type] -> selected) * (plot -> extra_cell[0]+1)*(plot -> extra_cell[1]+1)*(plot -> extra_cell[2]+1);
611 measure -> vertices = allocfloat (measure -> vert_buffer_size*measure -> num_vertices);
612 nbs = 0;
613 bonds_loop (wingl, 0, type, NULL);
614
615 if (plot -> mpattern != 2)
616 {
617 wingl -> ogl_glsl[MEASU][0][shaders] = init_shader_program (MEASU, GLSL_LINES, line_vertex, line_stipple, line_stipple_color, GL_LINES, 2, 7, FALSE, measure);
618 }
619 else
620 {
621 wingl -> ogl_glsl[MEASU][0][shaders] = init_shader_program (MEASU, GLSL_LINES, line_vertex, NULL, line_color, GL_LINES, 2, 7, FALSE, measure);
622 }
623 wingl -> ogl_glsl[MEASU][0][shaders] -> line_width = plot -> mwidth;
624 nshaders ++;
625 g_free (measure);
626
627 // The angles
628 if (plot -> selected[type] -> selected > 2)
629 {
630 measure = g_malloc0 (sizeof*measure);
631 measure -> vert_buffer_size = LINE_BUFF_SIZE;
632 measure -> num_vertices = 3 * num_angles (plot -> selected[type] -> selected) * (plot -> extra_cell[0]+1)*(plot -> extra_cell[1]+1)*(plot -> extra_cell[2]+1);
633 measure -> vertices = allocfloat (measure -> vert_buffer_size*measure -> num_vertices);
634 nbs = 0;
635 angles_loop (wingl, 0, type, NULL);
636 if (plot -> mpattern != 2)
637 {
638 wingl -> ogl_glsl[MEASU][0][shaders+1] = init_shader_program (MEASU, GLSL_LINES, angle_vertex, angle_stipple, line_stipple_color, GL_TRIANGLES, 2, 7, FALSE, measure);
639 }
640 else
641 {
642 wingl -> ogl_glsl[MEASU][0][shaders+1] = init_shader_program (MEASU, GLSL_LINES, angle_vertex, angle_stipple, angle_color, GL_TRIANGLES, 2, 7, FALSE, measure);
643 }
644 wingl -> ogl_glsl[MEASU][0][shaders+1] -> line_width = plot -> mwidth;
645 nshaders ++;
646 g_free (measure);
647 }
648 }
649 // When all labels are found we render the text if any
650 if (plot -> labels_list[3+type] != NULL)
651 {
652 measures_drawing = nshaders + shaders;
653 render_all_strings (MEASU, 3+type);
654 nshaders += (plot -> labels_render[3+type]+1) * (plot -> labels_list[3+type] -> last -> id + 1);
655 }
656 }
657 return nshaders;
658}
