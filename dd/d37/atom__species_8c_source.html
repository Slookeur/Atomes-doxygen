50chemical_data * duplicate_chemical_data (int spec, chemical_data * chem)
51{
52 chemical_data * newchem = g_malloc0 (sizeof*chem);
53 newchem -> label = g_malloc (spec*sizeof*newchem -> label);
54 newchem -> element = g_malloc (spec*sizeof*newchem -> element);
55 int i, j;
56 for (i=0; i<spec; i++)
57 {
58 newchem -> label[i] = g_strdup_printf ("%s", chem -> label[i]);
59 newchem -> element[i] = g_strdup_printf ("%s", chem -> element[i]);
60 }
61 newchem -> nsps = duplicate_int (spec, chem -> nsps);
62 newchem -> formula = duplicate_int (spec, chem -> formula);
63 newchem -> cutoffs = allocddouble (spec, spec);
64 newchem -> grtotcutoff = chem -> grtotcutoff;
65 newchem -> chem_prop = allocddouble (CHEM_PARAMS, spec);
66 for (i=0; i<spec; i++)
67 {
68 for (j=0; j<spec; j++) newchem -> cutoffs[i][j] = chem -> cutoffs[i][j];
69 for (j=0; j<CHEM_PARAMS; j++) newchem -> chem_prop[j][i] = chem -> chem_prop[j][i];
70 }
71 return newchem;
72}

107int search_for_new_spec (atom_edition * edit, atomic_object * object)
108{
109 double * tmpnzid;
110
111 coord_info * coord = edit -> coord;
112 int i, j, k, l, m;
113 i = 0;
114 for (j=0; j<object -> species; j++)
115 {
116 if (object -> old_z[j] > 0)
117 {
118 k = find_spec_id (coord -> species, object -> old_z[j], edit -> new_z);
119 if (k < 0)
120 {
121 i ++;
122 tmpnzid = allocdouble (coord -> species+i);
123 for (l=0; l<coord -> species+i-1; l++) tmpnzid[l] = edit -> new_z[l];
124 tmpnzid[l] = (double)object -> old_z[j];
125 if (edit -> new_z) g_free (edit -> new_z);
126 edit -> new_z = duplicate_double (coord -> species+i, tmpnzid);
127 g_free (tmpnzid);
128 }
129 }
130 }
131
132 if (i)
133 {
134 coord_info * tmp = duplicate_coord_info (edit -> coord);
135 for (j=0; j<2; j++)
136 {
137 if (coord -> species)
138 {
139 g_free (coord -> ntg[j]);
140 g_free (coord -> geolist[j]);
141 }
142 coord -> ntg[j] = allocint (coord -> species + i);
143 coord -> geolist[j] = g_malloc0 ((coord -> species + i)*sizeof* coord -> geolist[j]);
144 if (j)
145 {
146 if (coord -> species) g_free (coord -> partial_geo);
147 coord -> partial_geo = g_malloc0 ((coord -> species + i)*sizeof*coord -> partial_geo);
148 }
149 for (k=0; k<coord -> species; k++)
150 {
151 coord -> ntg[j][k] = tmp -> ntg[j][k];
152 coord -> geolist[j][k] = duplicate_int (tmp -> ntg[j][k], tmp -> geolist[j][k]);
153 if (j)
154 {
155 coord -> partial_geo[k] = allocdint (coord -> ntg[j][k], coord -> species + i);
156 for (l=0; l<tmp -> ntg[j][k]; l++)
157 {
158 for (m=0; m<coord -> species; m++)
159 {
160 coord -> partial_geo[k][l][m] = tmp -> partial_geo[k][l][m];
161 }
162 }
163 }
164 }
165 }
166 coord -> species += i;
167 g_free (tmp);
168 }
169 return i;
170}
