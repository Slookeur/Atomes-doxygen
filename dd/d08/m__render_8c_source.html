132G_MODULE_EXPORT void change_render_radio (GSimpleAction * action, GVariant * parameter, gpointer data)
133{
134 glwin * view = (glwin *)data;
135 const gchar * render = g_variant_get_string (parameter, NULL);
136 int lgt = strlen (render);
137 gchar * name = g_strdup_printf ("%c%c", render[lgt-2], render[lgt-1]);
138 if (g_strcmp0(name, ".1") == 0)
139 {
140 g_free (name);
141 name = g_strdup_printf ("%.*s.0", lgt-2, render);
142 g_action_group_activate_action ((GActionGroup *)view -> action_group, "set-render", g_variant_new_string((const gchar *)name));
143 g_free (name);
144 }
145 else
146 {
147 gchar * render_name = NULL;
148 int i;
149 for (i=0; i<OGL_RENDERS*2; i++)
150 {
151 render_name = g_strdup_printf ("set-render.%d.0", i);
152 if (g_strcmp0(render, (const gchar *)render_name) == 0)
153 {
154 set_render (NULL, & view -> colorp[i][0]);
155 g_free (render_name);
156 render_name = NULL;
157 break;
158 }
159 g_free (render_name);
160 render_name = NULL;
161 }
162 g_action_change_state (G_ACTION (action), parameter);
163 }
