53void add_bonds_to_project (project * this_proj, int removed, int nbd, int ** new_bond_list)
54{
55 int i, j;
56 int ** tmpbondid = NULL;
57 if (nbd)
58 {
59 i = this_proj -> modelgl -> bonds[0][0];
60 tmpbondid = allocdint (i+nbd, 2);
61 for (j=0; j<i; j++)
62 {
63 tmpbondid[j][0] = this_proj -> modelgl -> bondid[0][0][j][0];
64 tmpbondid[j][1] = this_proj -> modelgl -> bondid[0][0][j][1];
65 }
66 if (this_proj -> modelgl -> allbonds[0]) g_free (this_proj -> modelgl -> bondid[0][0]);
67 this_proj -> modelgl -> bondid[0][0] = allocdint (i+nbd, 2);
68 for (j=0; j<i; j++)
69 {
70 this_proj -> modelgl -> bondid[0][0][j][0] = tmpbondid[j][0];
71 this_proj -> modelgl -> bondid[0][0][j][1] = tmpbondid[j][1];
72 }
73 for (j=0; j<nbd; j++)
74 {
75 this_proj -> modelgl -> bondid[0][0][j+i][0] = new_bond_list[j][0] + this_proj -> natomes - removed;
76 this_proj -> modelgl -> bondid[0][0][j+i][1] = new_bond_list[j][1] + this_proj -> natomes - removed;
77 }
78 this_proj -> modelgl -> bonds[0][0] += nbd;
79 this_proj -> modelgl -> allbonds[0] += nbd;
80 g_free (tmpbondid);
81 }
82}

131G_MODULE_EXPORT void set_atoms_to_insert (GtkComboBox * box, gpointer data)
132{
