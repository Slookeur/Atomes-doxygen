42{

47 i = max(last -> extra_cell[0], last -> extra_cell[1]);
48 i = max(i, last -> extra_cell[2]);
49 last -> p_depth /= (i+1);
50 last -> extra_cell[dat -> b] = gtk_spin_button_get_value_as_int(res) - 1;
51 i = max(last -> extra_cell[0], last -> extra_cell[1]);
52 i = max(i, last -> extra_cell[2]);

61 if (this_proj -> modelgl -> n_shaders[SLABS][0]) this_proj -> modelgl -> create_shaders[SLABS] = TRUE;
62 update (this_proj -> modelgl);
