159G_MODULE_EXPORT void run_cp2k_fix_molecule (GtkDialog * dial, gint response_id, gpointer data)
160{
161 gchar * str;
162 int i, j;
163 gboolean done = FALSE;
164 switch (response_id)
165 {
166 case GTK_RESPONSE_APPLY:
167 if (a_frag > 0)
168 {
169 if (a_frag > 1)
170 {
171 str = g_strdup_printf ("Fragments N°%d", fix_frag[0]+1);
172 if (a_frag > 2)
173 {
174 for (i=1; i<a_frag-1; i++)
175 {
176 str = g_strdup_printf ("%s, %d", str, fix_frag[i]+1);
177 }
178 }
179 str = g_strdup_printf ("%s and %d have been selected !", str, fix_frag[a_frag-1]+1);
180 str = g_strdup_printf ("%s\nConfirm this choice and fix these fragments ?", str);
181 }
182 else
183 {
184 str = g_strdup_printf ("Fragment N°%d has been selected !", fix_frag[0]+1);
185 str = g_strdup_printf ("%s\nConfirm this choice and fix this fragment ?", str);
186 }
187 field_question (str, G_CALLBACK(select_fixed_atom_confirm), NULL);
188 g_free (str);
189 if (sel_and_conf)
190 {
191 done = TRUE;
192 if (tmp_cp2k -> fixlist[1] != NULL)
193 {
194 g_free (tmp_cp2k -> fixlist[1]);
195 tmp_cp2k -> fixlist[1] = NULL;
196 if (tmp_cp2k -> fixcoord[1] != NULL)
197 {
198 g_free (tmp_cp2k -> fixcoord[1]);
199 tmp_cp2k -> fixcoord[1] = NULL;
200 }
201 }
202 if (a_frag > 0)
203 {
204 tmp_cp2k -> fixlist[1] = allocint (a_frag);
205 tmp_cp2k -> fixcoord[1] = allocdint (a_frag, 3);
206 tmp_cp2k -> fixat[1] = a_frag;
207 j = -1;
208 for (i=0; i<qm_coord -> totcoord[2]; i++)
209 {
210 if (fix_frag[i])
211 {
212 j ++;
213 tmp_cp2k -> fixlist[1][j] = i;
214 tmp_cp2k -> fixcoord[1][j][0] = old_fixed[i][0];
215 tmp_cp2k -> fixcoord[1][j][1] = old_fixed[i][1];
216 tmp_cp2k -> fixcoord[1][j][2] = old_fixed[i][2];
217 }
218 }
219 }
220 }
221 }
222 break;
223 case GTK_RESPONSE_CLOSE:
224 done = TRUE;
225 break;
226 }
227 if (done) destroy_this_dialog (dial);
228}

235void cp2k_fix_molecule ()
236{
237 int i, j, k;
238 at_col = 0;
239 gchar * str = g_strdup_printf ("Please select the fragment(s) to fix");
240 GtkWidget * fmol = dialogmodal (str, GTK_WINDOW(qm_assistant));
241 g_free (str);
242 gtk_dialog_add_button (GTK_DIALOG(fmol), "Apply", GTK_RESPONSE_APPLY);
243 GtkWidget * frag_tree = NULL;
244 GtkTreeIter iter;
245 GtkTreeViewColumn * frag_col[5];
246 GtkCellRenderer * frag_cell[5];
247 gchar * frag_title[5] = {"Fragment", "Viz.3D & Select", "x", "y", "z"};
248 gchar * ctype[5]={"text", "active", "active", "active", "active"};
249 GType col_type[5] = {G_TYPE_INT, G_TYPE_BOOLEAN, G_TYPE_BOOLEAN, G_TYPE_BOOLEAN, G_TYPE_BOOLEAN};
250 add_model = gtk_tree_store_newv (5, col_type);
251 frag_tree = gtk_tree_view_new_with_model(GTK_TREE_MODEL(add_model));
252 for (i=0; i<5; i++)
253 {
254 if (i == 0)
255 {
256 frag_cell[i] = gtk_cell_renderer_text_new ();
257 }
258 else
259 {
260 frag_cell[i] = gtk_cell_renderer_toggle_new ();
261 if (i == 1)
262 {
263 g_signal_connect (G_OBJECT(frag_cell[i]), "toggled", G_CALLBACK(select_frag), NULL);
264 }
265 else
266 {
267 j = i + 2;
268 g_signal_connect (G_OBJECT(frag_cell[i]), "toggled", G_CALLBACK(cp2k_select_coord_id), GINT_TO_POINTER(j));
269 }
270 }
271 frag_col[i] = gtk_tree_view_column_new_with_attributes (frag_title[i], frag_cell[i], ctype[i], i, NULL);
272 gtk_tree_view_append_column (GTK_TREE_VIEW(frag_tree), frag_col[i]);
273 gtk_tree_view_column_set_alignment (frag_col[i], 0.5);
274 if (i == 0)
275 {
276 gtk_tree_view_column_set_cell_data_func (frag_col[i], frag_cell[i], frag_set_color, NULL, NULL);
277 }
278 else if (i > 1)
279 {
280 gtk_tree_view_column_set_cell_data_func (frag_col[i], frag_cell[i], frag_set_visible, NULL, NULL);
281 }
282 }
283 // fill model
284 fix_frag = NULL;
285 fix_frag = allocint (qm_coord -> totcoord[2]);
286 old_fixed = allocdint (qm_coord -> totcoord[2], 3);
287 a_frag = 0;
288 for (i=0; i<qm_coord -> totcoord[2]; i++)
289 {
290 gtk_tree_store_append (add_model, & iter, NULL);
291 for (j=0; j<tmp_cp2k -> fixat[1]; j++)
292 {
293 if (tmp_cp2k -> fixlist[1][j] == i)
294 {
295 fix_frag[i] = 1;
296 a_frag ++;
297 for (k=0; k<3; k++) old_fixed[i][k] = tmp_cp2k -> fixcoord[1][j][k];
298 break;
299 }
300 }
301 gtk_tree_store_set (add_model, & iter, 0, i + 1,
302 1, fix_frag[i],
303 2, old_fixed[i][0],
304 3, old_fixed[i][1],
305 4, old_fixed[i][2], -1);
306 }
307 g_object_unref (add_model);
308 gtk_tree_view_expand_all (GTK_TREE_VIEW(frag_tree));
309
310 i = ((qm_coord -> totcoord[2]+1)*35 < 500) ? (qm_coord -> totcoord[2]+1)*35 : 500;
311 GtkWidget * scrollsets = create_scroll (dialog_get_content_area (fmol), 220, i, GTK_SHADOW_ETCHED_IN);
312 add_container_child (CONTAINER_SCR, scrollsets, frag_tree);
313 run_this_gtk_dialog (fmol, G_CALLBACK(run_cp2k_fix_molecule), NULL);
314 proj_unselect_all_atoms ();
315 g_free (fix_frag);
316}
