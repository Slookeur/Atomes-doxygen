210void add_container_child (int type, GtkWidget * widg, GtkWidget * child)
211{
212#ifdef GTK3
213 gtk_container_add (GTK_CONTAINER(widg), child);
214#else
215 switch (type)
216 {
217 case CONTAINER_WIN:
218 // gtk_window_set_child ((GtkWindow *)widg, NULL);
219 gtk_window_set_child ((GtkWindow *)widg, child);
220 break;
221 case CONTAINER_SCR:
222 // Bug in the management of scrolled window child
223/* Frequently getting warning messages at this point with GTK4
224(atomes:?????): GLib-GObject-CRITICAL **: ??:??:??.???: g_object_set: assertion 'G_IS_OBJECT (object)' failed
225(atomes:?????): Gtk-CRITICAL **: ??:??:??.???: gtk_widget_unparent: assertion 'GTK_IS_WIDGET (widget)' failed
226*/
227 gtk_scrolled_window_set_child ((GtkScrolledWindow *)widg, NULL);
228 gtk_scrolled_window_set_child ((GtkScrolledWindow *)widg, child);
229 break;
230 case CONTAINER_VIE:
231 // gtk_viewport_set_child ((GtkViewport *)widg, NULL);
232 gtk_viewport_set_child ((GtkViewport *)widg, child);
233 break;
234 case CONTAINER_BUT:
235 // gtk_button_set_child ((GtkButton *)widg, NULL);
236 gtk_button_set_child ((GtkButton *)widg, child);
237 break;
238 case CONTAINER_FRA:
239 // gtk_frame_set_child ((GtkFrame *)widg, NULL);
240 gtk_frame_set_child ((GtkFrame *)widg, child);
241 break;
242 case CONTAINER_EXP:
243 // gtk_expander_set_child ((GtkExpander *)widg, NULL);
244 gtk_expander_set_child ((GtkExpander *)widg, child);

740 if (scroll_handler != NULL) g_signal_connect (G_OBJECT(hscale), "change-value", scroll_handler, data);
741 return hscale;

1205static void convert_alpha (cairo_surface_t * surf, GdkPixbuf * pix, int src_x, int src_y, int width, int height)
1206{
1207 int x, y;
1208 guchar * src_data = cairo_image_surface_get_data (surf);
1209 int src_stride = cairo_image_surface_get_stride (surf);
1210 guchar * dest_data = gdk_pixbuf_get_pixels (pix);
1211 int dest_stride = gdk_pixbuf_get_rowstride (pix);
1212
1213 src_data += src_stride * src_y + src_x * 4;
1214
1215 for (y = 0; y < height; y++) {
1216 guint32 *src = (guint32 *) src_data;
1217
1218 for (x = 0; x < width; x++) {
1219 guint alpha = src[x] >> 24;
1220
1221 if (alpha == 0)
1222 {
1223 dest_data[x * 4 + 0] = 0;
1224 dest_data[x * 4 + 1] = 0;
1225 dest_data[x * 4 + 2] = 0;
1226 }
1227 else
1228 {
1229 dest_data[x * 4 + 0] = (((src[x] & 0xff0000) >> 16) * 255 + alpha / 2) / alpha;
1230 dest_data[x * 4 + 1] = (((src[x] & 0x00ff00) >> 8) * 255 + alpha / 2) / alpha;
1231 dest_data[x * 4 + 2] = (((src[x] & 0x0000ff) >> 0) * 255 + alpha / 2) / alpha;
1232 }
1233 dest_data[x * 4 + 3] = alpha;
1234 }
1235
1236 src_data += src_stride;
1237 dest_data += dest_stride;
1238 }
1239}

1249{

1251 int height = cairo_image_surface_get_height (surf);
1252 GdkPixbuf * pix = gdk_pixbuf_new (GDK_COLORSPACE_RGB, TRUE, 8, width , height);
1253 convert_alpha (surf, pix, 0, 0, width, height);
1254 return pix;
