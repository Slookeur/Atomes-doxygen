200void add_container_child (int type, GtkWidget * widg, GtkWidget * child)
201{
202#ifdef GTK3
203 gtk_container_add (GTK_CONTAINER(widg), child);
204#else
205 switch (type)
206 {
207 case CONTAINER_WIN:
208 // gtk_window_set_child ((GtkWindow *)widg, NULL);
209 gtk_window_set_child ((GtkWindow *)widg, child);
210 break;
211 case CONTAINER_SCR:
212 // Bug in the management of scrolled window child
213/* Frequently getting warning messages at this point with GTK4
214(atomes:?????): GLib-GObject-CRITICAL **: ??:??:??.???: g_object_set: assertion 'G_IS_OBJECT (object)' failed
215(atomes:?????): Gtk-CRITICAL **: ??:??:??.???: gtk_widget_unparent: assertion 'GTK_IS_WIDGET (widget)' failed
216*/
217 gtk_scrolled_window_set_child ((GtkScrolledWindow *)widg, NULL);
218 gtk_scrolled_window_set_child ((GtkScrolledWindow *)widg, child);
219 break;
220 case CONTAINER_VIE:
221 // gtk_viewport_set_child ((GtkViewport *)widg, NULL);
222 gtk_viewport_set_child ((GtkViewport *)widg, child);
223 break;
224 case CONTAINER_BUT:
225 // gtk_button_set_child ((GtkButton *)widg, NULL);
226 gtk_button_set_child ((GtkButton *)widg, child);
227 break;
228 case CONTAINER_FRA:
229 // gtk_frame_set_child ((GtkFrame *)widg, NULL);
230 gtk_frame_set_child ((GtkFrame *)widg, child);
231 break;

1194static void convert_alpha (cairo_surface_t * surf, GdkPixbuf * pix, int src_x, int src_y, int width, int height)
1195{
1196 int x, y;
1197 guchar * src_data = cairo_image_surface_get_data (surf);
1198 int src_stride = cairo_image_surface_get_stride (surf);
1199 guchar * dest_data = gdk_pixbuf_get_pixels (pix);
1200 int dest_stride = gdk_pixbuf_get_rowstride (pix);
1201
1202 src_data += src_stride * src_y + src_x * 4;
1203
1204 for (y = 0; y < height; y++) {
1205 guint32 *src = (guint32 *) src_data;
1206
1207 for (x = 0; x < width; x++) {
1208 guint alpha = src[x] >> 24;
1209
1210 if (alpha == 0)
1211 {
1212 dest_data[x * 4 + 0] = 0;
1213 dest_data[x * 4 + 1] = 0;
1214 dest_data[x * 4 + 2] = 0;
1215 }
1216 else
1217 {
1218 dest_data[x * 4 + 0] = (((src[x] & 0xff0000) >> 16) * 255 + alpha / 2) / alpha;
1219 dest_data[x * 4 + 1] = (((src[x] & 0x00ff00) >> 8) * 255 + alpha / 2) / alpha;
1220 dest_data[x * 4 + 2] = (((src[x] & 0x0000ff) >> 0) * 255 + alpha / 2) / alpha;
1221 }
1222 dest_data[x * 4 + 3] = alpha;
1223 }
1224
1225 src_data += src_stride;
1226 dest_data += dest_stride;
1227 }
1228}
