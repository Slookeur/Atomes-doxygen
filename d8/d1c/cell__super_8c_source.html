162void super_celling (glwin * view)
163{
164 gchar * txta = "You are about to change the periodicity of the 3D model,";
165 gchar * txtb = "this will affect the entire molecular dynamics trajectory,";
166 gchar * txtc = "\tand the action is irreversible, proceed anyway ?";
167 gchar * str;
168 if (get_project_by_id(view -> proj) -> steps > 1)
169 {
170 str = g_strdup_printf ("%s\n%s\n%s", txta, txtb, txtc);
171 }
172 else
173 {
174 str = g_strdup_printf ("%s\n%s", txta, txtc);
175 }
176 if (ask_yes_no ("Create a super-cell ?", str, GTK_MESSAGE_WARNING, view -> win))
177 {
178 int i, j, k, l;
179 k = activep;
180 image * last = view -> anim -> last -> img;
181 if (k != view -> proj) active_project_changed (view -> proj);
182 preserve_ogl_selection (view);
183 if (add_cells_ (& active_project -> natomes, & active_project -> steps, last -> extra_cell))
184 {
185 if (active_cell -> crystal)
186 {
187 vec3_t shift;
188 shift.x = active_box -> vect[0][0] + active_box -> vect[1][0] + active_box -> vect[2][0];
189 shift.y = active_box -> vect[0][1] + active_box -> vect[1][1] + active_box -> vect[2][1];
190 shift.z = active_box -> vect[0][2] + active_box -> vect[1][2] + active_box -> vect[2][2];
191 for (i=0; i<active_project -> steps; i++)
192 {
193 for (j=0; j<active_project -> natomes; j++)
194 {
195 active_project -> atoms[i][j].x += shift.x;
196 active_project -> atoms[i][j].y += shift.y;
197 active_project -> atoms[i][j].z += shift.z;
198 }
199 }
200 }
201 l = 1;
202 for (i=0; i<3; i++)
203 {
204 for (j=0; j<3; j++)
205 {
206 active_box -> vect[i][j] *= (last -> extra_cell[i] + 1);
207 }
208 l *= (last -> extra_cell[i] + 1);
209 active_box -> param[0][i] *= (last -> extra_cell[i] + 1);
210 last -> extra_cell[i] = 0;
211 if (active_glwin -> cell_win)
212 {
213 if (active_glwin -> cell_win -> ax_cell[i])
214 {
215 gtk_spin_button_set_value (GTK_SPIN_BUTTON(active_glwin -> cell_win -> ax_cell[i]), 1.0);
216 }
217 }
218 }
219 if (active_cell -> crystal)
220 {
221 vec3_t shift;
222 shift.x = active_box -> vect[0][0] + active_box -> vect[1][0] + active_box -> vect[2][0];
223 shift.y = active_box -> vect[0][1] + active_box -> vect[1][1] + active_box -> vect[2][1];
224 shift.z = active_box -> vect[0][2] + active_box -> vect[1][2] + active_box -> vect[2][2];
225 for (i=0; i<active_project -> steps; i++)
226 {
227 for (j=0; j<active_project -> natomes; j++)
228 {
229 active_project -> atoms[i][j].x -= shift.x/2.0;
230 active_project -> atoms[i][j].y -= shift.y/2.0;
231 active_project -> atoms[i][j].z -= shift.z/2.0;
232 }
233 }
234 }
235 for (i=0; i<active_project -> nspec; i++)
236 {
237 active_chem -> nsps[i] *= l;
238 }
239 init_curves_and_calc (active_project);
240 if (! active_project -> cell.crystal) center_molecule (active_project);
241
242 active_project_changed (view -> proj);
243 active_project -> dmtx = FALSE;
244 bonds_update = 1;
245 active_project -> runc[0] = FALSE;
246 frag_update = (active_project -> natomes > ATOM_LIMIT) ? 0 : 1;
247 mol_update = (frag_update) ? ((active_project -> steps > STEP_LIMIT) ? 0 : 1) : 0;
248 gboolean ** cshow = duplicate_geom_info (active_project);
249 gboolean ** pshow = duplicate_poly_info (active_project);
250 if (view -> rings)
251 {
252 view -> rings = FALSE;
253 for (i=0; i<5; i++)
254 {
255 clean_rings_data (i, view);
256#ifdef GTK3
257 update_rings_menus (view);
258#endif
259 }
260 }
261 if (view -> chains)
262 {
263 clean_chains_data (view);
264#ifdef GTK3
265 update_chains_menus (view);
266#endif
267 }
268 on_calc_bonds_released (NULL, NULL);
269 restore_coord_and_poly_info (active_project, cshow, pshow);
270 g_free (cshow);
271 g_free (pshow);
272 int shaders[1] = {POLYS};
273 re_create_md_shaders (1, shaders, active_project);
274 view -> create_shaders[PICKS] = TRUE;
275 view -> create_shaders[MDBOX] = TRUE;
276 view -> create_shaders[LABEL] = TRUE;
277 view -> create_shaders[MEASU] = TRUE;
278 }
279 else
280 {
281 show_warning ("Something went wrong, the 3D model was not updated", view -> win);
282 }
283 restore_ogl_selection (view);
284 fill_tool_model ();
285 clean_other_window_after_edit (active_project);
286 if (k != view -> proj) active_project_changed (k);
287 }
288 update (view);
289#ifdef GTK
290 update_menu_bar (view);
291#endif
292}
