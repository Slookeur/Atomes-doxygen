535void prepare_cif_list (int dba, int sym, int spg)
536{
537#ifdef G_OS_WIN32
538 data_xml = g_strdup_printf ("%s\\crystals\\%s_list.xml", PACKAGE_LIB_DIR, database_dir[dba]);

541#endif
542 xmlDoc * doc;
543 xmlTextReaderPtr reader;
544 const xmlChar cif_base[8]="cif-database";
545 xmlChar * cdata;

547 reader = xmlReaderForFile(filetoread, NULL, 0);
548 if (reader == NULL)
549 {
550 return 0;
551 }
552 else
553 {
554 doc = xmlParseFile(filetoread);
555 if (doc == NULL) return 0;
556 racine = xmlDocGetRootElement(doc);
557 if (strcmp((char *)(racine -> name), (char *)cif_base) != 0) return clean_xml_data (doc, reader);
558 node = findnode (racine -> children, "cif-list");
559 if (node == NULL) return clean_xml_data (doc, reader);
560 sym_node = findnode (sym_node, sym_list[sym]);
561 if (sym_node == NULL) return clean_xml_data (doc, reader);

571 xmlFreeDoc(doc);
572 xmlFreeTextReader(reader);
573 xmlCleanupParser();

584int open_cif_database (gchar * filetoread)
585{
586 xmlDoc * doc;
587 xmlTextReaderPtr reader;
588 const xmlChar cif_base[8]="cif-database";
589 xmlChar * cdata;
590 xmlNodePtr racine, node, sym_node, group_node, cif_node;
591 gchar * str;
592 /*
593 * build an xmlReader for that file
594 */
595 reader = xmlReaderForFile(filetoread, NULL, 0);
596 if (reader == NULL)
597 {
598 return 0;
599 }
600 else
601 {
602 doc = xmlParseFile(filetoread);
603 if (doc == NULL) return 0;
604 racine = xmlDocGetRootElement(doc);
605 if (strcmp((char *)(racine -> name), (char *)cif_base) != 0) return clean_xml_data (doc, reader);
606
607 node = racine -> children;
608 node = findnode (racine -> children, "num-cif");
609 if (node == NULL) return clean_xml_data (doc, reader);
610 total_num_cif = (int)atof((char *)xmlNodeGetContent(node));
611 sym_node = findnode (racine -> children, "sym-cif");
612 if (sym_node == NULL) return clean_xml_data (doc, reader);
613 int i, j;
614 for (i=0; i<SYMMETRIES; i++)
615 {
616 node = findnode (sym_node, sym_list[i]);
617 if (node == NULL) return clean_xml_data (doc, reader);
618 j = (int)atof((char *)xmlNodeGetContent(node));
619 group_names_by_sym[i] = calloc(j, sizeof*group_names_by_sym[i]);
620 num_cif_by_sym_group[i] = calloc(j, sizeof*num_cif_by_sym_group[i]);
621 num_group_by_sym[i] = j;
622 for (j=0; <num_group_by_sym[i]; j++)
623 {
624 group_node = findnode (node, "group");
625 if (group_node == NULL) return clean_xml_data (doc, reader);
626 group_spec = group_node -> properties;
627 while (group_spec)
628 {
629 cif_node = group_spec -> children;
630 if (cif_node == NULL) return clean_xml_data (doc, reader);
631 if (strcmp("num_cif", (char *)cif_node -> name) == 0)
632 {
633 num_cif_by_sym_group[i][j] = (int)atof((char *)xmlNodeGetContent(cif_node));
634 }
635 }
636 group_names_by_sym[i][j] = g_strdup_printf ("%s", (char *)xmlNodeGetContent(group));
637 }
638 }
639 sym_node = findnode (racine -> children, "cif-list");
640 if (sym_node == NULL) return clean_xml_data (doc, reader);
641 for (i=0; i<SYMMETRIES; i++)
642 {
643 node = findnode (sym_node, sym_list[i]);
644 if (node == NULL) return clean_xml_data (doc, reader);
645 for (j=0; j<num_group_by_sym[i]; j++)
646 {
647 group_node = findnode (node, group_names_by_sym[i][j]);
648 if (node == NULL) return clean_xml_data (doc, reader);
649 }
650 }
651 xmlFreeDoc(doc);
652 xmlFreeTextReader(reader);
653 xmlCleanupParser();
654 return 1;
655 }
656}

675void select_cif_data (GtkTreeView * tree_view,
676 GtkTreePath * path,
677 GtkTreeViewColumn * column,
678 gpointer data)
679{
680 GtkTreeIter row;
681 GtkTreeModel * model = gtk_tree_view_get_model(tree_view);
682 if (gtk_tree_model_get_iter (model, & row, path))
683 {
684 GValue val, vbl, vcl;
685 val = vbl = vcl = {0, };
686 gtk_tree_model_get_value (model, & row, 0, & val);
687 gtk_tree_model_get_value (model, & row, 1, & vbl);
688 gtk_tree_model_get_value (model, & row, 2, & vcl);
689 int i, j, k;
690 i = (int)g_value_get_int (& val);
691 j = (int)g_value_get_int (& vbl);
692 switch (j)
693 {
694 case 3:
695 the_cif = g_strdup_printf ("%s", (char *)g_value_get_string (& vcl));
696 prepare_cif_preview (i);
697 break;
698 case 2:
699 the_group = i;
700 break;
701 case 1:
702 the_symmetry = i;
703 the_group = 0;
704 break;
705 case 0:
706 the_database = i;
707 the_symmetry = the_group = 0;
708 k = prepare_database (i);
709 break;
710 }
711 if (j < 3)
712 {
713 gtk_list_store_clear (cif_store);
714 if (j == 0 && k) gtk_tree_selection_select_iter (dataselect[1], & first_symmetry_iter);
715 if (j == 0 || j == 1) gtk_tree_selection_select_iter (dataselect[2], & first_group_iter);
716 if (j || k)
717 {
718 prepare_cif_list (the_databas, the_symmetry, the_group);
719 fill_cif_tree (cif_store);
720 gtk_tree_selection_select_iter (dataselect[3], & first_cif_iter);
721 prepare_cif_preview (0);
722 }
723 }
724 }
