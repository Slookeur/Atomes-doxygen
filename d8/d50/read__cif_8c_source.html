134 ssize_t getline(char **lineptr, size_t *n, FILE *stream)
135 {
136 size_t pos;
137 int c;
138
139 if (lineptr == NULL || stream == NULL || n == NULL)
140 {
141 errno = EINVAL;
142 return -1;
143 }
144
145 c = getc(stream);
146 if (c == EOF) return -1;
147
148 if (*lineptr == NULL)
149 {
150 *lineptr = malloc(128);
151 if (*lineptr == NULL) return -1;
152 *n = 128;
153 }
154
155 pos = 0;
156 while(c != EOF)
157 {
158 if (pos + 1 >= *n)
159 {
160 size_t new_size = *n + (*n >> 2);
161 if (new_size < 128)
162 {
163 new_size = 128;
164 }
165 char *new_ptr = realloc(*lineptr, new_size);
166 if (new_ptr == NULL) return -1;
167 *n = new_size;
168 *lineptr = new_ptr;
169 }
170
171 ((unsigned char *)(*lineptr))[pos ++] = c;
172 if (c == '\n') break;
173 c = getc(stream);
174 }
175
176 (*lineptr)[pos] = '\0';
177 return pos;
178 }
