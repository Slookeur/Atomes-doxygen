122G_MODULE_EXPORT void choose_format (GtkComboBox * box, gpointer cid)
123{
124 forme = gtk_combo_box_get_active (box);
125 if (forme != -1)
126 {
127 dwidget * imf = (dwidget *) cid;
128#ifdef GTK4
129 gtk_check_button_set_active (GTK_CHECK_BUTTON(imf -> a), 0);

131 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON(imf -> a), 0);

151 break;
152 case 3:
153 gtk_image_set_from_file (GTK_IMAGE(imf -> b), PACKAGE_EPS);
154 break;
155 }
156 }

280G_MODULE_EXPORT void run_save_image (GtkDialog * save_img, gint response_id, gpointer data)
281{
282 gboolean done = FALSE;
283 while (! done)
284 {
285 switch (response_id)
286 {
287 case GTK_RESPONSE_APPLY:
288 if (forme > -1)
289 {
290 write_image (data);
291 done = TRUE;
292 }
293 else
294 {
295 done = FALSE;
296 }
297 break;
298 default:
299 done = TRUE;
300 break;
301 }
302 }
303 if (done) destroy_this_dialog (save_img);
304}
