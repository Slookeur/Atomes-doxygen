126G_MODULE_EXPORT void set_axis_min (GtkEntry * res, gpointer data)
127{
128 tint * ad = (tint *)data;
129 a = ad -> a;
130 b = ad -> b;
131 c = ad -> c;
132 struct project * this_proj = get_project_by_id(a);
133 int i = get_active_axis();
134 if (i > -1)
135 {
136 const gchar * m;
137 m = entry_get_text (res);
138 if (atof(m) < this_proj -> curves[b][c] -> axmax[i])
139 {
140 this_proj -> curves[b][c] -> axmin[i] = atof(m);
141 }
142 else
143 {
144 show_warning ("Axis min must be < to axis max", this_proj -> curves[b][c] -> window);
145 }
146 update_entry_double (res, this_proj -> curves[b][c] -> axmin[i]);
147 update_curve (data);
148 }
149}
