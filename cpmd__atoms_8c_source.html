343G_MODULE_EXPORT void run_remove_dummy (GtkDialog * dialog, gint response_id, gpointer data)
344{
345 int i, j;
346 gboolean done = FALSE;
347 int num_to_remove = GPOINTER_TO_INT(data);
348 struct dummy_atom * dumm;
349 gchar * str;
350 switch (response_id)
351 {
352 case GTK_RESPONSE_APPLY:
353 if (n_dummy == num_to_remove)
354 {
355 done = TRUE;
356 // Now we remove all selected dummy atoms
357 for (i=0; i<n_dummy; i++)
358 {
359 dumm = tmp_cpmd -> dummy;
360 for (j=0; j<tmp_cpmd -> dummies; j++)
361 {
362 if (dumm -> id == old_dummy[i])
363 {
364 if (dumm -> next != NULL)
365 {
366 if (dumm -> prev != NULL)
367 {
368 dumm -> next -> prev = dumm -> prev;
369 dumm -> prev -> next = dumm -> next;
370 }
371 else
372 {
373 tmp_cpmd -> dummy = dumm -> next;
374 tmp_cpmd -> dummy -> prev = NULL;
375 }
376 g_free (dumm);
377 }
378 else if (dumm -> prev != NULL)
379 {
380 dumm -> prev -> next = NULL;
381 g_free (dumm);
382 }
383 else
384 {
385 g_free (tmp_cpmd -> dummy);
386 tmp_cpmd -> dummy = NULL;
387 }
388 break;
389 }
390 if (dumm -> next != NULL) dumm = dumm -> next;
391 }
392 }
393 tmp_cpmd -> dummies -= num_to_remove;
394 if (tmp_cpmd -> dummies > 0)
395 {
396 dumm = tmp_cpmd -> dummy;
397 for (j=0; j<tmp_cpmd -> dummies; j++)
398 {
399 dumm -> id = j;
400 if (dumm -> next != NULL) dumm = dumm -> next;
401 }
402 }
403 clean_dummy_widgets ();
404 the_dummy_box = create_dummy_box ();
405 add_box_child_start (GTK_ORIENTATION_VERTICAL, dummy_param_box, the_dummy_box, FALSE, FALSE, 0);
406 show_the_widgets (dummy_param_box);
407 }
408 else
409 {
410 str = g_strdup_printf ("You must select %d dummy atom(s) to be deleted !", num_to_remove);
411 show_warning (str, qm_assistant);
412 g_free (str);
413 }
414 break;
415 default:
416 // field_unselect_all ();
417 done = TRUE;
418 break;
419 }
420 if (done) destroy_this_dialog (dialog);
421}

430void remove_dummy (int num_to_remove)
431{
432 int i, j, k;
433 gchar * str;
434 str = g_strdup_printf ("Select the %d dummy atom(s) to be removed", num_to_remove);
435 GtkWidget * rdummy = dialogmodal (str, GTK_WINDOW(qm_assistant));
436 g_free (str);
437 gtk_dialog_add_button (GTK_DIALOG(rdummy), "Apply", GTK_RESPONSE_APPLY);
438
439 gchar * mol_title[5] = {"Id", "Type", "Atom(s)", " ", "Select"};
440 gchar * ctype[5] = {"text", "text", "text", "text", "active"};
441 GType col_type[5] = {G_TYPE_INT, G_TYPE_STRING, G_TYPE_INT, G_TYPE_STRING, G_TYPE_BOOLEAN};
442
443 GtkTreeIter dummy_level, atom_level;
444 n_dummy = 0;
445 old_dummy = allocint(tmp_cpmd -> dummies);
446 GtkTreeStore * remove_model = gtk_tree_store_newv (5, col_type);
447 GtkWidget * remove_tree = gtk_tree_view_new_with_model(GTK_TREE_MODEL(remove_model));
448 for (i=0; i<5; i++)
449 {
450 if (i < 4)
451 {
452 dummy_renderer[i] = gtk_cell_renderer_text_new ();
453 }
454 else
455 {
456 dummy_renderer[i] = gtk_cell_renderer_toggle_new ();
457 gtk_cell_renderer_toggle_set_radio (GTK_CELL_RENDERER_TOGGLE(dummy_renderer[i]), TRUE);
458 g_signal_connect (G_OBJECT(dummy_renderer[i]), "toggled", G_CALLBACK(select_dummy), & remove_model);
459 }
460 dummy_col[i] = gtk_tree_view_column_new_with_attributes (mol_title[i], dummy_renderer[i], ctype[i], i, NULL);
461 gtk_tree_view_column_set_alignment (dummy_col[i], 0.5);
462 gtk_tree_view_append_column (GTK_TREE_VIEW(remove_tree), dummy_col[i]);
463 gtk_tree_view_column_set_cell_data_func (dummy_col[i], dummy_renderer[i], dummy_set_visible, GINT_TO_POINTER(i), NULL);
464 }
465 // fill model
466 struct dummy_atom * dumm = tmp_cpmd -> dummy;
467 for (i=0; i<tmp_cpmd -> dummies; i++)
468 {
469 gtk_tree_store_append (remove_model, & dummy_level, NULL);
470 gtk_tree_store_set (remove_model, & dummy_level, 0, i+1,
471 1, g_strdup_printf ("Type %d", dumm -> type + 1),
472 2, dumm -> natoms,
473 3, NULL,
474 4, 0, -1);
475 for (j=0; j<dumm -> natoms; j++)
476 {
477 gtk_tree_store_append (remove_model, & atom_level, & dummy_level);
478 k = qm_proj -> atoms[0][dumm -> list[j]].sp;
479 str = g_strdup_printf ("%s<sub>%d</sub>", exact_name(qm_proj -> chemistry -> label[k]), dumm -> list[j]+1);
480 gtk_tree_store_set (remove_model, & atom_level, 0, -(i+1), 1, 0.0, 2, 0.0, 3, str, 4, 0, -1);
481 g_free (str);
482 }
483 if (dumm -> next != NULL) dumm = dumm -> next;
484 }
485 num_cpmd_objects = tmp_cpmd -> dummies;
486 g_object_unref (remove_model);
487
488 i = ((tmp_cpmd -> dummies+1)*40 < 500) ? (tmp_cpmd -> dummies+1)*40 : 500;
489 GtkWidget * scrollsets = create_scroll (dialog_get_content_area (rdummy), 450, i, GTK_SHADOW_ETCHED_IN);
490 add_container_child (CONTAINER_SCR, scrollsets, remove_tree);
491 run_this_gtk_dialog (rdummy, G_CALLBACK(run_remove_dummy), GINT_TO_POINTER(num_to_remove));
492}

502void add_dummy (int extra)
503{
504 int i;
505 if (tmp_cpmd -> dummy == NULL)
506 {
507 tmp_cpmd -> dummy = init_dummy (0, 0);
508 tmp_cpmd -> dummies = 1;
509 extra --;
510 }
511 struct dummy_atom * dumm = tmp_cpmd -> dummy;
512 while (dumm -> next != NULL) dumm = dumm -> next;
513 for (i=0; i<extra; i++)
514 {
515 dumm -> next = init_dummy (0, i+tmp_cpmd -> dummies);
516 dumm -> next -> prev = dumm;
517 dumm = dumm -> next;
518 }
519
520 tmp_cpmd -> dummies += extra;
521 // finaly update the combobox
522 create_selection_combo (1, tmp_cpmd -> dummies, 0, G_CALLBACK(changed_dummy_id_box));

533G_MODULE_EXPORT void add_or_remove_dummy (GtkSpinButton * res, gpointer data)
534{
535 int id = gtk_spin_button_get_value_as_int(res);
536 int k;
537 gchar * str;

539 if (id != tmp_cpmd -> dummies)
540 {
541 if (id > tmp_cpmd -> dummies)
542 {
543 // adding dummy
544 k = id - tmp_cpmd -> dummies;
545 if (k > 1)
