45void add_bonds_to_project (struct project * this_proj, int removed, int nbd, int ** new_bond_list)
46{
47 int i, j;
48 int ** tmpbondid = NULL;
49 if (nbd)
50 {
51 i = this_proj -> modelgl -> bonds[0][0];
52 tmpbondid = allocdint (i+nbd, 2);
53 for (j=0; j<i; j++)
54 {
55 tmpbondid[j][0] = this_proj -> modelgl -> bondid[0][0][j][0];
56 tmpbondid[j][1] = this_proj -> modelgl -> bondid[0][0][j][1];
57 }
58 if (this_proj -> modelgl -> allbonds[0]) g_free (this_proj -> modelgl -> bondid[0][0]);
59 this_proj -> modelgl -> bondid[0][0] = allocdint (i+nbd, 2);
60 for (j=0; j<i; j++)
61 {
62 this_proj -> modelgl -> bondid[0][0][j][0] = tmpbondid[j][0];
63 this_proj -> modelgl -> bondid[0][0][j][1] = tmpbondid[j][1];
64 }
65 for (j=0; j<nbd; j++)
66 {
67 this_proj -> modelgl -> bondid[0][0][j+i][0] = new_bond_list[j][0] + this_proj -> natomes - removed;
68 this_proj -> modelgl -> bondid[0][0][j+i][1] = new_bond_list[j][1] + this_proj -> natomes - removed;
69 }
70 this_proj -> modelgl -> bonds[0][0] += nbd;
71 this_proj -> modelgl -> allbonds[0] += nbd;
72 g_free (tmpbondid);
73 }
74}

123G_MODULE_EXPORT void set_atoms_to_insert (GtkComboBox * box, gpointer data)
124{
125 GValue val = {0, };
126 GtkTreeModel * cmodel = gtk_combo_box_get_model (box);
127 GtkTreeIter iter;
128 if (gtk_combo_box_get_active_iter (box, & iter))
129 {
130 gtk_tree_model_get_value (cmodel, & iter, 0, & val);
131 tint * dat = (tint *)data;
132 gchar * str = g_strdup_printf ("%s", (char *)g_value_get_string (& val));
133 struct project * this_proj = get_project_by_id (dat -> a);
134 prepare_to_instert (str, this_proj, this_proj -> modelgl -> search_widg[dat -> c], TRUE);
135 g_free (str);
136 }
137 gtk_combo_box_set_active (box, 0);
