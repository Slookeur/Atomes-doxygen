193G_MODULE_EXPORT void set_thermo_param (GtkEntry * res, gpointer data)
194{
195 int i;
196 i = GPOINTER_TO_INT(data);
197 const gchar * m = entry_get_text (res);
198 double v = atof(m);
199 if (i == 9)
200 {
201 if ((v == 0.0 || v >= 1.0) && v != tmp_field -> thermo_opts[i])
202 {
203 tmp_field -> thermo_opts[i] = v;
204 }
205 else if (v < 1.0 && v != 0.0)
206 {
207 show_warning ("Target temperature for the pseudo bath must be &#8805; 1.0 K,\n"
208 "Alternatively set value to 0.0 K to system temperature.", field_assistant);
209 }
210 }
211 else if (v >= 0.0 && v != tmp_field -> thermo_opts[i])
212 {
213 tmp_field -> thermo_opts[i] = v;
214 }
215 update_entry_double(res, tmp_field -> thermo_opts[i]);
216}

1877 int i;

1879 const gchar * m = entry_get_text (res);

1907 show_warning ("Padding must be > 0.0 &#xC5;", field_assistant);
