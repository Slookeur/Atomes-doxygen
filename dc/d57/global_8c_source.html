710gchar * calculation_time (gboolean modelv, double ctime)
711{
712 int i, j, k;
713 gchar * t_string = (modelv) ? "\n \tAnalysis was performed in: ": "";
714 if (ctime < 60.0)
715 {
716 return g_strdup_printf ("%s%f s", t_string, ctime);
717 }
718 else if (ctime < 3600.0)
719 {
720 i = ctime/60.0;
721 return g_strdup_printf ("%s%d m %f s", t_string, i, ctime-i*60.0);
722 }
723 else if (ctime < 86400.0)
724 {
725 // Really ?!
726 i = ctime/3600.0;
727 j = (ctime - i*3600.0)/60.0;
728 return g_strdup_printf ("%s%d h %d m %f s", t_string, i, j, ctime-i*3600.0-j*60.0);
729 }
730 else
731 {
732 // Seriously ?!
733 i = ctime/86400.0;
734 j = (ctime - i*86400.0)/3600.0;
735 k = (ctime - i*86400.0 - j*3600.0)/60.0;
736 return g_strdup_printf ("%s%d d %d h %d m %f s", t_string, i, j, k, ctime-i*86400.0-j*3600.0-k*60.0);
737 }
