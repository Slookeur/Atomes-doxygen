57void get_color (ColRGBA * but, int cid)
58{
59 int rid = cid / 4;
60 int bid = cid - rid*4;
61 but -> red = (rid < 6) ? 0.0 : (rid < 11) ? 0.2 * (rid - 5) : 1.0;
62 but -> green = (rid < 6) ? 0.2*rid : (rid < 11) ? 1.0 - 0.2 * (rid - 5) : 0.2 * (rid - 10);
63 but -> blue = bid * 0.333;
64 if (bid == 3) but -> blue = 1.0;
65}

78cairo_surface_t * col_surface (double r, double g, double b, int x, int y)
79{
80 cairo_surface_t * cst;
81 cairo_t * tcst;
82 cst = cairo_image_surface_create(CAIRO_FORMAT_ARGB32, x, y);
83 tcst = cairo_create (cst);
84 cairo_set_source_rgb (tcst, r, g, b);
85 cairo_paint (tcst);
86 cairo_destroy (tcst);
87 return cst;
88}
