73int find_bond_in_bonds (struct project * this_proj, int i, int j, int b, int id)
74{
75 int k, l, m;
76 for (k=0; k < this_proj -> modelgl -> bonds[j][b]; k++)
77 {
78 l = this_proj -> modelgl -> bondid[j][b][k][0];
79 m = this_proj -> modelgl -> bondid[j][b][k][1];
80 if (this_proj -> atoms[j][l].show[b]) i ++;
81 if (i == id) return k;
82 if (b)
83 {
84 if (this_proj -> atoms[j][l].show[b]) i ++;
85 if (i == id) return k;
86 }
87 if (this_proj -> atoms[j][m].show[b]) i ++;
88 if (i == id) return k;
89 if (b)
90 {
91 if (this_proj -> atoms[j][m].show[b]) i ++;
92 if (i == id) return k;
93 }
94 }
95 return -i;
96}

106int find_selected_bond (struct project * this_proj, int id)
107{
108 int i, j, k;
109
110 i = this_proj -> modelgl -> clones_to_be_picked - 1;
111 j = this_proj -> modelgl -> anim -> last -> img -> step;
112 k = find_bond_in_bonds (this_proj, i, j, 0, id);

114 {

122 return k;

131 return k;

143int find_selected_atom (struct project * this_proj, int id)
144{
145 int i, j, k, l, m;
146 i = -1;
147 j = this_proj -> modelgl -> anim -> last -> img -> step;
148 for (k=0; k<this_proj -> natomes; k++)
149 {
150 if (this_proj -> atoms[j][k].show[0]) i++;
151 if (i == id) return k;
152 }
153 if (this_proj -> modelgl -> anim -> last -> img -> draw_clones)
154 {
155 for (k=0; k < this_proj -> modelgl -> bonds[j][1]; k++)
156 {
157 l = this_proj -> modelgl -> bondid[step][1][k][0];
158 m = this_proj -> modelgl -> bondid[step][1][k][1];
159 if (this_proj -> atoms[j][l].show[1]) i++;
160 if (i == id) return l;
161 if (this_proj -> atoms[j][m].show[1]) i++;
162 if (i == id) return m;
163 }
164 }
165 return -1;
166}

437void update_selection_list (struct atom_selection * at_list, struct atom * at, gboolean add)
438{
439 int i;
440 struct selatom * selection = at_list -> first;
441
442 if (add)
443 {
444 selection = at_list -> last;
445 if (at_list -> selected == 0)
446 {
447 at_list -> first = new_selatom (at -> id, at -> sp);
448 at_list -> last = at_list -> first;
449 }
450 else
451 {
452 selection -> next = new_selatom (at -> id, at -> sp);
453 selection -> next -> prev = selection;
454 at_list -> last = selection -> next;
455 }
456 }
457 else
458 {
459 selection = at_list -> first;
460 for (i=0; i<at_list -> selected; i++)
461 {
462 if (selection -> id == at -> id)
463 {
464 if (i == 0 && at_list -> selected == 1)
465 {
466 at_list -> last = NULL;
467 at_list -> first = NULL;
468 }
469 else if (i == 0)
470 {
471 at_list -> first = at_list -> first -> next;
472 at_list -> first -> prev = NULL;
473 }
474 else
475 {
476 if (i == at_list -> selected-1)
477 {
478 at_list -> last = selection -> prev;
479 at_list -> last -> next = NULL;
480 }
481 else
482 {
483 selection -> next -> prev = selection -> prev;
484 selection -> prev -> next = selection -> next;
485 }
486 }
487 break;
488 }
489 if (selection -> next != NULL) selection = selection -> next;
490 }
491 }

506void process_selected_atom (struct project * this_proj, glwin * view, int id, int ac, int se, int pi)
507{
508 int i;
509 i = view -> anim -> last -> img -> step;
510 if (this_proj -> atoms[i][id].pick[pi])
511 {
512 if (! se)
513 {
514 this_proj -> atoms[i][id].pick[pi] = FALSE;
515 update_selection_list (view -> anim -> last -> img -> selected[pi], & this_proj -> atoms[i][id], FALSE);
516 view -> anim -> last -> img -> selected[pi] -> selected --;
517 view -> prepare_motion = TRUE;
518 }
519 else if (this_proj -> atoms[i][id].label[ac])
520 {
521 this_proj -> atoms[i][id].pick[pi] = FALSE;
522 this_proj -> atoms[i][id].label[ac] = FALSE;
523 view -> labelled = check_label_numbers (this_proj, 2);
524 update_selection_list (view -> anim -> last -> img -> selected[pi], & this_proj -> atoms[i][id], FALSE);
525 view -> anim -> last -> img -> selected[pi] -> selected --;
526 view -> prepare_motion = TRUE;
527 }
528 else
529 {
530 this_proj -> atoms[i][id].label[ac] = TRUE;
531 view -> labelled = check_label_numbers (this_proj, 2);
532 }
533 }
534 else
535 {
536 this_proj -> atoms[i][id].pick[pi] = TRUE;
537 update_selection_list (view -> anim -> last -> img -> selected[pi], & this_proj -> atoms[i][id], TRUE);
538 view -> anim -> last -> img -> selected[pi] -> selected ++;
539 view -> prepare_motion = TRUE;
540 }

555{

558 k = this_proj -> atoms[j][id].pick[pi];

582void process_the_hits (glwin * view, gint event_button, double ptx, double pty)
583{
584 int i, j, k, l, m, n, o, p, q;
585 view -> picked = FALSE;
586 GLubyte pixel[4];
587 GLint viewport[4];
588 int scale = gtk_widget_get_scale_factor (view -> win);
589 glGetIntegerv (GL_VIEWPORT, viewport);
590 glPixelStorei (GL_UNPACK_ALIGNMENT, 1);
591 glReadPixels (scale * view -> mouseX, viewport[3] - scale * view -> mouseY, 1, 1, GL_RGBA, GL_UNSIGNED_BYTE, pixel);
592
593 int picked_id = pixel[0] + 256*pixel[1] + 256*256*pixel[2];
594 for (i=0; i<view -> to_be_picked; i++)
595 {
596 if (picked_id == view -> color_to_pick[i])
597 {
598 j = i;
599 view -> picked = TRUE;
600 break;
601 }
602 }
603 to_pop.action = 0;
604 to_pop.x = 0.0;
605 to_pop.y = 0.0;
606 to_pop.pts[0] = to_pop.pts[1] = to_pop.pts[2] = to_pop.pts[3] = to_pop.pts[4] = -1;
607 if (view -> picked)
608 {
609 struct project * this_proj = get_project_by_id(view -> proj);
610 k = (is_atom_win_active(view) || (view -> mode == EDITION && view -> selection_mode == NSELECTION-1)) ? 1 : 0;
611 l = view -> anim -> last -> img -> step;
612 if (j < view -> clones_to_be_picked)
613 {

630 {
