192G_MODULE_EXPORT void on_calc_gr_released (GtkWidget * widg, gpointer data)
193{
194 int i;
195 if (! active_project -> initok[GR]) initgr (GR);
196 clean_curves_data (GR, 0, active_project -> numc[GR]);
197 active_project -> delta[GR] = active_project -> max[GR] / active_project -> num_delta[GR];
198 prepostcalc (widg, FALSE, GR, 0, opac);
199 clock_gettime (CLOCK_MONOTONIC, & start_time);
200 i = g_of_r_ (& active_project -> num_delta[GR], & active_project -> delta[GR], & fitc);
201 clock_gettime (CLOCK_MONOTONIC, & stop_time);
202 active_project -> calc_time[GR] = get_calc_time (start_time, stop_time);
203 prepostcalc (widg, TRUE, GR, i, 1.0);
204 if (! i)
205 {
206 remove_action (analyze_action_names[SQ]);
207 show_error ("The RDF's calculation has failed", 0, widg);
208 }
209 else
210 {
211 add_action (analyze_actions[SQ]);
212 update_rdf_view (active_project, GR);
213 gtk_widget_show (curvetoolbox);
214 }
215 fill_tool_model ();
216 for (i=0; i<4; i=i+3) update_after_calc (i);
217}
