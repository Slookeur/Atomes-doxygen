67double get_object_dim (struct insert_object * object)
68{
69 double dmax = 0.0;
70 vec3_t at, bt;
71 vec3_t dist;
72 int i, j;
73 for (i=0; i<object -> atoms-1; i++)
74 {
75 at = vec3 (object -> at_list[i].x, object -> at_list[i].y, object -> at_list[i].z);
76 for (j=i+1; j<object -> atoms; j++)
77 {
78 bt = vec3 (object -> at_list[j].x, object -> at_list[j].y, object -> at_list[j].z);
79 dist = v3_sub(at, bt);
80 dmax = max (dmax, v3_length(dist));
81 }
82 }
83 return dmax+1.0;
84}

94void correct_pos_and_get_dim (struct insert_object * object, gboolean adjust)
95{
96 int i;
97 if (object -> baryc) g_free(object -> baryc);
98 object -> baryc = allocdouble(3);
99 for (i=0; i<object -> atoms; i++)
100 {
101 object -> baryc[0] += object -> at_list[i].x;
102 object -> baryc[1] += object -> at_list[i].y;
103 object -> baryc[2] += object -> at_list[i].z;
104 }
105 for (i=0; i<3; i++) object -> baryc[i] /= object -> atoms;
106 if (adjust)
107 {
108 for (i=0; i<object -> atoms; i++)
109 {
110 object -> at_list[i].x -= object -> baryc[0];
111 object -> at_list[i].y -= object -> baryc[1];
112 object -> at_list[i].z -= object -> baryc[2];
113 }
114 }
115 object -> dim = get_object_dim (object);
116}

131gboolean rebuild_atom_neighbors (struct project * this_proj, int step, struct insert_object * object, int target, int aid, struct atom * at, gboolean * checked_at)
132{
133 int i, j;
134 struct distance dist;
135 for (i=0; i<at -> numv; i++)
136 {
137 j = at -> vois[i];
138 dist = distance_3d (& this_proj -> cell, step, at, & object -> at_list[j]);
139 if (dist.pbc && ! checked_at[j])
140 {
141 object -> at_list[j].x = at -> x - dist.x;
142 object -> at_list[j].y = at -> y - dist.y;
143 object -> at_list[j].z = at -> z - dist.z;
144 }
145 }
146 checked_at[aid] = TRUE;
147 i = 0;
148 for (j=0; j<object -> atoms; j++) if (checked_at[j]) i ++;
149 if (i == target) return TRUE;
150 for (i=0; i<at -> numv; i++)
151 {
152 j = at -> vois[i];
153 if (! checked_at[j])
154 {
155 if (rebuild_atom_neighbors (this_proj, step, object, target, j, & object -> at_list[j], checked_at)) return TRUE;
156 }
157 }
158 return FALSE;
159}

170void reconstruct_bonds (struct project * this_proj, int ifcl, int * bcid)
171{
172 int i, j, k, l;
173 int o_step = this_proj -> modelgl -> anim -> last -> img -> step;
174 int ** old_bid = NULL;
175 old_bid = allocdint (this_proj -> modelgl -> bonds[o_step][0], 2);
176 for (i=0; i<this_proj -> modelgl -> bonds[o_step][0]; i++)
177 {
178 for (j=0; j<2; j++) old_bid[i][j] = this_proj -> modelgl -> bondid[o_step][0][i][j];
179 }
180 g_free (this_proj -> modelgl -> bondid[o_step][0]);
181 this_proj -> modelgl -> bondid[o_step][0] = allocdint (this_proj -> modelgl -> bonds[o_step][0]+ ifcl, 2);
182 for (i=0; i<this_proj -> modelgl -> bonds[o_step][0]; i++)
183 {
184 for (j=0; j<2; j++) this_proj -> modelgl -> bondid[o_step][0][i][j] = old_bid[i][j];
185 }
186 if (old_bid) g_free (old_bid);
187 old_bid = NULL;
188
189 for (i=0; i<ifcl; i++)
190 {
191 j = bcid[i];
192 k = this_proj -> modelgl -> bonds[o_step][0];
193 for (l=0; l<2; l++)
194 {
195 this_proj -> modelgl -> bondid[o_step][0][k][l] = this_proj -> modelgl -> bondid[o_step][1][j][l];
196 }
197 this_proj -> modelgl -> bonds[o_step][0] ++;
198 this_proj -> modelgl -> allbonds[0] ++;
199 this_proj -> modelgl -> bondid[o_step][1][j][0] = -1;
200 }
201
202 vec3_t * old_clo = NULL;
203 i = this_proj -> modelgl -> bonds[o_step][1] - ifcl;
204 l = 0;
205 if (i)
206 {
207 old_clo = g_malloc0 (i*sizeof*old_clo);
208 old_bid = allocdint (i, 2);
209 for (i=0; i<this_proj -> modelgl -> bonds[o_step][1]; i++)
210 {
211 if (this_proj -> modelgl -> bondid[o_step][1][i][0] > -1)
212 {
213 for (k=0; k<2; k++) old_bid[l][k] = this_proj -> modelgl -> bondid[o_step][1][i][k];
214 old_clo[l].x = this_proj -> modelgl -> clones[o_step][i].x;
215 old_clo[l].y = this_proj -> modelgl -> clones[o_step][i].y;
216 old_clo[l].z = this_proj -> modelgl -> clones[o_step][i].z;
217 l ++;
218 }
219 }
220 }
221 g_free (this_proj -> modelgl -> bondid[o_step][1]);
222 this_proj -> modelgl -> bondid[o_step][1] = NULL;
223 g_free (this_proj -> modelgl -> clones[o_step]);
224 this_proj -> modelgl -> clones[o_step] = NULL;
225 this_proj -> modelgl -> bonds[o_step][1] = this_proj -> modelgl -> allbonds[1] = l;
226 if (l)
227 {
228 this_proj -> modelgl -> bondid[o_step][1] = allocdint (l, 2);
229 this_proj -> modelgl -> clones[o_step] = g_malloc0 (l*sizeof*this_proj -> modelgl -> clones[o_step]);
230 for (i=0; i<l; i++)
231 {
232 for (j=0; j<2; j++) this_proj -> modelgl -> bondid[o_step][1][i][j] = old_bid[i][j];
233 this_proj -> modelgl -> clones[o_step][i].x = old_clo[i].x;
234 this_proj -> modelgl -> clones[o_step][i].y = old_clo[i].y;
235 this_proj -> modelgl -> clones[o_step][i].z = old_clo[i].z;
236 }
237 if (old_clo) g_free (old_clo);
238 old_clo = NULL;
239 if (old_bid) g_free (old_bid);
240 old_bid = NULL;
241 }

253void reconstruct_coordinates_for_object (struct project * this_proj, struct insert_object * this_object, gboolean upcoord)
254{
255 int h, i, j, k;
256 if (this_proj -> coord -> totcoord[2])
257 {
258 int * tmp_multi = allocint (this_proj -> coord -> totcoord[2]);
259 for (i=0; i<this_object -> atoms; i++)
260 {
261 j = this_object -> at_list[i].coord[2];
262 tmp_multi[j] = 1;
263 }
264 gboolean * checked_at = allocbool (this_object -> atoms);
265 h = (this_proj -> cell.npt) ? this_proj -> modelgl -> anim -> last -> img -> step : 0;
266 for (i=0; i<this_proj -> coord -> totcoord[2]; i++)
267 {
268 if (tmp_multi[i])
269 {
270 k = 0;
271 for (j=0; j<this_object -> atoms; j++)
272 {
273 checked_at[j] = FALSE;
274 if (this_object -> at_list[j].coord[2] == i)
275 {
276 k ++;
277 }
278 }
279 for (j=0; j<this_object -> atoms; j++)
280 {
281 if (this_object -> at_list[j].coord[2] == i)
282 {
283 rebuild_atom_neighbors (this_proj, h, this_object, k, j, & this_object -> at_list[j], checked_at);
284 }
285 }
286 }
287 }
288 g_free (checked_at);
289 }
290 correct_pos_and_get_dim (this_object, TRUE);
291 if (upcoord)
292 {
293 reconstruct_bonds (this_proj, this_object -> ifcl, this_object -> bcid);
294 for (i=0; i<this_object -> atoms; i++)
295 {
296 j = this_object -> at_list[i].id;
297 this_proj -> atoms[0][j].x = this_object -> at_list[i].x + this_object -> baryc[0];
298 this_proj -> atoms[0][j].y = this_object -> at_list[i].y + this_object -> baryc[1];
299 this_proj -> atoms[0][j].z = this_object -> at_list[i].z + this_object -> baryc[2];
300 this_proj -> atoms[0][j].cloned = FALSE;
301 }
302 for (i=0; i<3; i++) this_proj -> modelgl -> saved_coord[i] = save_coordinates (this_proj, i);
303 }
304}

344tint ulam_coord (glwin * view)
345{
346 tint pos;
347 int m;
348 pos.a = pos.b = pos.c = 0;
349 if (view -> builder_win) return pos;
350 int p = view -> nth_copy;
351 if (p > 0)
352 {
353 pos.c = p/9;
354 m = p - pos.c*9;
355 pos.b = m/3;
356 pos.a = p - pos.c*9 - pos.b*3;
357 }
358 return pos;

398 }

532void clean_object_bonds (struct project * proj, int o_step, struct insert_object * object, int * new_id, gboolean movtion)
533{
534 int h, i, j, k, l, m;
535 gboolean doit;
536 int ** tmpibonds = allocdint (proj -> modelgl -> bonds[o_step][0]+proj -> modelgl -> bonds[o_step][1], 2);
537 if (proj -> modelgl -> bonds[o_step][1])
538 {
539 h = (movtion) ? 2 : 1;
540 }
541 else
542 {
543 h = 1;
544 }
545 i = 0;
546 if (new_id) object -> bcid = allocint (proj -> modelgl -> bonds[o_step][1]);
547 for (j=0; j<h; j++)
548 {
549 for (k=0; k<proj -> modelgl -> bonds[o_step][j]; k++)
550 {
551 l = proj -> modelgl -> bondid[o_step][j][k][0];
552 m = proj -> modelgl -> bondid[o_step][j][k][1];
553 doit = FALSE;
554 if (new_id)
555 {
556 if (new_id[l] && new_id[m]) doit = TRUE;
557 }
558 else
559 {
560 doit = TRUE;
561 }
562 if (doit)
563 {
564 if (new_id)
565 {
566 tmpibonds[i][0] = new_id[l] - 1;
567 tmpibonds[i][1] = new_id[m] - 1;
568 }
569 else
570 {
571 tmpibonds[i][0] = l;
572 tmpibonds[i][1] = m;
573 }
574 i ++;
575 if (j && new_id)
576 {
577 object -> bcid[object -> ifcl] = k;
578 object -> ifcl ++;
579 }
580 }
581 }
582 }
583 if (i)
584 {
585 object -> ibonds = allocdint (i, 2);
586 for (j=0; j<i; j++)
587 {
588 object -> ibonds[j][0] = tmpibonds[j][0];
589 object -> ibonds[j][1] = tmpibonds[j][1];
590 }
591 }
592 g_free (tmpibonds);
593 object -> bonds = i;
594 if (new_id) clean_object_vois (proj, object, new_id, movtion);
595}

611void add_object_atoms (struct insert_object * this_object, struct project * this_proj,
612 int o_step, int numa, int * old_id, gboolean check_bonding, atom_search * remove)
613{
614 int i, j;

616 this_object -> at_list = g_malloc0 (this_object -> atoms*sizeof*this_object -> at_list);
617 int * new_id = allocint (this_proj -> natomes);
618 for (i=0; i<this_object -> atoms; i++)
619 {
620 j = old_id[i]-1;
621 new_id[j] = i+1;
622 if (remove) remove -> todo[j] = 1;
623 this_object -> at_list[i] = * duplicate_atom (& this_proj -> atoms[o_step][j]);
624 if (i)
625 {
626 this_object -> at_list[i].prev = & this_object -> at_list[i-1];
627 this_object -> at_list[i-1].next = & this_object -> at_list[i];
628 }
629 }
630 gboolean movtion = ((object_motion && this_proj -> modelgl -> rebuild[0][0]) || (! object_motion && this_proj -> modelgl -> rebuild[1][0]));
631 if (check_bonding)
632 {
633 clean_object_bonds (this_proj, o_step, this_object, new_id, movtion);
634 }
635 correct_coordinates_for_object (this_proj, this_object, movtion);
636 if (check_bonding)
637 {
638 check_coord_modification (this_proj, NULL, & this_object -> at_list[0], this_object, FALSE, FALSE);
639 }
640 g_free (old_id);
641 g_free (new_id);
642}
