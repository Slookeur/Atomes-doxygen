150void init_labels_colors (image * img, int sp, int id)
151{
152 int i;
153 for (i = 0; i < sp; i++)
154 {
155 img -> labels_color[id][i].red = img -> at_color[i+id*sp].red;
156 img -> labels_color[id][i].green = img -> at_color[i+id*sp].green;
157 img -> labels_color[id][i].blue = img -> at_color[i+id*sp].blue;
158 img -> labels_color[id][i].alpha = 1.0;
159 GdkRGBA col = colrgba_togtkrgba (img -> labels_color[id][i]);
160 gtk_color_chooser_set_rgba (GTK_COLOR_CHOOSER(color_title[i]), & col);
161 }
162}
