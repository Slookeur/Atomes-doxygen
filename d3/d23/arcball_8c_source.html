38vec3_t get_arc_ball_vector (glwin * view, int x, int y)
39{
40 vec3_t vect;
41 vect.x = 2.0*x/view -> pixels[0] - 1.0;
42 y = view -> pixels[1] - y;
43 vect.y = 2.0*y/view -> pixels[1] - 1.0;
44 float norm_arc = vect.x*vect.x + vect.y*vect.y;
45 if (norm_arc > 1.0)
46 {
47 vect.z = 0.0;
48 }
49 else
50 {
51 vect.z = sqrt(1.0 - norm_arc);
52 }
53 return v3_norm (vect);
54}

65void arc_ball_rotation (glwin * view, int x, int y)
66{
67 arc_ball_new = get_arc_ball_vector (view, x, y);
68 vec3_t rot_axis = v3_cross (arc_ball_init, arc_ball_new);
69 if (v3_length(rot_axis) > 0.0)
70 {
71 double rot_angle = - acos(v3_dot (arc_ball_init, arc_ball_new));
72 vec4_t quat = axis_to_quat (rot_axis, rot_angle);
73 if (view -> mode != EDITION)
74 {
75 view -> anim -> last -> img -> rotation_quaternion = q4_mul (old_rotation_quaternion, quat);
76 }
77 else
78 {
79 rotate_quat (get_project_by_id(view -> proj), quat, 1, 1);
80 }
81 }
