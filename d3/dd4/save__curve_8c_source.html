42gboolean write_data_layout (FILE * fp, DataLayout * layout)
43{
44 if (fwrite (& layout -> datacolor, sizeof(ColRGBA), 1, fp) != 1) return FALSE;
45 if (fwrite (& layout -> thickness, sizeof(double), 1, fp) != 1) return FALSE;
