59field_molecule * get_active_field_molecule_from_model_id (project * this_proj, int aid)
60{
61 int i;
62 field_molecule * fmol = this_proj -> force_field[activef] -> first_molecule;
63 field_atom* fat;
64 while (fmol)
65 {
66 fat = fmol -> first_atom;
67 while (fat)
68 {
69 for (i=0; i<fat -> num; i++)
70 {
71 if (fat -> list[i] == aid) return fmol;
72 }
73 fat = fat -> next;
74 }
75 fmol = fmol -> next;
76 }
77 return NULL;
78}

286field_prop * get_active_prop_using_atoms (struct field_prop * pr, int ti, int * ids)
287{
288 field_prop * prop = NULL;
289 prop = pr;
290 gboolean done;
291 int i;
292 while (prop != NULL)
293 {
294 done = TRUE;
295 for (i=0; i<ti; i++)
296 {
297 if (prop -> aid[i] != ids[i])
298 {
299 done = FALSE;
300 break;
301 }
302 }
303 if (done) break;
304 prop = prop -> next;
305 }
306 return prop;
307}
