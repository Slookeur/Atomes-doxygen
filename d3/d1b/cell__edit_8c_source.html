133GtkWidget * create_cell_notebook (struct project * this_proj, GtkWidget * vbox)
134{
135 GtkWidget * notebook = gtk_notebook_new ();
136 gchar * str;
137 add_box_child_start (GTK_ORIENTATION_VERTICAL, vbox, notebook, TRUE, TRUE, 0);
138
139 int i, j;
140 gboolean doit;
141#ifdef DEBUG
142 for (i=0; i<5; i ++)
143#else
144 for (i=0; i<4; i ++)
145#endif
146 {
147#ifdef DEBUG
148 doit = (i == 4 && ! this_proj -> pix[0]) ? FALSE : TRUE;
149#else
150 doit = TRUE;
151#endif
152 if (doit)
153 {
154 j = (i < 2) ? i+1: i+2;
155 str = g_strdup_printf ("<b>%s</b>", edit_names[j]);
156 gtk_notebook_insert_page (GTK_NOTEBOOK(notebook), cell_tab (i, this_proj), markup_label(str, -1, -1, 0.0, 0.5), i);
157 if (i > 1) widget_set_sensitive (gtk_notebook_get_nth_page (GTK_NOTEBOOK (notebook), i), (this_proj -> steps) > 1 ? 0 : 1);
158 g_free (str);
159 }
160 }
161
162 show_the_widgets (notebook);
163 return notebook;
164}

174GtkWidget * create_cell_edition_window (struct project * this_proj, gpointer data)
175{
176 gchar * str = g_strdup_printf ("Cell edition - %s", this_proj -> name);
177 GtkWidget * win = create_win (str, this_proj -> modelgl -> win, FALSE, FALSE);
178 g_free (str);
179 GtkWidget * vbox = create_vbox (5);
180 add_container_child (CONTAINER_WIN, win, vbox);
181 gtk_widget_set_size_request (vbox, 700 , -1);
182
183 int i;
184 for (i=0; i<3; i++) this_proj -> modelgl -> cell_win -> cparam[i] = this_proj -> modelgl -> cshift[i];
185 this_proj -> modelgl -> cell_win -> notebook = create_cell_notebook (this_proj, vbox);
186
187 GtkWidget * hbox = create_hbox (5);
188 add_box_child_start (GTK_ORIENTATION_VERTICAL, vbox, hbox, TRUE, FALSE, 0);
189 GtkWidget * but = create_button ("Close", IMG_STOCK, FCLOSE, -1, -1, GTK_RELIEF_NORMAL, G_CALLBACK(close_cell_edit), this_proj);
190 add_box_child_end (hbox, but, FALSE, FALSE, 5);
191 add_gtk_close_event (win, G_CALLBACK(close_cell), this_proj);
192
193 return win;
194}
