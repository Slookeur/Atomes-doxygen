330gchar * replace_markup (char * init, char * key, char * rep)
331{
332 char * buffer = NULL;
333 char * p = NULL;
334 int inilen = strlen(init);
335 int oldlen = strlen(key);
336 int newlen = 0;
337 if (rep) newlen = strlen(rep);
338
339 // Is key in init
340 if (!(p = strstr(init, key))) return init;
341 buffer = g_malloc0((inilen+newlen-oldlen+1)*sizeof*buffer);
342
343 strncpy (buffer + strlen(buffer), init, p - init);
344 if (rep)
345 {
346 sprintf (buffer + strlen(buffer), "%s", rep);
347 //strcpy (buffer + strlen(buffer), rep);
348 }
349 sprintf (buffer + strlen(buffer), "%s", p + oldlen);
350 //strcpy (buffer + strlen(buffer), p + oldlen);
351 return g_strdup_printf ("%s", buffer);
352}

363gchar * substitute_string (gchar * init, gchar * o_motif, gchar * n_motif)
364{
365 gchar * str_a, * str_b;
366 str_a = g_strdup_printf ("%s", replace_markup (init, o_motif, n_motif));
367 str_b = g_strdup_printf ("%s", init);
368 while (g_strcmp0 (str_a, str_b))
369 {
370 str_b = g_strdup_printf ("%s", str_a);
371 str_a = g_strdup_printf ("%s", replace_markup (str_a, o_motif, n_motif));
372 }
373 g_free (str_b);
374 return str_a;
375}

403int sml_preview (const char * filetoread)
404{
405 int i, j, k;
406 xmlDoc * doc;
407 xmlTextReaderPtr reader;
408 xmlNodePtr racine, name_node, chem_node;
409 xmlNodePtr n_node, at_node, sp_node, coord_node;
410 xmlNodePtr spec_node, lab_node, lot_node, pbc_node;
411 xmlAttrPtr xspec;
412
413 if (lib_proj != NULL) close_project (lib_proj);
414 init_project (TRUE);
415 lib_proj = active_project;
416
417 reader = xmlReaderForFile(filetoread, NULL, 0);
418 doc = xmlParseFile(filetoread);
419 racine = xmlDocGetRootElement(doc);
420
421 name_node = findnode (racine -> children, "names");
422 if (name_node == NULL) return clean_xml_data (doc, reader);
423 n_node = findnode (name_node -> children, "library-name");
424 if (n_node == NULL) return clean_xml_data (doc, reader);
425 lib_proj -> name = g_strdup_printf ("%s", (gchar *)xmlNodeGetContent(n_node));
426 lib_proj -> name = check_xml_string (lib_proj -> name);
427 n_node = findnode (name_node -> children, "iupac-name");
428 if (n_node == NULL) return clean_xml_data (doc, reader);
429 o_names = 0;
430 other_name[0] = g_strdup_printf ("%s", (gchar *)xmlNodeGetContent(n_node));
431 o_names ++;
432 n_node = findnode (name_node -> children, "other-names");
433 if (n_node == NULL) return clean_xml_data (doc, reader);
434 n_node = n_node -> children;
435 for (name_node = n_node; name_node && o_names < 5; name_node = name_node->next)
436 {
437 if (name_node -> type == XML_ELEMENT_NODE)
438 {
439 other_name[o_names] = g_strdup_printf ("%s", (gchar *)xmlNodeGetContent(name_node));
440 o_names ++;
441 }
442 }
443 chem_node = findnode(racine -> children, "chemistry");
444 if (chem_node == NULL) return clean_xml_data (doc, reader);
445 at_node = findnode (chem_node -> children, "atoms");
446 if (at_node == NULL) return clean_xml_data (doc, reader);
447 lib_proj -> natomes = (int)atof((char *)xmlNodeGetContent(at_node));
448
449 sp_node = findnode (chem_node -> children, "species");
450 if (sp_node == NULL) return clean_xml_data (doc, reader);
451 spec_node = sp_node -> properties -> children;
452 if (spec_node == NULL) return clean_xml_data (doc, reader);
453 lib_proj -> nspec = (int)atof((char *)xmlNodeGetContent(spec_node));
454 if (lib_proj -> natomes < 1 || lib_proj -> nspec < 1) return clean_xml_data (doc, reader);
455 alloc_proj_data (lib_proj, 1);
456 lab_node = sp_node -> children;
457 if (lab_node == NULL) return clean_xml_data (doc, reader);
458 j = 0;
459 for (i=0; i<lib_proj -> nspec; i++)
460 {
461 lab_node = findnode (lab_node, "label");
462 if (lab_node == NULL) return clean_xml_data (doc, reader);
463 lib_proj -> chemistry -> label[i] = g_strdup_printf ("%s", (char *)xmlNodeGetContent(lab_node));
464 xspec = lab_node -> properties;
465 if (xspec == NULL) return clean_xml_data (doc, reader);
466 while (xspec)
467 {
468 lot_node = xspec -> children;
469 if (lot_node == NULL) return clean_xml_data (doc, reader);
470 if (g_strcmp0 ("num",(char *)xspec -> name) == 0)
471 {
472 lib_proj -> chemistry -> nsps[i] = (int)atof((char *)xmlNodeGetContent(lot_node));
473 lib_proj -> chemistry -> chem_prop[CHEM_Z][i] = get_z_from_periodic_table (lib_proj -> chemistry -> label[i]);
474 k = (int)lib_proj -> chemistry -> chem_prop[CHEM_Z][i];
475 lib_proj -> chemistry -> chem_prop[CHEM_M][i] = periodic_table_info[k].M;
476 lib_proj -> chemistry -> chem_prop[CHEM_R][i] = set_radius_ (& k, & j);
477 }
478 xspec = xspec -> next;
479 }
480 lab_node = lab_node -> next;
481 }
482 active_chem = lib_proj -> chemistry;
483 initcutoffs (active_chem, lib_proj -> nspec);
484 at_node = findnode (racine -> children, "coordinates");
485 if (at_node == NULL) return clean_xml_data (doc, reader);
486 coord_node = at_node -> children;
487 if (coord_node == NULL) return clean_xml_data (doc, reader);
488 for (i=0; i<lib_proj -> natomes; i++)
489 {
490 coord_node = findnode (coord_node, "atom");
491 if (coord_node == NULL) return clean_xml_data (doc, reader);
492 xspec = coord_node -> properties;
493 if (xspec == NULL) return clean_xml_data (doc, reader);
494 while (xspec)
495 {
496 lot_node = xspec -> children;
497 if (lot_node == NULL) return clean_xml_data (doc, reader);
498 if (g_strcmp0 ("x",(char *)xspec -> name) == 0)
499 {
500 lib_proj -> atoms[0][i].x = atof((char *)xmlNodeGetContent(lot_node));
501 }
502 else if (g_strcmp0 ("y",(char *)xspec -> name) == 0)
503 {
504 lib_proj -> atoms[0][i].y = atof((char *)xmlNodeGetContent(lot_node));
505 }
506 else if (g_strcmp0 ("z",(char *)xspec -> name) == 0)
507 {
508 lib_proj -> atoms[0][i].z = atof((char *)xmlNodeGetContent(lot_node));
509 }
510 else if (g_strcmp0 ("sp",(char *)xspec -> name) == 0)

514 }
515 xspec = xspec -> next;
516 }
517 coord_node = coord_node -> next;
518 }
519 pbc_node = findnode (racine -> children, "lattice");
520 if (pbc_node != NULL)
521 {
522
523 }
524 xmlFreeDoc(doc);
525 xmlFreeTextReader(reader);
526 xmlCleanupParser();
527 return 1;

538gchar * open_sml_file (const char * filetoread, int fam)
539{
540 xmlDoc * doc;
541 xmlTextReaderPtr reader;
542 const xmlChar sml[8]="scl-xml";
543 xmlChar * cdata;
544 xmlNodePtr racine, node;
545 /*
546 * build an xmlReader for that file
547 */
548 reader = xmlReaderForFile(filetoread, NULL, 0);
549 if (reader == NULL)
550 {
551 return NULL;
552 }
553 else
554 {
555 doc = xmlParseFile(filetoread);
556 if (doc == NULL) return NULL;
557 racine = xmlDocGetRootElement(doc);
558 if (g_strcmp0 ((char *)(racine -> name), (char *)sml) != 0)
559 {
560 clean_xml_data (doc, reader);
561 return NULL;
562 }
563 node = racine -> children;
564 node = findnode(node, "class");
565 cdata = xmlNodeGetContent(node);
566 if (g_strcmp0 ((gchar *)cdata, family_list[fam]) != 0)
567 {
568 clean_xml_data (doc, reader);
569 return NULL;
570 }
571 node = findnode (racine -> children, "names");
572 if (node == NULL)
573 {
574 clean_xml_data (doc, reader);
575 return NULL;
576 }
577 node = findnode (node -> children, "library-name");
578 if (node == NULL)
579 {
580 clean_xml_data (doc, reader);
581 return NULL;
582 }
583 cdata = xmlNodeGetContent(node);
584 xmlFreeDoc(doc);
585 xmlFreeTextReader(reader);
586 xmlCleanupParser();
587 return (gchar *)cdata;
588 }
589}
