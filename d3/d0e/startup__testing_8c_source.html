180gboolean is_GLExtension_Supported (const char * extension)
181{
182 int i, j;
183 i = j = 0;
184 glGetIntegerv (GL_NUM_EXTENSIONS, & i);
185 for (j=0; j<i; j++)
186 {
187 if (g_strcmp0 (extension, (const char*)glGetStringi(GL_EXTENSIONS, j)) == 0) return TRUE;
188 }
189 return FALSE;
190}

197void init_opengl ()
198{
199 glEnable (GL_DEPTH_TEST);
200 glDepthMask (GL_TRUE);
201 glDepthFunc (GL_LEQUAL);
202 glDepthRange (0.0f, 1.0f);
203 glClearDepth (1.0f);
204 glEnable (GL_NORMALIZE);
205
206 glShadeModel(GL_SMOOTH);
207 glCullFace(GL_BACK);
208 glEnable (GL_CULL_FACE); // Incompatible with polyhedra viz
209
210 glEnable (GL_COLOR_SUM_EXT);
211
212 glEnable (GL_PROGRAM_POINT_SIZE);
213 glEnable (GL_VERTEX_PROGRAM_POINT_SIZE);
214 glEnable (GL_POINT_SPRITE);
215
216 glEnable (GL_POINT_SMOOTH);
217 glHint (GL_POINT_SMOOTH_HINT, GL_NICEST);
218 glEnable (GL_LINE_SMOOTH); // Lines antialiasing
219 glHint (GL_LINE_SMOOTH_HINT, GL_NICEST);
220
221 glDisable (GL_POLYGON_SMOOTH); // To disable ploygon antialiasing
222 glEnable (GL_POLYGON_STIPPLE);
223 glEnable (GL_POLYGON_OFFSET_FILL);
224
225 glEnable (GL_BLEND);
226 glBlendFunc (GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
227
228 glPolygonOffset (1.0, 1.0);
229 glPolygonMode (GL_FRONT_AND_BACK, GL_FILL);
230
231 glHint (GL_FOG_HINT, GL_NICEST);
232 glHint (GL_PERSPECTIVE_CORRECTION_HINT, GL_NICEST); // Really Nice Perspective Calculations
233}

331void gtk_window_change_gdk_visual (GtkWidget * win)
332{
333 // GTK+ > 3.15.1 uses an X11 visual optimized for GTK+'s OpenGL stuff
334 // since revid dae447728d: https://github.com/GNOME/gtk/commit/dae447728d
335 // However, in some cases it simply cannot start an OpenGL context.
336 // This changes to the default X11 visual instead the GTK's default.
337 GdkScreen * screen = gdk_screen_get_default ();
338 GList * visuals = gdk_screen_list_visuals (screen);
339 // printf("n visuals: %u\n", g_list_length(visuals));
340 GdkX11Screen* x11_screen = GDK_X11_SCREEN (screen);
341 g_assert (x11_screen != NULL);
342 Visual * default_xvisual = DefaultVisual (GDK_SCREEN_XDISPLAY(x11_screen), GDK_SCREEN_XNUMBER(x11_screen));
343 GdkVisual * default_visual = NULL;
344 // int i = 0;
345 while (visuals != NULL)
346 {
347 GdkVisual * visual = GDK_X11_VISUAL (visuals -> data);
348 if (default_xvisual -> visualid == gdk_x11_visual_get_xvisual(GDK_X11_VISUAL (visuals -> data)) -> visualid)
349 {
350 // printf("Default visual %d\n", i);
351 default_visual = visual;
352 }
353 // i++;
354 visuals = visuals -> next;
355 }
356 gtk_widget_set_visual (win, default_visual);
357}

432int main (int argc, char * argv[])
433{
434 gboolean RUNC = TRUE;
435 switch (argc)
436 {
437 case 2:
438 if (g_strcmp0 (argv[1], "-h") == 0 || g_strcmp0 (argv[1], "--help") == 0)
439 {
440 printhelp();
441 RUNC=FALSE;
442 }
443 else if (g_strcmp0 (argv[1], "-v") == 0 || g_strcmp0 (argv[1], "--version") == 0)
444 {
445 printversion();
446 RUNC=FALSE;
447 }
448 break;
449 default:
450 break;
451 }
452#ifdef G_OS_WIN32
453#ifndef DEBUG
454 FreeConsole ();
455#endif
456#endif
457 if (RUNC)
458 {
459 GtkApplication * TestOpenGLApp;
460#if GLIB_MINOR_VERSION < 74
461 TestOpenGLApp = gtk_application_new (g_strdup_printf ("atomes.ogl-%d", (int)clock()), G_APPLICATION_FLAGS_NONE);
462#else
463 TestOpenGLApp = gtk_application_new (g_strdup_printf ("atomes.ogl-%d", (int)clock()), G_APPLICATION_DEFAULT_FLAGS);
464#endif
465 GError * error = NULL;
466 g_application_register (G_APPLICATION(TestOpenGLApp), NULL, & error);
467 g_signal_connect (G_OBJECT(TestOpenGLApp), "activate", G_CALLBACK(run_opengl_test), NULL);
468 int status = g_application_run (G_APPLICATION (TestOpenGLApp), 0, NULL);
469 g_object_unref (TestOpenGLApp);
470 switch (opengl_visual)
471 {
472 case 0:
473 // X11 default
474 return -1;
475 break;
476 case -1:
477 // Impossible
478 return -2;
479 break;
480 default:
481 // Gtk default (0) or impossible with fatal error (1)
482 return status;
483 break;
484 }
485 }
486 return 0;
